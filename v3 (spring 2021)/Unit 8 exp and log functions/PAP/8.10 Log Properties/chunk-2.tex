\section{Using Logarithm Properties to Solve Equations}

Use log \gap{properties} 
to solve equations that contain \gap{multiple} log expressions
with the \gap{same} base.

\begin{myConceptSteps}{~to solve log equations using log properties\dots}
    \myStep{collect}{Move all the logs to the \gap{same} side of the equation. }
    \myStep{log properties}{Use log properties to condense the logs.}
    \myStep{solve}{Solve the log equation.}
    \myStep{check}{Check for \gap{extraneous} solutions.}
\end{myConceptSteps}


\begin{my2Problems}{6.6in}[Use log properties to solve these equations.]
    {
        $ ln\left( x^2 + 2x \right)  -  ln(3x)  = 4$
    }
    {
        $  log_3{x} = 3 - log_3(x-6)  $
    }
\end{my2Problems}
