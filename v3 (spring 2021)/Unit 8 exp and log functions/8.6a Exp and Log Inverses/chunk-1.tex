\section{Finding Inverses}

\begin{myConceptSteps}{~To find the inverse of $f(x)$ algebraically\dots}
    \setlength{\itemsep}{0.75\onelineskip}%
    \myStep{replace}{Replace $f(x)$ by $y$.}
    \myStep{swap}{Swap $x$ and $y$ in the equation.}
    \myStep{isolate}{Get the exponential or logarithm \gap{by itself}.}
    \myStep{Solve for $y$}{%
        \begin{itemize}[nosep, left=-1em]
            \item exponential form $\rightarrow$ logarithmic form
            \item logarithmic form $\rightarrow$ exponential form
        \end{itemize}
    }
    \myStep{$f^{-1}$}{Replace $y$ by $f^{-1}(x)$.}
\end{myConceptSteps}
