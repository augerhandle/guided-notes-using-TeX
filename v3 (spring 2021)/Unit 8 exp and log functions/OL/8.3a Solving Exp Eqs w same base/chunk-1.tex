\begin{minipage}{0.55\textwidth}
    Solve exponential equations using
    \gap{identical} bases when both sides of the equation
    can be (re)written with the \gap{same base}.
\end{minipage}
\begin{minipage}{0.44\textwidth}
    \begin{myCenteredBox}[width=2.75in,]
        \centering\large
        $b^C = b^D$ \quad {\Large $\Longleftrightarrow$} \quad $C = D$
    \end{myCenteredBox}
\end{minipage}

\begin{myCenteredBox}[width=5in]
    When the \gap{base} of both sides is \gap{equal},
    you can \gap{cancel} the base.
\end{myCenteredBox}

\begin{my2Problems}[\normalsize]{3in}[Solve these exponential equations.]
    {
        $5^{2x+7} = 5^{x+10}$
    }
    {
        $e^{5x} = e^{x+12}$
    }
\end{my2Problems}
