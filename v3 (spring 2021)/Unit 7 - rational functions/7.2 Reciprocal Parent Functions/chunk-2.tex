        \item range
        \item asymptotes
    \end{itemize}

    \begin{myCenteredBox}[width=3.75in,]
        \begin{center}
            {
            \Large
            \begin{tabular}{r|l}
                {\bfseries\itshape domain}     & $x \neq 0 $ \\
                {\bfseries\itshape range}      & $y \neq 0 $ \\
                {\bfseries\itshape horizontal asymptote} & $ y = 0$  \\
                {\bfseries\itshape vertical asymptote} & $ x = 0$  \\
            \end{tabular}
            }
        \end{center}
    \end{myCenteredBox}
