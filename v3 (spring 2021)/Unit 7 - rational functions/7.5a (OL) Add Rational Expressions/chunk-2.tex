\section{Adding rational expressions}


\begin{myConceptSteps}{~to {\bfseries\itshape add} two rational expressions\dots}
    \myStep{factor}{Factor the denominators.}
    \myStep{LCD}{Find the {\bfseries\itshape least common denominator} \gap{LCD} of the two rational expressions.}
    \myStep{multiply}{%
        Multiply \gap{both} the numerator and denominator 
        of the expressions by the factors needed to make their \gap{denominators}
        equal to the \gap{LCD}.
    }
    \myStep{add}{%
        Add the \gap{numerators}.
        \begin{itemize}[nosep]
            \item[$\circ$]{\bfseries\itshape Do not} add the denominators.
            The denominator becomes the \gap{LCD}. 
        \end{itemize}
    }
    \myStep{simplify}{%
        Factor the numerator and simplify the expression.
        \begin{itemize}[nosep]
            \item[$\circ$]Remember to specify the domain of the simplified expression. 
        \end{itemize}
    }
\end{myConceptSteps}


\begin{my2Problems}{3in}[Add these rational expressions.]
    {

        \frac{x+3}{2(x-1)}
        +
