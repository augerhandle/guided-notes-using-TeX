\section{Multiplying rational expressions}

\begin{myConceptSteps}{~to {\bfseries\itshape multiply} two rational expressions\dots}
    \myStep{factor}{%
        Factor the numerators and denominators in \gap{both} expressions.
    }
    \myStep{multiply}{%
        Multiply the numerators. 
        Multiply the denominators.
        \begin{itemize}[nosep]
            \item [$\circ$] Write the factors next to each other. \gap{Do not} distribute.
            \item [$\circ$] \gap{Do not} cancel!
        \end{itemize}
    }
    \myStep{domain}{%
        Find the domain as the \gap{zeros} of all the factors 
        in the denominator.
    }
    \myStep{simplify}{%
        Cancel all common factors.
    }
    \myStep{result}{%
        The result is the simplified expression including the \gap{domain}.
    }
\end{myConceptSteps}

\begin{myCenteredBox}[width=4.25in,]
    The domain \gap{excludes} any values of $x$ that force division by zero.
\end{myCenteredBox}
