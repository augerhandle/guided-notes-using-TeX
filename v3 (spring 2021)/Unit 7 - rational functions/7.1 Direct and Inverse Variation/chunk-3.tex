\section{Distinguishing between direct and inverse variation}

\begin{myConceptSteps}{~To distinguish direct and inverse variation (from data)\dots}
    \myStep{table}{Examine the data table or create a table using the data.}
    \myStep{constant}{%
        Check if the \gap{ratio} or \gap{product} of $x$ and $y$ is constant.
        \begin{itemize}[nosep]
            \item constant ratio $\longrightarrow$ \gap{direct} variation
            \item constant product $\longrightarrow$ \gap{inverse} variation
        \end{itemize}
        }
    \myStep{equation}{%
        Find the equation for the relationship.
        \begin{itemize}[nosep]
            \item Calculate $k$ using one set of $x$ and $y$ data.
            \item Write the direct or inverse variation equation: $y=kx$ or $y=\frac{k}{x}$.
        \end{itemize}
    }
\end{myConceptSteps}
