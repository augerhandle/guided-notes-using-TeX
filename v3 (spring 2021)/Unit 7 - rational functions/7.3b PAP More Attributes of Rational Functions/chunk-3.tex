\section{Graphs of rational functions with holes}

\begin{myConceptSteps}{%
    ~to sketch the graph of a rational function {\bfseries\itshape with holes}\dots
}
\myStep{holes}{%
    Find all the \gap{holes}.
    Plot them as \gap{empty circles}.
}
\myStep{asymptotes}{%
    Sketch the horizontal and vertical asymptotes of the simplified function.
}
\myStep{the graph}{%
    Sketch the graph of the simplified function (using {\scshape Desmos} to help).
    Do \gap{not} draw through the holes.
}
\end{myConceptSteps}

\begin{myWideProblem}{5.5in}[Sketch the graph of $H(x)$.]
    {
        $
        H(x) = \frac
            {x^2 + 10x}
            {x^3 +4x^2-5x}
        $
    }
\end{myWideProblem}
