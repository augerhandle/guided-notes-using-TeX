\begin{myConceptSteps}{%
    ~to solve rational \gap{equations}\dots}
    \myStep{LCD}{%
        Find the {\bfseries\itshape least common denominator} \gap{LCD}
        of  \gap{all} the fractions.
    }
    \myStep{clear the fractions}{%
        Multiply \gap{both} sides of the equation by the \gap{LCD}.
        \begin{itemize}[nosep]
            \item[$\circ$]\gap{Distribute} the LCD to {\bfseries\itshape every} term on both sides. 
        \end{itemize}
    }
    \myStep{solve}{%
        Solve the resulting equation.
    }
    \myStep{check}{Check for \gap{extraneous} solutions by comparing them to the \gap{excluded} values in the domain.}
\end{myConceptSteps}
