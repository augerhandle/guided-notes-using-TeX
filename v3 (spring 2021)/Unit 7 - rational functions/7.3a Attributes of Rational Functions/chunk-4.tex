\section{Simplifying rational functions}

\begin{myCenteredBox}[width=4in,]
    We must exclude \gap{holes} from the domain.
    So find the domain \gap{before} you cancel common factors.
\end{myCenteredBox}

\begin{myConceptSteps}{~to simplify a rational function\dots}
    \myStep{domain}{%
        Find the \gap{domain} of the function. 
    }
    \myStep{simplify}{%
        \gap{Factor} the numerator/denominator.
        \gap{Cancel} common factors.
        }
    \myStep{rewrite}{%
        Rewrite the function as the \gap{simplified} function. Include the domain.
    }
\end{myConceptSteps}
