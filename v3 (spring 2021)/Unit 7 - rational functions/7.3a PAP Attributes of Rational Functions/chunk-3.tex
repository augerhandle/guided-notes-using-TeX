\section{Holes of rational functions}

\gap{Holes}  occur when a \gap{common factor} cancels out 
in the numerator and denominator.
We must \gap{exclude} them from the domain.

\begin{myConceptSteps}{~to find the {\bfseries\itshape holes} of a rational function\dots}
    \myStep{canceled factors}{%
        Simplify by {\bfseries\itshape factoring} and
        \gap{canceling} common factors.
        }
    \myStep{$x$ values}{%
        Find the \gap{zeros} of each canceled factor.
    }
    \myStep{$y$ values}{%
        Evaluate the \gap{simplified} function at \gap{each} zero.
    }
    \myStep{holes}{%
        The holes have $(x,y)$ \gap{coordinates} that come from the canceled factors.
    }
\end{myConceptSteps}


\begin{my2Problems}{3.25in}[Find the holes (if any) of $g(x)$.]
    {
        $g(x) = \frac{x^2 - 81}{x+3}$
    }
    {
        $g(x) = \frac{x-2}{x^2 + 6x -16}$
    }
\end{my2Problems}
