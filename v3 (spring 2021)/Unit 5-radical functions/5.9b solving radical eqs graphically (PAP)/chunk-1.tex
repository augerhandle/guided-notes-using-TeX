\section{Solving equations by graphing}

\begin{myConceptSteps}{ To solve an algebraic equation graphically (using {\normalfont\scshape Desmos} or a graphing calculator)\dots}
    \myStep{left}{%
        Let the \gap{left} side of the equation be called \gap{$L(x)$}.
        Graph $y=L(x)$.
        }
        \myStep{right}{%
        Let the \gap{right} side of the equation be called \gap{$R(x)$}.
        Graph $y=R(x)$.
        }
    \myStep{intersections}{%
        Find the point(s) where these two graphs intersect.
        }
    \myStep{$x$-coordinates}{%
        The solution(s) to the original equation are the $x$-coordiates 
        of the intersection point(s).
        }
\end{myConceptSteps}
