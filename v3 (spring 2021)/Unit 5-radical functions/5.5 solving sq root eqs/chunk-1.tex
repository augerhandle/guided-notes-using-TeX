\begin{myConceptSteps}{To solve square root equations\dots}
    \myStep{isolate}{
        Move the radical to \gap{one side} of the equation.
        \begin{itemize}[nosep]
            \item[$\circ$] If there are two radicals, put them on \gap{opposite} 
            sides of the equation.
        \end{itemize}
        }
    \myStep{square}{Square both sides of the equation.}
    \myStep{solve} Solve the equation for the variable.
    \myStep{extraneous solutions}{
        Check all your solutions.
        \begin{itemize}[nosep]
            \item[$\circ$] Anything that does not check, 
            is \gap{extraneous}. 
            {\bfseries\itshape Reject} it.
        \end{itemize}
        }
\end{myConceptSteps}
