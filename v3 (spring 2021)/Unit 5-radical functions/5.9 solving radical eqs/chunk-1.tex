\section{Solving Square Root Equations (again)}

\begin{myConceptSteps}{To solve square root equations\dots}
    \myStep{isolate}{%
        Move the radical to \gap{one side} of the equation.
        }
    \myStep{square}{%
        Square \gap{both sides} of the equation. 
        }
    \myStep{solve}{%
        Solve the equation.
        }
    \myStep{check}{%
        Look for extraneous solutions, and reject them.
        }
\end{myConceptSteps}
