\section{Solving Cube Root Equations}

\begin{myConceptSteps}{To solve cube root equations\dots}
    \myStep{isolate}{%
        Move the radical to \gap{one side} of the equation.
        }
    \myStep{square}{%
        Cube \gap{both sides} of the equation. 
        }
    \myStep{solve}{%
        Solve the equation.
        }
\end{myConceptSteps}

\begin{myCenteredBox}[width=5.75in]
    \large
    Because \gap{cubing} will never eliminate a negative sign,
    when you solve cube root equations,
    you {\bfseries\itshape do not} need to check for \gap{extraneous} solutions.
\end{myCenteredBox}
