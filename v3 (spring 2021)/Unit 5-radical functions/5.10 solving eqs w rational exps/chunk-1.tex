\begin{myCenteredBox}[width=5.25in,colback=white,]
    When evaluating expressions, 
    always put a \gap{$\pm$} in front when you have
    \begin{itemize}[nosep]
        \item[$\circ$] radicals with an \gap{even} index
        \item[$\circ$] rational exponents with an \gap{even} denominator
    \end{itemize}
\end{myCenteredBox}
