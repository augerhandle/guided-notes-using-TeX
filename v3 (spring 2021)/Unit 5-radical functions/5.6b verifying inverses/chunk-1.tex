
\begin{myConcept}{\,To verify that two functions, $f$ and $g$, are inverses\dots}
    Two functions, 
    $f$ and $g$,
    are inverses of each other if
    \vspace{1em}
    {
        \large
        \begin{itemize}[nosep]
            \item $f(g(x)) = x$
            \item $g(f(x)) = x$
        \end{itemize}
    }
    \vspace{1em}
    If both results equal $x$,
    then they are inverses.
    Otherwise not.
\end{myConcept}
