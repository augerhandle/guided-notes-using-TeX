
\section{Factoring Differences of Perfect Squares}

\begin{myCenteredBox}[width=4.75in,]
    You can \gap{factor} the {\bfseries\itshape difference of perfect squares} like this.
    \Large
    \begin{equation*}
        a^2 - b^2 = (a+b)(a-b)
    \end{equation*}
\end{myCenteredBox}
