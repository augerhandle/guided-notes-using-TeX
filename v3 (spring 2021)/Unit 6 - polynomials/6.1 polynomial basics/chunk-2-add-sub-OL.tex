\section{Adding and Subtracting Polynomials}

\begin{myConceptSteps}{To solve equations with rational exponents\dots}
    \myStep{isolate}{%
        Get the term with the rational exponent by itself.
        (the term raised to a fractional power).
        }
    \myStep{reciprocal}{%
        Raise both sides of the equation to the \gap{reciprocal} of the fractional power. 
        Make sure to \gap{simplify} the fractions.
        }
    \myStep{solve}{%
        Solve the equation.
        Include \gap{$\pm$} if necessary.
        }
    \myStep{check}{%
        If you raised both sides to an \gap{even} power, \gap{check} for extraneous roots.
    }
\end{myConceptSteps}



\begin{my2Problems}[\large]{1.5in}[
        Solve these equations with rational exponents.
        You may use a calculator to evaluate, but watch out for {\Large$\pm$}.
        ]
    {

    }
    {
        $ x^{\frac{3}{2}} = 729 $
    }
\end{my2Problems}
