\section{Adding, Subtracting and Multiplying Polynomials}


\begin{myCenteredBox}[colback=white,]
    To {\bfseries\itshape add} and {\bfseries\itshape subtract} polynomials,
    \begin{itemize}[nosep]
        \item[$\circ$] \gap{combine} {\bfseries\itshape like terms}.
    \end{itemize}
    \vspace{1em}
    To {\bfseries\itshape multiply} polynomials,
    \begin{itemize}[nosep]
        \item[$\circ$] \gap{distribute} and then 
        \gap{combine} {\bfseries\itshape like terms}.
    \end{itemize}
\end{myCenteredBox}
