\section{Adding and subtracting rational numbers}

\begin{myConceptSteps}{To add or subtract rational numbers\dots}
    \myStep{multiply}{%
        Multiply the \gap{numerator} and \gap{denominator}
        of each fraction by the denominator 
        of the \gap{other} \gap{fraction}.
        \begin{itemize}
            \item This will give you two fractions with a \gap{common} denominator.
        \end{itemize}
    }
    \myStep{combine numerators}{%
        Add (or subtract) the \gap{numerators} 
        as a \gap{single} \gap{fraction}
        over the common denominator. 
    }
    \myStep{simplify}{Simplify the fraction.}
\end{myConceptSteps}


\myProblems[Add/subtract these rational numbers.]
{
    $
    \frac{1}{3} % 10/30
    +
    \frac{1}{10} % 3/30 = 13/30 
    $
}
{
    $
    \frac{3}{2} % 18/12
    -
    \frac{2}{6} % 4/12 = (18-4)/12 = 14/12 = 7/6
    $
}
{1in}
[\myAnswer{$\frac{13}{30}$}]
[\myAnswer{$\frac{11}{10}$\hfill example with simplifying}]
