
\begin{myConceptSteps}{To solve an equation with rational expressions\dots}
    \myStep{excluded values}{%
        Find the \myEmph{excluded} \gap{values} by making sure that 
        \gap{neither} denominator is zero.
    }
    \myStep{new equation}{%
        Write a new equation by \gap{cross}--\gap{multiplying}.
        \begin{itemize}[nosep]
            \item Be careful to \gap{distribute} correctly.
            \item This equation will not have fractions.
            \item This is known at the \gap{butterfly} \gap{method}.
        \end{itemize}
    }
    \myStep{solve}{%
        Solve the new equation.
        \begin{itemize}[nosep]
            \item You will get \gap{1} or \gap{2} solutions.
            \item \gap{Reject} any excluded values.
            \item Your final answer will have \gap{2} or \gap{1} or \gap{no} solutions.
        \end{itemize}
    }
\end{myConceptSteps}
