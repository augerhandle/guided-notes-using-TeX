\section{Excluded values}

The term \gap{excluded} \gap{values} 
means values which will result in 
division by zero.
We can \gap{never} divide by zero.

\begin{myConceptSteps}
    {To find the \myEmph{excluded} \gap{values} of a rational expression\dots}
    \myStep{factor}{Completely \gap{factor} the denominator.}
    \myStep{zeros}{The \gap{excluded} \gap{values} are the
        \gap{zeros} of the denominator.
    }
\end{myConceptSteps}

\myProblems[Find the excluded values of these rational expressions.]
    {
        \large
        $\frac{2x}{(x-2)(x+8)}$
    }
    {
        \large
        $\frac{(x^2 + 7x + 13)}{x(5x-2)}$
    }
{0.5in}
[\myAnswer{$x \ne 2, -8$}]
[\myAnswer{$x \ne 0, \frac{2}{5}$}]
