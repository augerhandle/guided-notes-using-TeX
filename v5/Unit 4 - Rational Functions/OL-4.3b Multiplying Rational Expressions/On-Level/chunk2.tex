\section{Multiplying rational expressions}

\begin{myConceptSteps}{To {\bfseries\itshape multiply} two rational expressions\dots}
    \myStep{multiply}{\gap{Multiply} the numerators and denominators.
        \begin{itemize}[nosep]
            \item Write the polynomials next to each other. \gap{Do not} distribute.
            \item This will give you \gap{one} rational expression.
        \end{itemize}
    }
    \myStep{simplify}{%
        Simplify the rational expression. 
    }
\end{myConceptSteps}

\myWideProblem[Multiply these rational expressions.]
    {
        \large
        $
        \frac
        {2(x+1)(x+2)}
        {(x+3)}
        \cdot
        \frac
        {(x+3)}
        {(x+1)(x-2)}
        $
        \hfill
        \myAnswer{$\frac{2(x+2)}{(x-2)} ; x\ne -3, -1, 2$}
    }
{1.5in}
\myWideProblem[Multiply these rational expressions.]
    {
        \large
        $
        \frac
        {3x+15}
        {x-2}
        %
        \cdot
        %
        \frac
        {2x-4}
        {x^2 -2x -35}
        $
        \hfill
        \myAnswer{$  \frac{6}{x-7} ; x \ne 2, -5, 7$}
    }
{3in}
