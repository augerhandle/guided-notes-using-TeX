\section{Standard to Vertex Form}


\begin{myConceptSteps}{To convert a quadratic function from standard to vertex form\dots}
    \myStep{start with standard form}{%
        \dotfill 
            $ f(x) = 
            2x^2 + 12x + 10
            $
    }
    \myStep{factor $a$ from first two terms}{%
        \dotfill 
            $ f(x) = 
                2 \left[ x^2 + 6x \right] + 10
            $
    }
    \myStep{complete the square}{%
        \dotfill
            $ f(x) = 
            2 \left[ 
                \left(
                    x + 3
                \right)^2 
                - 
                9 
            \right] 
            + 10
            $
    }
    \myStep{distribute $a$}{%
        \dotfill 
            $ f(x) = 
            \left[ 
                2 \left(
                    x + 3
                \right)^2 
                - 
                18 
            \right] 
            + 10
            $
    }
    \myStep{combine like terms}{%
        \dotfill 
            $ f(x) = 
                2 \left(
                    x + 3
                \right)^2 
            - 8
            $
    }
\end{myConceptSteps}



\myWideProblem[Convert this function to vertex form.]
{
    $ g(x) = 3x^2 - 30x +73 $
}
{1.6in}
[\myAnswer{$ g(x) = 3(x-5)^2 - 2 $}]
