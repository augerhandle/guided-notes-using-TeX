\section{Vertex to Standard Form}


\begin{myConceptSteps}{To convert a quadratic function from vertex to standard form\dots}
    [
        \qquad\hfill 
        $
            f(x) = 2(x-3)^2 + 4
        $
    ]
    \myStep{multiply}{%
         Expand \gap{$(x-h)^2$}.
        \dotfill
        $
        \quad = 2\left[ x^2 -6x + 9 \right] + 4 
        $
        \begin{itemize}
            \item Use the \gap{box} \gap{method}.
        \end{itemize}
    }
    \myStep{distribute}{%
        Distribute \gap{$a$}.
        \dotfill
            $
                \quad = 2x^2 -12x + 18 + 4 
            $
    }
    \myStep{simplify}{%
        Combine \gap{like} \gap{terms}.
        \dotfill
            $\quad = 2x^2 - 12x + 22$%
    }%
\end{myConceptSteps}
