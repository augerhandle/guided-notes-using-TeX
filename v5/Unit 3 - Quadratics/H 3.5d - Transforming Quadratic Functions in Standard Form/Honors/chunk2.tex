\myWideProblemWithContent[Describe the transformations of this function relative to the quadratic parent function.]
{
    \centering $g(x) = 2x^2 + 12x + 10$
    \tcblower
    \whenSTUDENT{\vspace{2in}}
    \whenTEACHER{
        \small
        $g(x) = 2\left[x^2 + 6x\right] +10$ \qquad $B=6$, $\frac{B}{2}=3$, $(\frac{B}{2})^2=9$\\
        $g(x) = 2\left[(x+3)^2 - 9)\right] +10$\\
        $g(x) = 2(x+3)^2 - 18 +10$\\
        $g(x) = 2(x+3)^2 - 8$\\[2\onelineskip]
        $a = 2$ \quad $h=-3$ \quad $k=-8$\\
        transformations:
        \begin{itemize}[nosep]
            \item vertical stretch by 2 
            \item horizontal shift left by 3 
            \item vertical shift down by 8
        \end{itemize}
    }
}

\myWideProblemWithContent[Find the specified attributes for this quadratic function.]
{
    \centering $g(x) = -x^2 + 16x + 15$
    \tcblower 
    \begin{minipage}[t]{0.6\textwidth}
    vertex coordinates: \gap{$(8,4)$}\\
    equation of the AOS: \gap{$x=8$}\\
    domain: \gap{all real numbers}\\
    range: \gap{$y\le 4$}\\
    opens up or down? \gap{down}\\
    vertex is a maximum or minimum? \gap{max} \\
    left end behavior: \gap{as $x\rightarrow-\infty$, $y\rightarrow-\infty$}\\
    right end behavior: \gap{as $x\rightarrow\infty$, $y\rightarrow-\infty$}\\
    \end{minipage}
    \begin{minipage}[t]{0.39\textwidth}
        \whenTEACHER{
            \small
            $g(x) = -x^2 + 16x - 60$\\
            $g(x) = -\left[x^2 - 16x\right] - 60$\\ 
            $B=-16$, $\frac{B}{2}=-8$, $(\frac{B}{2})^2=64$\\[1\onelineskip]
            $g(x) = -\left[(x-8)^2 - 64)\right] - 60$\\
            $g(x) = -(x-8)^2 + 64 - 60$\\
            $g(x) = -(x-8)^2 + 4 $\\[2\onelineskip]
            $a = -1$ \quad $h=8$ \quad $k=4$\\
            \vspace{1in}
        }
        \whenSTUDENT{\vspace{3.5in}}
    \end{minipage}
}
