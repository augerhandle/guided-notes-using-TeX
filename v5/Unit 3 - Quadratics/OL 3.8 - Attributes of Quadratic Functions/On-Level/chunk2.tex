\section{Attributes in Vertex Form}

\begin{myConcept}
    {%
    To find the attributes of a quadratic function 
    in \myEmph{vertex form}\dots
    }
    \begin{enumerate}[fullwidth]
    \item \myEmph{\textcolor{blue}{$a$, $h$, $k$}.} {Find the values of \gap{$a$}, \gap{$h$}, and \gap{$k$}.}%
    \item \myEmph{\textcolor{blue}{attributes}.} {Use this \gap{table} to find the attributes.}%
    \end{enumerate}
    %
    \begin{center} 
        \newcolumntype{P}[1]{>{\raggedright\arraybackslash}p{#1}}
        \renewcommand{\myDropStrutLength}{-10ex}
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}[t]{P{1.5in}|P{2in}|P{2in}}
            {\itshape attributes} 
                & {\itshape when $a$ is \gap{positive}} 
                & {\itshape when $a$ is \gap{negative}} \\
            \toprule
            opens \myEmph{up}/\myEmph{down}, 
            \mbox{vertex} \myEmph{max}/\myEmph{min} 
                & \myAnswer{opens up, min\qquad---sketch parabola + vertex} \myDropStrut 
                & \myAnswer{opens down, max\qquad---sketch parabola + vertex} \myDropStrut\\
            \midrule
            \myEmph{vertex coordinates} 
                & \myAnswer{$(h,k)$\qquad---sketch parabola + vertex} \myDropStrut 
                & \myAnswer{$(h,k)$\qquad---sketch parabola + vertex} \myDropStrut \\ 
            \midrule
            \myEmph{equation of AOS} 
                & \myAnswer{$x=h$\qquad---sketch parabola + AOS} \myDropStrut 
                & \myAnswer{$x=h$\qquad---sketch parabola + AOS} \myDropStrut \\
            \midrule
            \myEmph{range} 
                & \myAnswer{$y \ge k$} 
                & \myAnswer{$y \le k$}   
                \\
            \midrule
            \myEmph{domain}
                & \myAnswer{all real numbers} 
                &  \myAnswer{all real numbers} \\
        \end{tabular}
    \end{center}
\end{myConcept}
