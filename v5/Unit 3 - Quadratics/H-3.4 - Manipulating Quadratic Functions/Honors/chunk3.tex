\section{Converting from Vertex to Standard Form}


\begin{myConceptSteps}{To convert a quadratic function from vertex to standard form\dots}
    \myStep{multiply}{%
        Expand \gap{$(x-h)^2$} by multiplying.
        \dotfill
        $
            f(x) = 2(x-3)^2 + 4
        $
    }
    \myStep{distribute}{%
        Multiply \gap{$a$} times the expanded result.
        \dotfill
            $
                f(x) = 2\left[ x^2 -6x + 9 \right] + 4 
            $
    }
    \myStep{simplify}{%
        Multiply and combine \gap{like} \gap{terms}.
        \dotfill
            $f(x) = 2x^2 - 12x + 22$%
    }%
\end{myConceptSteps}
\vspace{-2\onelineskip}
\myWideProblem[Convert this function to vertex form.]
{
    $ g(x) = 3(x-4)^2 - 5 $
}
{1in}
[\myAnswer{$ g(x) = 3x^2 - 24x + 43 $}]
