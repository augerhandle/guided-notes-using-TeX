\section{Quadratic Inequalities}

\begin{myConceptSteps}{To solve a quadratic inequality\dots}
    \myStep{standard form}{Rewrite the inequality with \gap{zero} on one side.}
    \myStep{equation}{%
        Replace the inequality with \gap{$=$}
        to make a quadratic \gap{equation}.
    }
    \myStep{critical points}{%
        Solve the quadratic quation  
        \begin{itemize}
            \item by \gap{factoring} (which might not work) or
            \item using the \gap{quadratic} \gap{formula}.
        \end{itemize}
    }
    \myStep{number line}{Draw a number line showing the \gap{critical} \gap{points}.}
    \myStep{test points}{%
        Pick an $x$ value in each interval between the critical points.
        \begin{itemize}
            \item \gap{Evaluate} the function at the values
                using \myKey{Y=} and \myKey{2ND} \myKey{TRACE} \myKey{VALUE}.
            \item Substitute the result into the inequality from step 1.
            \item If the inequality is \gap{true}, that interval is part of the solution.
            \item If the inequality is \gap{false}, that interval is \myEmph{not} part of the solution.
        \end{itemize}
    }
    \myStep{solution}{%
        \begin{itemize}
            \item Draw the solution set on the number line.
                \begin{itemize}
                    \item \gap{Solid} circles at the critical points for $\le$ or $\ge$.
                    \item \gap{Empty} circles at the critical points for $<$ or $>$.
                    \item \gap{Filled}-\gap{in} line segments for each interval that is part of the solution.
                \end{itemize}
            \item Write the solution in \gap{interval} \gap{notation}.
            \item Write the solution as \gap{inequalities}.
        \end{itemize}
    }
\end{myConceptSteps}
