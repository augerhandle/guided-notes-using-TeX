\section{How to do it}

\begin{myConceptSteps}{To rewrite $x^2 + Bx$ by \myEmph{completing the square}\dots}
    \myStep{Find $B$}{Write down the value of $B$. This is a \gap{number}.}
    \myStep{$\frac{B}{2}$}{Calculate the values of 
        \:
        \gap{$\frac{B}{2}$}
        \:
        and
        \:
        \gap{$\left(\frac{B}{2}\right)^2$}.
        These are both \gap{numbers}.
    }
    \myStep{substitute}{%
        Rewrite $x^2 + Bx$ by substituting the numbers into:
        \Large
        \begin{equation*} 
            x^2 + Bx
            =
            \left( 
                x +
                \text{\gap{ $\frac{B}{2}$ }}
            \right)^2 
            -
            \text{\gap{ $\left( \frac{B}{2} \right)^2$ }}
        \end{equation*}
    }
\end{myConceptSteps}

\begin{tcolorbox}[center,width=5in]
    {\scshape Warning!}\qquad
    If $B$ is a negative number, 
    then there will be a \gap{subtraction} inside the square.
\end{tcolorbox}
