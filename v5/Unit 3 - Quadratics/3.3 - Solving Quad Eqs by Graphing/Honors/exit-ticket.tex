\myProblemsWithContent[]
{
    Solve this quadratic equation 
    using the \myEmph{Zero Product Property}.
    %
    \begin{center}
        $
        \left(
            x-\left[  1+\myRoot{7} \right]
        \right)
        \left(
            x-\left[  1-\myRoot{7}  \right]
        \right)
        = 0
        $
    \end{center}
    \tcblower
    \vspace{2in}
    \myAnswer{$x= 1 \pm \myRoot{7}$}
}
{
    Solve this quadratic equation 
    by \myEmph{factoring}.
    %
    \begin{center}
        $  3x^2 - 2x - 10 = 2x^2 - 4  $
    \end{center}
    \tcblower
    \vspace{2in}
    \myAnswer{$x= 1 \pm \myRoot{7}$}
}
