\section{Evaluating}

\begin{myConceptSteps}{To evaluate a quadratic function in vertex form at \gap{$x=a$}\dots}
    \myStep{Substitute}{Replace $x$ by \gap{$a$}.}
    \myStep{Simplify}{Simplify the expression. The answer will be a \gap{number}.}
\end{myConceptSteps}

\myProblems[Evaluate these quadratic functions at the given value of $x$.]
{
    $f(x) = (x+1)^2 + 3$
    %
    % (4+1)^2 + 3 = 5^2 + 3 = 25 + 3 = 28
    \quad at \quad
    $x = 4$
}
{
    $f(x) = -(x - 5)^2 - 17 $
    \quad at \quad
    $x = -3$
    %
    % -(-3-5)^2 - 17  =  -(-8)^2 - 17  =  -64-17  =  -91
    }
{0.9in}
[\myAnswer{$f(4) = 28$}]
[\myAnswer{$f(-3) =-81$}]
