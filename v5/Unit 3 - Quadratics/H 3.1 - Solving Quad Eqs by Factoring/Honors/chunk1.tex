\section{The Zero Product Property}


\begin{myConceptSteps}{To solve factored quadratic equations\dots}[
    \begin{tcolorbox}[
        center,
        width=4.75in,
        coltitle=black, colbacktitle=blue!10!white,
        title={The Zero Product Property}
        ]
        If any number of factors multiplied together are \gap{zero}, 
        then at least one of the factors equals \gap{zero}.
    \end{tcolorbox}
    Use this property by following these steps.
    ]
    %
    \myStep{separate equations}{Write equations setting each factor equal to \gap{zero}.}
    \myStep{solve}{Solve the \gap{equations}.}
    \myStep{collect}{Write all the solutions \gap{together} separated by a comma.}
\end{myConceptSteps}
