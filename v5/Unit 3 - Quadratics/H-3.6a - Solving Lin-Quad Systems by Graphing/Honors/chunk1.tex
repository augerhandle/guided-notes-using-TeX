
\begin{myConceptSteps}{To solve a linear-quadratic system by graphing\dots}
    \myStep{$y=$}{Rewrite both equations so that $y$ is \gap{by} \gap{itself}.}
    \myStep{graph}{%
        Find the graph of the \gap{line} and \gap{parabola}.
        \begin{itemize}
            \item Use \myKey{Y=} on the {\scshape TI-84}.
            \item Enter both functions.
            \item \myKey{GRAPH} will show the graphs.
        \end{itemize}
    }
    \myStep{intersect}{%
        Find the points where they \gap{intersect}.
        \begin{itemize}
            \item Use \myKey{2ND} \myKey{TRACE} \myKey{Intersect} 
             to find the coordinates of the points of intersection.
        \end{itemize}
    }
    \myStep{solution}{Write the solutions as \gap{$(x,y)$} points.}
\end{myConceptSteps}
