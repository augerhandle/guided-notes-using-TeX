

\myProblemsWithContent[Solve these systems of linear-quadratic equations by graphing.]
{
    $y = x^2 + x - 6$\\
    $y = -2x  - 2$
    \tcblower 
    \begin{myTikzpictureGrid}{0.4} {10}{10}
        \whenTEACHER{
            \tkzFct[ solid, red, thick, samples=100, domain=-10:10,]
            {(x)**2 + x - 6 }
            \tkzFct[ solid, red, thick, samples=100, domain=-10:10,]
            {-2*x - 2 }
            \draw [fill,red] (-4,6) circle (3mm);
            \draw [fill,red] (1,-4) circle (3mm);
        }
    \end{myTikzpictureGrid}    
}
{
    $x-y=-5$\\
    $x^2+2y=18$
    \tcblower 
    \begin{myTikzpictureGrid}{0.4} {10}{10}
        \whenTEACHER{
            \tkzFct[ solid, red, thick, samples=100, domain=-10:10,]
            {-(0.5)*x**2 + (18/2) }
            \tkzFct[ solid, red, thick, samples=100, domain=-10:10,]
            {x+5}
            \draw [fill,red] (-4,1) circle (3mm);
            \draw [fill,red] (2,7) circle (3mm);
        }
    \end{myTikzpictureGrid}    
}
