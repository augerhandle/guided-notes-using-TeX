

\myProblemsWithContent[Rewrite these expressions by completing the square.]
{
    $x^2 + 8x$
    \tcblower 
    \whenTEACHER{
        \large
        $B = 8$  \hfill  $\frac{B}{2} = 4$  \hfill  $(\frac{B}{2})^2 = 16$
        \\ 
        \vspace{3\onelineskip}
        \\
        \begin{center}
        $
        x^2 + 8x = (x  \text{\Large $+\,4$})^2 - \text{\Large 16}
        $
        \end{center}
    }
    \whenSTUDENT{\vspace{2.5in}}
}
{
    $x^2 - 12x$
    \tcblower 
    \whenTEACHER{
        $B = -12$\hfill  $\frac{B}{2} = -6$  \hfill  $(\frac{B}{2})^2 = 36$
        \\ 
        \vspace{3\onelineskip}
        \\
        \begin{center}
        $
        x^2 + 8x = (x  \text{\Large $-6\,$})^2 - \text{\Large 36}
        $
        \end{center}
    }
}
