\gap{Completing} \gap{the} \gap{square} 
is a way to rewrite\dots
{
    \Large
    \begin{equation*}
        x^2 + Bx 
        \text{\qquad \LARGE $\Rightarrow$ \qquad}
        (x + \text{\fbox{\phantom{x}}}\,)^2 - \text{\fbox{\phantom{x}}}
    \end{equation*}
}

\begin{myConceptSteps}{To rewrite \gap{$x^2 + Bx$} by completing the square\dots}
    \myStep{Find $B$}{Write down the value of $B$. This is a \gap{number}.}
    \myStep{$\frac{B}{2}$}{Calculate the values of 
        \:
        \gap{$\frac{B}{2}$}
        \:
        and
        \:
        \gap{$\left(\frac{B}{2}\right)^2$}.
        These are both \gap{numbers}.
    }
    \myStep{substitute}{%
        Rewrite $x^2 + Bx$ by substituting the numbers into:
        \Large
        \begin{equation*} 
            x^2 + Bx
            =
            \left( 
                x +
                \text{\gap{ $\frac{B}{2}$ }}
            \right)^2 
            -
            \text{\gap{ $\left( \frac{B}{2} \right)^2$ }}
        \end{equation*}
    }
\end{myConceptSteps}
