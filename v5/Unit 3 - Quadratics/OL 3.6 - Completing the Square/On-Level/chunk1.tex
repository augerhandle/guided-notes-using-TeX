\gap{Completing} \gap{the} \gap{square} 
is a way to rewrite\dots
{
    \Large
    \begin{equation*}
        x^2 + Bx 
        \text{\qquad \LARGE $\Rightarrow$ \qquad}
        (x + \text{\fbox{\phantom{x}}}\,)^2 - \text{\fbox{\phantom{x}}}
    \end{equation*}
}

\begin{myConceptSteps}{To rewrite \gap{$x^2 + Bx$} by completing the square\dots}
    \myStep{$B$}{Write down the value of $B$. }
    \myStep{$\frac{B}{2}$}{%
        Write down \gap{$\frac{1}{2}$} of that number. 
        That's $\frac{B}{2}$.
    }
    \myStep{$(\frac{B}{2})^2$}{%
        Write down the \gap{square} of that number. 
        That's $(\frac{B}{2})^2$. 
    }
    \myStep{substitute}{%
        Substitute $\frac{B}{2}$ and $(\frac{B}{2})^2$
        into 
        \Large
        \begin{equation*} 
            \left( 
                x +
                \text{\gap{ $\frac{B}{2}$ }}
            \right)^2 
            -
            \text{\gap{ $\left( \frac{B}{2} \right)^2$ }}
        \end{equation*}
    }
\end{myConceptSteps}
