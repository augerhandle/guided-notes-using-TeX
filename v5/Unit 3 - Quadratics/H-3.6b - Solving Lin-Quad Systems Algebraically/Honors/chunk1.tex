\begin{tcolorbox}[center,width=4in,colback=white,]
    We can solve linear-quadratic equations without doing any graphing%
    ---just using \gap{algebra}.
\end{tcolorbox}

\begin{myConceptSteps}{To solve a linear-quadratic system algebraically\dots}
    % \begin{itemize}
    %     \item 
    %         Solve the \gap{linear} equation for $y$.
    %         The result will be an \gap{expression}.
    %     \item 
    %         Substitute the expression into the \gap{quadratic} equation.
    %     \item 
    %         Solve the quadratic equation for $x$.
    %         You will get \gap{two}, \gap{one}, or \gap{no} values.
    %     \item 
    %         Substitute each $x$ value into the \gap{linear} equation 
    %         and solve for the matching $y$.
    %     \item Write the answer as $(x,y)$ pairs.
    % \end{itemize}
    \myStep{Solve the \gap{linear} equation}{
        \begin{itemize}
            \item Rewrite the linear equation as $y = \dots$.
            \item This will give you an expression.
        \end{itemize}
    }
    \myStep{Substitute the expression into the \gap{quadratic} equation}{
        \begin{itemize}
            \item This will give you a quadratic equation in $x$.
        \end{itemize}
    }
    \myStep{Solve the quadratic equation}{%
        \begin{itemize}
            \item Solve by \gap{factoring}? 
            \item Solve with the \gap{quadratic} \gap{formula}?
            \item This will give you \gap{two}, \gap{one}, or \gap{no} values.
        \end{itemize}
    }
    \myStep{Find matching $y$}{
        \begin{itemize}
            \item Substitute each $x$ into the linear equation.
            \item Solve for $y$.
        \end{itemize}
    }
    \myStep{Write the answer.}{   
        \begin{itemize}
            \item Write the solution as \gap{$(x,y)$} pairs.
        \end{itemize}
    }
\end{myConceptSteps}
