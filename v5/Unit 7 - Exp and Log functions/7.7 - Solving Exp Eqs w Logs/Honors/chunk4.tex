\section{Solving Exponential Equations with Inverse Functions}

\begin{myConceptSteps}{To solve equations with inverse functions\dots}
    \myStep{isolate}{Get the exponential term \gap{by} \gap{itself}.}
    \myStep{base}{Find the base $b$ of the exponential term.}
    \myStep{$log_b(\dots) = log_b(\dots)$}{Take the logarithm base $b$ of \gap{both} \gap{sides} of the equation.}
    \myStep{solve}{Solve for the variable. Wait until the very end to use the calculator.}
\end{myConceptSteps}

\myProblems
{
    Solve this equation without a calculator.)\\[0.5\onelineskip]
    $ 2 e^{x+1} - 7 = 5 $
    \myAnswer{\hfill\tiny $x=ln(6)-1$ }
}
{
    Solve this equation using a calculator.\\[0.5\onelineskip]
    $ 2 + 10^{5x+8} = 13 $
    \myAnswer{\hfill\tiny $x \approx -1.392$}
}
