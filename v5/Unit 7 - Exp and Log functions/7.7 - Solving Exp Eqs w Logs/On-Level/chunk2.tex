If the \gap{base} of the exponential equation is not $10$ or $e$, 
to get an approximate solution, 
we'll need to use the \myEmph{Change of Base Formula}.

\myProblems[
Solve these exponential equations.
\vspace{-1\onelineskip}
\begin{itemize}[nosep]
    \item First solve them exactly.
    \item Rewrite the log in the exact answer using the Change of Base Formula.
    \item Then find the approximate solution using a calculator.
    \item Write the approximate solution with three digits after the decimal point.
\end{itemize}
]
{
    $3^{2x-1} -4 = 21$
}
{
    $2\cdot 5^{(3b-8)} + 4 = 82$
}
{5in}
[\whenTEACHER{
    \tiny
    $x=\frac{log_3{25}+1}{2} = \frac{\,\,\frac{log{25}}{log{3}}+1\,\,}{2}\approx 1.965$ 
}]

    \tiny$b=\frac{log_5{39} + 8}{3} = \frac{\,\,\frac{log(39)}{log(5)}+8\,\,}{3} \approx 3.425$ 
}]
