\begin{myConceptSteps}{To solve equations with one exponential term by converting to log form\dots}
    \myStep{isolate}{Get the exponential term \gap{by} \gap{itself}.}
    \myStep{log form}{Convert the equation to \gap{logarithmic} \gap{form}.}
    \myStep{solve}{Solve for the variable. Wait until the very end to use the calculator.}
\end{myConceptSteps}


\myProblems[Solve these exponential equations.]
{
    Solve this equation \myEmph{exactly}. Do not use a calculator to evaluate the logarithm.
    \\[0.5\onelineskip]
    $10^{x+5} + 3 = 28$
}
{
    Solve this equation \myEmph{approximately} 
    using a calculator. Round your answer to 3 digits after the decimal point.
    \\[0.5\onelineskip]
    $2 \, e^{r+6} - 3 = 6.4$
}
