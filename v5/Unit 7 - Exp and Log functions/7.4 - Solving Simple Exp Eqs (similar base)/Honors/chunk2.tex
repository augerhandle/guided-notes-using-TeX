\myProblems[Solve these exponential equations.]
{
    $2^{5x+1} = 2^{x-3}$ \myAnswer{\small\hfill $x=-1$}
}
{
    $e^{2z+5} = 1$  \myAnswer{\small\hfill $z=-\frac{5}{2}$}
}
