\begin{myConceptSteps}{To solve exponential equations with \gap{similar} bases\dots}
    \myStep{same base}{\gap{Convert} the bases to that they are the \gap{same}.}
        \begin{itemize}[nosep]
            \item Use the 2-3-4-5 cheat sheet.
        \end{itemize}
    \myStep{cancel}{Cancel the base on each side of the equation.}
    \myStep{solve}{This sets the \gap{exponents} equal to each other. Solve that equation.} 
\end{myConceptSteps}


\myProblems
{
    Convert $1$ to be a power of $10$.
}
{
    Convert $\frac{1}{27}$ to be a power of $3$.
}
{1in}
\myProblems
{
    Convert $8^{5x+4}$ to be a power of $2$.
}
{
    Convert $\left(\frac{1}{9}\right)^{x-5}$ to be a power of $3$.
}
{1in}
[\myAnswer{
    \tiny 
    $
    8^{5x+4} 
    = 
    \left(2^3\right)^{5x+4} 
    = 
    2^{3(5x+4)} 
    = 
    2^{15x+12}
    $
}]
[\myAnswer{
    \tiny 
    $
    \left(\frac{1}{9}\right)^{x-5} 
    = 
    \left(\frac{1}{3^2}\right)^{x-5} 
    = 
    \left(3^{-2}\right)^{x-5} 
    = 
    3^{-2(x-5)} 
    = 
    3^{-(2x-10)}
    $
}]
