    Exponential equations are easy to solve then they have the 
    \gap{same} \gap{base}.
    \begin{tcolorbox}[center,colback=white,width=5.5in]
        \begin{itemize}
            \item You cancel the base\dots 
            \item but \myEmph{only when} 
            the base on both sides is the \gap{same}.
        \end{itemize}
    \end{tcolorbox}

    \begin{center}
        \renewcommand{\arraystretch}{1.5}
        \begin{tabular}{c|c}
            {\itshape same base}              & {\itshape different base}         \\ 
            \midrule
            {\large$ 2^{3x+4} = 2^{6x+7} $}   & {\large $ 2^{3x+4} = 10^{6x+7} $} \\
        \end{tabular}
    \end{center}

    Sometimes you can \gap{convert} the bases to be the same. 

    \whenTEACHER{
        \tiny 
        These problems introduce what we mean by "convert". 
        This is a crucial skill the kids must master before they'll be able to 
        solve today's problems. 
        To do this, we should encourage them to use the "cheat sheet" that shows 
        positive/negative powers of 2, 3, 4, 5.
        The SECOND ONE is the hardest.
    }

    \myProblems
    {
        Convert $8^{5x+4}$ so that is has a base of $2$.
    }
    {
        Convert $\left(\frac{1}{9}\right)^{x-5}$ so that is has a base of $3$.
    }
    {1in}
    [\myAnswer{
        \tiny 
        $
        8^{5x+4} 
        = 
        \left(2^3\right)^{5x+4} 
        = 
        2^{3(5x+4)} 
        = 
        2^{15x+12}
        $
    }]
    [\myAnswer{
        \tiny 
        $
        \left(\frac{1}{9}\right)^{x-5} 
        = 
        \left(\frac{1}{3^2}\right)^{x-5} 
        = 
        \left(3^{-2}\right)^{x-5} 
        = 
        3^{-2(x-5)} 
        = 
        3^{-(2x-10)}
        $
    }]

    \whenTEACHER{
        \tiny 
        TRANSITION... Do these conversions to make the problems AS EASY AS YESTERDAY.
    }
