\myLesson{Solving Exponential Equations\\ with Similar Bases}[4]


\begin{myObjectives}
    \myObjective{solve}{exponential equations with \gap{similar} \gap{bases} on both sides}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{base}{the number ``holding up'' the \gap{exponent}: { $\bm{2}^{7x-3}$}}
    \myDefinition{same base}{both bases are \myEmph{exactly} the same: { $2^{3x} = 2^{7x-3}$}}
    \myDefinition{similar base}{both bases can be \myEmph{converted} to be the same: { $8 = 2^{7x-3} \Longrightarrow 2^3 = 2^{7x-3}$}}
\end{myVocabulary}
