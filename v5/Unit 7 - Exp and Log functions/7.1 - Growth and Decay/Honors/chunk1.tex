\hfil
{
    \begin{minipage}{0.3\textwidth}
        \small
        The general form of an {\bfseries\itshape exponential function} is \gap{$y = b^x$}, 
        where $b$ is \gap{positive} and $b\neq 1$.\par
        \vskip\onelineskip
        If \gap{$b>1$}, the function is exponential \gap{growth}.\par
        \vskip\onelineskip
        If \gap{$b<1$}, the function is exponential \gap{decay}.
    \end{minipage}
}
