\myProblems[%
The base $b$ is given.
Find $a$, $h$, $k$.
Then write the equation for a transformed exponential functions involving the following transformations.
]
{
    $b=2$ \myAnswer{\small\hfill $2^{x+5} - 2$}
    \begin{itemize}[nosep]
        \item shifted left by 5 
        \item shifted down by 2 
    \end{itemize}
}
{
    $b = 10$  \myAnswer{\small\hfill $-3 \cdot 10^{x-1} + 4$}
    \begin{itemize}[nosep]
        \item reflected across the $x$-axis 
        \item stretched by a factor of $3$
        \item shifted right by 1 
        \item shifted up by 4 
    \end{itemize}
}
{0.25in}
