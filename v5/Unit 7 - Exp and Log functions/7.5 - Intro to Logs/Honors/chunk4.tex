\section{Converting Between Exponential and Logarithmic Form}

\begin{myConceptSteps}{To convert from exponential to logarithmic form\dots}[]
    \myStep{compare}{Compare the equation to $b^x=A$}
    \myStep{values}{Find the values of 
        \begin{itemize}[nosep]
            \item the ``growth rate'' $b$ 
            \item the ``time'' $x$ 
            \item the ``final amount'' $A$
        \end{itemize}
    }
    \myStep{rewrite}{Rewrite as $log_b(A)=x$}
\end{myConceptSteps}

\myProblems[Rewrite these in logarithmic form.]
{
    $4^3 = 64$
}{
    $3^2 = 9$
}
{0.25in}
[\myAnswer{\tiny b=4, x=3, A=64 ... $log_4(64)=3$}]
[\myAnswer{\tiny b=3, x=2, A=9 ... $log_3(9)=2$}]

\begin{myConceptSteps}{To convert from logarithmic to exponential form\dots}[]
    \myStep{compare}{Compare the equation to $log_b(A)=x$}
    \myStep{values}{Find the values of 
        \begin{itemize}[nosep]
            \item the ``growth rate'' $b$ 
            \item the ``time'' $x$ 
            \item the ``final amount'' $A$
        \end{itemize}
    }
    \myStep{rewrite}{Rewrite as $b^x=A$}
\end{myConceptSteps}

\myProblems[Rewrite these in logarithmic form.]
{
    $log_2(8) = 3$
}{
    $log_6(\frac{1}{36}) = 2$
}
{0.25in}
[\myAnswer{\tiny b=2, x=3, A=8 ... $2^3=8$}]
[\myAnswer{\tiny b=6, x=2, A=1/36 ... $6^{-2}=\frac{1}{36}$}]
