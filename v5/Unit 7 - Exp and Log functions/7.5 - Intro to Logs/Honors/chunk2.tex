\section{Exponential Form}

\begin{minipage}{0.45\textwidth}
\myWideProblemWithContent[What does this exponential expression mean?]
{
    \large
    $2^5 = $ \gap{32}\whenTEACHER{\tiny 2*2*2*2*2 ... doubling 5 times}
    \vspace{3.5\onelineskip}
}
\end{minipage}
\begin{minipage}{0.55\textwidth}
    \vfill
    \begin{itemize}
        \item the \myEmph{base} {\Large$\bm{\rightsquigarrow}$} a \gap{growth} \gap{rate}
        \item the \myEmph{exponent} {\Large$\bm{\rightsquigarrow}$} \gap{time}
        \item the \myEmph{result} {\Large$\bm{\rightsquigarrow}$} the \gap{final} \gap{amount}
    \end{itemize}
\end{minipage}
\whenTEACHER{
    \tiny
    Draw a 2-col table with $x$ and $2^x$. 
}

\myProblems[So what do these mean?]
{
    $2^3 = $ \gap{$2\cdot 2\cdot 2$} 
}
{
    $3^1 = $ \gap{$3$}
}
{0.5in}
[\myAnswer{\tiny the ``final amount'' after DOUBLING (b=2) THREE times (x=3)}]
[\myAnswer{\tiny the ``final amount'' after TRIPLING (b=3) once (x=1)}]
\myProblems
{
    $2^0 = $ \gap{$0$}
}
{
    $2^{-1} = $ \gap{$\frac{1}{2}$}
}
{0.5in}
[\myAnswer{\tiny the ``final amount'' after never (x=0) DOUBLING (b=2) }]
[\myAnswer{\tiny the ``final after DOUBLING (b=2) but going ONE STEP BACK in time (x=-1) }]

This is known as \gap{exponential} \gap{form}. 
\begin{tcolorbox}[center,colback=white,width=3in]
    \myEmph{exponential form:}
    \Large
    \[ b^x = A \]
\end{tcolorbox}
