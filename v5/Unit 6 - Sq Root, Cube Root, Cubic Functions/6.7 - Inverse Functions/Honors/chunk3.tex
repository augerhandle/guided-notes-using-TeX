\begin{myConceptSteps}{To find the inverse of a function $f(x)$\dots } 
    \myStep{$y=f(x)$}{Start by writing out the equation for the function using $y=\cdots$.}
    \myStep{swap}{Switch $x$ and $y$ everywhere in the equation.}
    \myStep{solve}{Solve that equation for $y$.
        This will give you a new equation $y=\cdots$, 
        where the right side of the equation is the inverse function. 
    }
    \myStep{result}{
        Write out the inverse function using the notation, $f^{-1}(x)=\cdots$.
    }
\end{myConceptSteps}

\myProblemsWithContent[Find the inverse of these functions. Sketch the graphs of $f$ and $f^{-1}$.]
{
    $f(x) = 3x+6$   \myAnswer{\hfill $f^{-1}=\frac{1}{3}x - 2$}
    \tcblower
    \hfill
    \begin{myTikzpictureGrid}{0.4} {5}{5}

            \tkzFct[ red, solid, thick, samples=100, domain=-5:5,]{3*x+6}
            \tkzFct[ blue, dashed, ultra thick, samples=100, domain=-5:5,]{x/3 - 2}
        }
    \end{myTikzpictureGrid}
}

\begin{tcolorbox}[center,colback=white,width=6.5in]
    The graph of a function and its inverse 
    are \gap{mirror} \gap{images} 
    across the main diagonal, $y=x$.
\end{tcolorbox}

\vspace{-1\onelineskip}
\myProblemsWithContent[%
    Find the inverse of these functions. 
    Sketch the graphs of $f$ and $f^{-1}$.
    Use a TI-84 to help you make sure the two graphs are 
    mirror images across the main diagonal line.
    ]
{
    $f(x) = x^3$   \myAnswer{\hfill $f^{-1}=\myRoot[3]{x}$}
    \tcblower
    \hfill
    \begin{myTikzpictureGrid}{0.35} {5}{5}
        \whenTEACHER{
            \tkzFct[ red, solid, thick, samples=100, domain=-5:5,]{x**3}
            \tkzFct[ blue, dashed, ultra thick, samples=100, domain=0:5,]{x**(1.0/3.0)}
            \tkzFct[ blue, dashed, ultra thick, samples=100, domain=-5:0,]{-(-x)**(1.0/3.0)}
        }
    \end{myTikzpictureGrid}
    \vspace{1.5\onelineskip}
}
{
    $f(x) = (x+1)^3 - 2)$   \myAnswer{\hfill $f^{-1}=\myRoot[3]{x+2}-1$}
    \tcblower
    \hfill
    \begin{myTikzpictureGrid}{0.35} {5}{5}
        \whenTEACHER{
            \tkzFct[ red, solid, thick, samples=100, domain=-5:5,]{(x+1)**3-2)}
            \tkzFct[ blue, dashed, ultra thick, samples=100, domain=-5:-2,]{-(-x-2)**(1.0/3.0) - 1}
            \tkzFct[ blue, dashed, ultra thick, samples=100, domain=-2:5,]{(x+2)**(1.0/3.0) - 1}
        }
    \end{myTikzpictureGrid}
}
