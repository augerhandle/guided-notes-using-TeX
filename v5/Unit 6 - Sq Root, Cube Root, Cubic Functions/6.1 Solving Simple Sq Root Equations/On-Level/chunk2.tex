\section{Solving Simple Square Root Equations}

\begin{myConceptSteps}{To solve square root equations\dots}
    \myStep{isolate}{Get the square root expression \gap{by} \gap{itself}.}
    \myStep{square}{Square both sides of the equation.}
    \myStep{solve}{solve the \gap{simpler} equation}
    \myStep{check}{Reject extraneous solutions.}
\end{myConceptSteps}

\newpage
\whenTEACHER{
    \tiny 
    When working these problems, I humbly suggest that you 
    hit hard on two things they should copy in how you 
    solve the problem. 
    \vspace{-1\onelineskip}
    \begin{itemize}[nosep]
        \item emphasize that none of this works until the radical is BY ITSELF
        \item explicitly show squaring both sides 
        \item label the "check" step very clearly
    \end{itemize}
    \vspace{-1\onelineskip}
    For example:
    \begin{align*} 
        3\myRoot{2x-1} + 2 &= 17 \\ 
        3\myRoot{2x-1} &= 15 \\ 
        \myRoot{2x-1} &= 5 \\ 
        (\phantom{\myRoot{2x-1}})^2 &= (\phantom 5)^2 \quad\longleftarrow\text{I think this might be helpful!} \\
        (\myRoot{2x-1})^2 &= (5)^2 \\
        2x-1 &= 25 \\
        \text{etc...}\\
        \text{\underline{CHECK}:}\\
        \text{etc...}\\
    \end{align*}
}
