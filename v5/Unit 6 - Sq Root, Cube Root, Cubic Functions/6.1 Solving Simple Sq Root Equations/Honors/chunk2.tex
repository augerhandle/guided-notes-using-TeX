\section{Solving Square Root Equations}

\begin{myConceptSteps}{To solve square root equations\dots}
    \myStep{isolate}{Get the square root expression \gap{by} \gap{itself}.}
    \myStep{square}{%
        \gap{Square} \myEmph{both sides} of the equation: 
        $ (\quad\cdots\quad)^2 = (\quad\cdots\quad)^2 $
    }
    \myStep{solve}{solve the \gap{simpler} equation}
    \myStep{check}{Reject extraneous solutions.}
\end{myConceptSteps}

\begin{myWarningBox}[center,width=4in,]
    \centering
    {\large$\left(\quad\cdots\quad\right)^\frac{1}{2}$} 
    is the \myEmph{same} as 
    {\large$\myRoot{\quad\cdots\quad}$}
\end{myWarningBox}

\myProblems[Solve these square root equations.]

    $2\myRoot{3x-5} - 3 = 13$
}
