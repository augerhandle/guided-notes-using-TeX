\section{Shifting the Parent Function}

\begin{tcolorbox}[center,width=6.75in,]
    \centering
    When you shift the parent function, 
    start by shifting its \gap{inflection} \gap{point}.
\end{tcolorbox}

\myWideProblemWithContent[%
Use a graphing calculator to graph these functions.\vspace{-0.8\baselineskip}
\begin{itemize}[nosep]
    \item Sketch the graphs of $f$ and $g$.
    \item Find the attributes of $g$.
\end{itemize}
]
{
    $f(x) = x^3$ \hspace{1in} $g(x) = x^3 - 4$
    \tcblower
    \begin{minipage}{0.4\textwidth}
        \begin{center}
        \begin{myTikzpictureGrid}{0.3} {5}[-8]{6}
            \whenTEACHER{
                % parent function, f
                \tkzFct[ dashed, very thick, samples=1000, domain=-5:5]{x**3}
                \draw[black,thin,fill=red] (0,0) circle (0.3 cm);
                \tkzText(2.5,2.5){$f$}
                % transformed function, g
                \tkzFct[ solid, ultra thick, samples=1000, domain=-5:5,]{x**3 - 4}
                \draw[black,thin,fill=red] (0,-4) circle (0.3 cm);
                \tkzText(2.5,-3.0){$g$}
            }
        \end{myTikzpictureGrid}
        \end{center}
    \end{minipage}
    \begin{minipage}{0.6\textwidth}
        \begin{itemize}[fullwidth]
            \item inflection point: \gap{$(0,-4)$}.
            \item domain: \gap{all real numbers}.
            \item range: \gap{all real numbers}.
            \item left end behavior: as $x \rightarrow$ \gap{$-\infty$}, $y \rightarrow$ \gap{$-\infty$}
            \item right end behavior: as $x \rightarrow$ \gap{$+\infty$}, $y \rightarrow$ \gap{$+\infty$}
        \end{itemize}
    \end{minipage}
}
