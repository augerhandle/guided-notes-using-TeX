\section{Square Root and Cube Root Radicals}

The \gap{square} \gap{root} of a number $x$ 
is a number $a$ that equals $x$ when it is squared
(raised to the second power).

\begin{minipage}{0.4\textwidth}
    \begin{tabular}{lcr}
        $\myRoot{9} = 3$ & {\itshape because} & \gap{$9 = 3^2$} \\
        $\myRoot{25} = 5$ & {\itshape because} & \gap{$25 = 5^2$} \\
        $\myRoot{81} = 9$ & {\itshape because} & \gap{$81 = 9^2$} \\
    \end{tabular}
\end{minipage}
\begin{minipage}{0.59\textwidth}
    {\huge $\Rightarrow$}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{ccc}
        \myEmph{radical form} & & \myEmph{exponential form} \\ \hline
        $\myRoot{x} = a$ & {\itshape because} & \gap{$x=a^2$} \\
    \end{tabular}
    % $\myRoot{x} = a$ \quad because\quad \gap{$x=a^2$}
\end{minipage}

\vspace{1\onelineskip}
\begin{myWarningBox}[center,width=3.5in,]
    \centering
    $\myRoot{\cdots}$ 
    is an easier way of writing 
    $\myRoot[2]{\cdots}$ 
\end{myWarningBox}
\vspace{2\onelineskip}

The \gap{cube} \gap{root} of a number $x$ 
is a number $a$ that equals $x$ when it is cubed
(raised to the third power).


\begin{minipage}{0.42\textwidth}
    \begin{tabular}{lcr}
        $\myRoot[3]{8} = 2$ & {\itshape because} & \gap{$8 = 2^3$} \\
        $\myRoot[3]{64} = 4$ & {\itshape because} & \gap{$64 = 4^3$} \\
        $\myRoot[3]{216} = 6$ & {\itshape because} & \gap{$216 = 6^3$} \\
    \end{tabular}
    % \begin{alignat*}{7}
    %     \myRoot[3]{8}   &= &&~2 &&\qqtext{\itshape because} &&9   &&= &&~2^3 \\
    %     \myRoot[3]{64}  &= &&~4 &&\qqtext{\itshape because} &&64  &&= &&~4^3 \\
    %     \myRoot[3]{216} &= &&~6 &&\qqtext{\itshape because} &&216 &&= &&~6^3
    % \end{alignat*}
\end{minipage}
\begin{minipage}{0.57\textwidth}
    {\huge $\Rightarrow$}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{ccc}
        \myEmph{radical form} & & \myEmph{exponential form} \\ \hline
        $\myRoot[3]{x} = a$ & {\itshape because} & \gap{$x=a^3$} \\
    \end{tabular}
\end{minipage}

\myProblems[Rewrite these radical expressions in exponential form.]
{
    $\myRoot{x} = 7$
    \myAnswer{\hfill $x=7^2$}
}
{
    $\myRoot{2x-3} = 5$
    \myAnswer{\hfill $2x-5=5^2$}
}
