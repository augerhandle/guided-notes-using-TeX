\myProblems[Rewrite these radical equations in exponential form.]
{
    $\myRoot[3]{x} = 8$
    \myAnswer{\hfill $x=8^3$}
}
{
    $\myRoot[3]{x+1} = 2$
    \myAnswer{\hfill $x+1=2^3$}
}
{0.5in}
\myProblems
{
    $\myRoot[3]{4x+6} = 5$
    \myAnswer{\hfill $4x+5=5^3$}
}
{
    $\myRoot[3]{-2x-7} = 4$
    \myAnswer{\hfill $-2x-7=4^3$}
}
{0.5in}

% \vspace{\onelineskip}

\myProblems[In the following problems,
\begin{itemize}[nosep]
    \item Rewrite the radical expression in exponential form (as above).
    \item Then evaluate the exponential expression (e.g., $4^3 = 4\cdot4\cdot4 = 64$).
    \item This should give you an equation which \myEmph{you can solve}. So solve it for $x$.
\end{itemize}
]
{
    $\myRoot[3]{x+2} = 4$
    \myAnswer{\hfill x+2=$4^3$ \hfill x=62}
}
{
    $\myRoot[3]{x-4} = 5$
    \myAnswer{\hfill x-4=$5^3$ \hfill x=129}
}
{2in}
\myProblems 
{
    $\myRoot[3]{4x+1} = 2$
    \myAnswer{\hfill 4x+1=$2^3$ \hfill x=$\frac{7}{4}$}
}
{
    $\myRoot[3]{7-2x} = 1$
    \myAnswer{\hfill 7-2x=$1^3$ \hfill x=3}
}
{2in}
