% It's a way to enable "augmented matrices" using a new optional argument:
%
% \begin{pmatrix}[cc|c]
%     1 & 2 & 3\\
%     4 & 5 & 9
%   \end{pmatrix}
%
\makeatletter

  \hskip -\arraycolsep
  \let\@ifnextchar\new@ifnextchar
  \array{#1}}
