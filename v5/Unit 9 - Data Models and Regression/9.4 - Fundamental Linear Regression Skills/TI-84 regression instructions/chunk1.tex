\section*{How to do Linear Regression using Desmos or TI-84 Calculators}
\Large
%
\begin{center}
\begin{minipage}[t]{0.75\textwidth}
    \begin{myConcept}{To find the line of best fit using a {\scshape TI-84}\dots}
        \begin{enumerate}%[nosep]
            \item Reset the calculator using 
                \fbox{\ttfamily 2ND} \fbox{\ttfamily +} {\footnotesize\ttfamily (MEM)}
                \fbox{\ttfamily 7} \fbox{\ttfamily 1} \fbox{\ttfamily 2}
            \item Select \fbox{\ttfamily CLEAR}
            \item Enable display of $r$.
                \begin{itemize}
                    \item Select 
                        \fbox{\ttfamily 2ND}
                        \fbox{\ttfamily 0}\,\,{\footnotesize\ttfamily (CATALOG)}
                    \item Select 
                        \fbox{$x^{-1}$}\,\,{\footnotesize (jump down the the Ds)}
                    \item Click the Down-Arrow until you see \fbox{\ttfamily DiagnosticOn}.
                    \item Select 
                        \fbox{\ttfamily DiagnosticOn}
                        \fbox{\ttfamily ENTER} 
                        \fbox{\ttfamily ENTER}
                    \item Select \fbox{\ttfamily CLEAR}
                \end{itemize}
            \item Enable scatterplots.
                \begin{itemize}
                    \item Select 
                        \fbox{\ttfamily 2ND}
                        \raisebox{0.2em}{\fbox{\ttfamily y=}}\,\,{\footnotesize\ttfamily (STAT PLOT)}
                    \item Select 
                        \fbox{\ttfamily Plot1}
                        \fbox{\ttfamily On}
                    \item Make sure that \fbox{\ttfamily On} is selected and blinking.
                    \item Make sure that {\ttfamily Type} is the scatterplot (the first one).
                    \item Make sure that {\ttfamily XList} is $\text{\ttfamily L}_1$.
                    \item Make sure that {\ttfamily YList} is $\text{\ttfamily L}_2$.
                    \item Select \fbox{\ttfamily CLEAR}
                \end{itemize}
            \item Enter your $x$ and $y$ data into a table.
                \begin{itemize}
                    \item Select 
                        \fbox{\ttfamily stat 1} {\footnotesize\ttfamily (EDIT)}
                    \item Put $x, y$ data into $L_1, L_2$ columns.
                    \item Use cursor keys to switch columns.
                    \item To clear columns, move to the header and select 
                        \fbox{\ttfamily CLEAR} 
                        \fbox{\ttfamily ENTER}
                \end{itemize}
            \item Set up the plot window.
                \begin{itemize}
                    \item Select 
                        \fbox{\ttfamily ZOOM}
                        \fbox{\ttfamily 9}\,\,{\footnotesize\ttfamily (ZoomStat)}
                    \item You should see a scatterplot of all your data.
                    \item Select \fbox{\ttfamily CLEAR}
                \end{itemize}
            % \item Double check your data by looking at the scatterplot.
            %     \begin{itemize}
            %         \item Select 
            %             \fbox{\ttfamily GRAPH} 
            %         \item If you want to inspect the coordinates of each point, 
            %             select 
            %             \fbox{\ttfamily TRACE} and use the cursor to move from point to point.
            %     \end{itemize}
            \item Perform linear regression.
                \begin{itemize}
                    \item Select 
                    \fbox{\ttfamily STAT}
                    \fbox{\ttfamily >}
                    \fbox{\ttfamily CALC}
                        \fbox{\ttfamily 4}\,\,{(\small\ttfamily LinReg(ax+b))}
                    \item Move the cursor down to  
                        \fbox{\ttfamily Calculate} 
                        and select \fbox{\ttfamily ENTER}
                \end{itemize}
        \end{enumerate}
    \end{myConcept}
    \end{minipage}
\end{center}
