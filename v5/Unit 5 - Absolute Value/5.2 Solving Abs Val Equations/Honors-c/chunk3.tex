\section{When the absolute value is not by itself}


\begin{myConceptSteps}{To solve general absolute value inequalities\dots}
    \myStep{isolate}{%
        Get the absolute value \gap{by} \gap{itself}.
        \begin{myWarningBox}[width=5.5in,center]
            \small\centering
            \gap{Multiplying}/\gap{dividing} by negatives \gap{flips} the inequality.
        \end{myWarningBox}
    }
    \myStep{solve}{Solve the inequality based on what we did above.}
\end{myConceptSteps}

\myProblems[Solve these absolute value inequalities.]
{
    $ 5|x+4| + 6 \ge 11$
    \myAnswer{
        \hfill
        {\tiny 
            $x \le -5$ OR $x \ge -3$
        }
    }
}
{
    $ 8 - |3x-7| \ge 6$
    \myAnswer{
        \hfill
        {\tiny 
            $\frac{5}{3} \le x \le 3$
        }
    }
}
