\section{Distance on the Number Line}

\begin{tcolorbox}
    \begin{itemize}[nosep]
        \item The \gap{distance} of a point from the origin is \gap{how} \gap{far} it is from $\bm{0}$.
        \item Distance is \gap{always} \gap{positive}.
    \end{itemize}
\end{tcolorbox}

\myProblemsWithContent[Find the {\bfseries\itshape distance} of the black point from the origin.]
{
    {
    \small
    \vspace{0.75\baselineskip}
    \begin{center}
        \begin{tikzpicture}[scale=0.75]
            \myDrawNumberlineCircle{0}{white}
            \myDrawNumberlineCircle{4}{black}
            \myDrawNumberline{5}
        \end{tikzpicture}
    \end{center}
    }
    {\itshape distance from origin} = \gap{4}
}
{
    {
    \small
    \vspace{0.75\baselineskip}
    \begin{center}
        \begin{tikzpicture}[scale=0.75]
            \myDrawNumberlineCircle{0}{white}
            \myDrawNumberlineCircle{-3}{black}
            \myDrawNumberline{5}
        \end{tikzpicture}
    \end{center}
    }
    {\itshape distance from origin} = \gap{3}
}
