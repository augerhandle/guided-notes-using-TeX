\section{Evaluating Absolute Value Functions}

\begin{tcolorbox}[center,width=6.5in,colback=white,]
    To evaluate a \gap{function} with absolute values, 
    you just \gap{evaluate} the expression for the function 
    at the input value.
\end{tcolorbox}

\whenTEACHER{
    \tiny
    Notice that the following problems have the EXACT SAME EXPRESSIONS 
    as the four previous problems. YOU DECIDE if you will got through 
    evaluating those 4 expressions a second time. Hasan and Hammit plan to.
}

\myProblems[Evaluate these functions at the given input value.]
{ 
    $
        f(x) =
        |2x|
    $
    \qquad
    Find $f(6)$.
}
{
    $
        g(x) =
        3|-5x|
    $
    \qquad
    Find $g(30)$.
}
{10\onelineskip}

\myProblems
{
    $
        t(z) =
        |z - 8|
    $
    \qquad
    Find $t(2)$.
}
{
    $
        s(w) =
        1 - 2\left|w + 3\right|
    $
    \qquad
    Find $s(-5)$.
}
{10\onelineskip}
