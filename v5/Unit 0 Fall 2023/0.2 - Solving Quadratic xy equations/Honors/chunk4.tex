
\begin{tcolorbox}[center,width=5in]
    \setlength{\parskip}{1em}
    When the discriminant is \myEmph{not a perfect square},
    there will be \gap{square} \gap{roots} in the solutions.

    Factor and simplify the expressions 
    \myEmph{as much as possible}.
\end{tcolorbox}

\myProblems
{
    $  v^2 - 4v - 16 = 0  $
}
{
    $  2x^2 + 4x - 11 = 0  $
}
{3.75in}
[\myAnswer{$v=2+\myRoot{5}, 2-\myRoot{5}$}]
[\myAnswer{$x=-1 \pm \frac{\myRoot{26}}{2}$}]
