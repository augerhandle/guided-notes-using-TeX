
\begin{myConceptSteps}{To find a quadratic regression equation using a \myTi\dots}
    \small
    \myStep{clear graphs}{
        \begin{itemize}
            \item Clear any functions that are in \myKey{Y=}.
        \end{itemize}
    }
    \myStep{$xy$ table}{
        \begin{itemize}
            \item Press \myKey{STAT} \myKey{EDIT} \myKey{ENTER}.
            \item Clear the columns by clicking on the \myEmph{top} of each column and pressing \myKey{CLEAR} \myKey{ENTER}.
            \item Put $x$'s into {\ttfamily L1}. Put $y$'s into {\ttfamily L2}.
        \end{itemize}
    }
    \myStep{plot settings}{
        \begin{itemize}
            \item Press \myKey{2ND} \myKey{Y=}.
            \item Press \myKey{Plot1} \myKey{ENTER}.
            \item Turn the plot \myKey{On}.
            \item Select the scatterplot. (Click the dots.)
            \item Make sure that {\ttfamily Xlist} is {\ttfamily L1} and {\ttfamily Ylist} is {\ttfamily L2}.
            \item Press \myKey{2ND} \myKey{QUIT}.
        \end{itemize}
    }
    \myStep{plot display}{%
        Make sure all the points get plotted on the display.
        \begin{itemize}
            \item Press \myKey{ZOOM} \myKey{9:ZoomStat}.
            \item You should see a plot of all your points. 
        \end{itemize}
    }
    \myStep{quadratic regression}{%
        Find the quadratic regression equation.
        \begin{itemize}
            \item Press \myKey{STAT} \myKey{CALC} \myKey{5:QuadReg}.
            \item Make sure that {\ttfamily Xlist} is {\ttfamily L1} and {\ttfamily Ylist} is {\ttfamily L2}.
            \item Press \myKey{Calculate} \myKey{Enter}.
            \item The regression equation is $y = ax^2 + bx + c$ where $a, b, c$ are displayed on the calculator.
        \end{itemize}
    }
\end{myConceptSteps}
