
\begin{myConceptSteps}{To solve a linear equation with {two} variables\dots}
    \myStep{simplify}{\gap{Combine} like terms.}
    \myStep{isolate}{Get $y$ by \gap{itself} using inverse operations.}
    \myStep{simplify}{\gap{Combine} like terms again to get \gap{$y=\dots$}.}
\end{myConceptSteps}

\myProblems[Solve these equations for $y$.]
{
    \centering
    $y - 2x -3 = 0$
    % y -2x = 3
    % y = 2x + 3 
    \myAnswer{\hfill\tiny $y = 2x + 3$ }
}
{
    \centering
    $ 2x - 6y + x = -12 $ 
    % 3x - 6y = -12 
    % 3x = 6y - 12 
    % 3x + 12 = 6y
    % 1/2 x + 2 = y
    \myAnswer{\hfill\tiny $y = \frac{1}{2} x + 2$ }
}
