
\begin{myConceptSteps}{To factor a quadratic form trinomial\dots}
    \myStep{Check}{
        \begin{itemize}
            \item Make sure the trinomial is in \gap{descending} \gap{order}.
            \item Make sure $m = $\gap{$2n$}.
        \end{itemize}
    }
    \myStep{$u^2$ and $u$}{%
        Write down 
        \begin{itemize}
            \item $u^2 =$ \gap{$x^m$} ($m$ is the \gap{larger} exponent)
            \item $u =$   \gap{$x^n$} ($n$ is \gap{one} \gap{half} of $m$)
        \end{itemize}
    }
    \myStep{$u$-substitution}{%
        Replace 
        \begin{itemize}
            \item \gap{$x^m$} by $u^2$ (in the \gap{leading} term) 
            \item \gap{$x^n$} by $u$ (in the \gap{middle} term)
        \end{itemize}
    }
    \myStep{solve}{%
        Solve \gap{ $au^2 + bu + c$} using Lesson 2.7.
    }
\end{myConceptSteps}
