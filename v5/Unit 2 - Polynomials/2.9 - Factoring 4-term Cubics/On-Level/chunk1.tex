\begin{myConceptSteps}{To factor a 4-term cubic polynomial\dots}
    \myStep{Check}{
        \begin{itemize}
            \item Make sure the trinomial is in \gap{descending} \gap{order}.
            \item Make sure it has the form \gap{$ax^3+bx^2+cx + d$}.
        \end{itemize}
    }
    \myStep{constants}{Write down $a$, $b$, $c$, $d$.}
    \myStep{inside the box}{Fill in the blanks.\\[0.25\onelineskip]
    \begin{minipage}{0.5\textwidth}
        \begin{itemize}[nosep]
            \item put \gap{$a$} in the {\bfseries\itshape top-left} blank.
            \item put \gap{$b$} in the {\bfseries\itshape top-right} blank.
        \end{itemize}
    \end{minipage}
    \begin{minipage}{0.49\textwidth}
        \begin{itemize}[nosep]
            \item put \gap{$c$} in the {\bfseries\itshape bottom left} blank.
            \item put \gap{$d$} in the {\bfseries\itshape bottom-right} blank.
        \end{itemize}
    \end{minipage}
}
    \myStep{outside the box}{
        \begin{itemize}[nosep]
            \item Find the GCF of the \gap{top} \gap{row}. Write it outside on the left.
            \item Fill in the rest of the outside by thinking about 
                a \gap{times} \gap{table}.
        \end{itemize}
    }
    \myStep{result}{%
        The factors are \gap{left} outside column and 
        the \gap{top} outside row.
    }
\end{myConceptSteps}
