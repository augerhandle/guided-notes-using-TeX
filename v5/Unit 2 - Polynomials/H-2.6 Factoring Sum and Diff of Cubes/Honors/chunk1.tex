\section{Factoring Sums of Cubes}

\myWideProblem[Multiply these polynomials using the {\bfseries\itshape Box Method}]
    {
        \centering $ (A+B) (A^2-AB+B^2) $
    }
    {1.5in}

\vfill 

\begin{tcolorbox}[center,width=5.25in]
    Polynomials that are \gap{sums} or \gap{differences} 
    \gap{of} \gap{cubes},
    may be rewritten like this:
    {
    \Large
    \begin{align*}{}
        (A^3 + B^3) &= (A+B)~(A^2-2AB + B^2) \\[0.5\onelineskip]
        (A^3 - B^3) &= (A-B)~(A^2+2AB + B^2)
    \end{align*}
    }
    where $A$ is the \gap{cube} \gap{root} of the first term 
    and $B$ is the \gap{cube} \gap{root} of the second term.
\end{tcolorbox}
