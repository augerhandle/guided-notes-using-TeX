\begin{myConceptSteps}{To factor a difference of two cubes\dots}
    \myStep{Find $A$}{
        Look at the \gap{first} {term}. $A$ is the \myEmph{product} of:
        \begin{itemize}
            \item the \gap{cube} \gap{root} of the coefficient
            \item the variable with an exponent {\large $=$} the exponent in the first term divided by \gap{3}.
        \end{itemize}
        \begin{center} 
            \begin{tabular}{cc}
                {\itshape first term} & {$A$} \\
                \midrule 
                8 & 2 \\ 
                $x^3$ & $x$ \\
                $27x^3$ & $3x$ \\
                $216x^6$ & $6x^2$ \\
            \end{tabular}
        \end{center}
    }
    \myStep{Find $B$}{Do the same thing but for the \gap{second} term.}
    \myStep{fill in}{Substitute \gap{$A$} and \gap{$B$} into either
    \begin{align*}
    &(A+B)~(A^2-2AB + B^2) \\[0.5\onelineskip]
    &(A-B)~(A^2+2AB + B^2)
    \end{align*}
    }
\end{myConceptSteps}
