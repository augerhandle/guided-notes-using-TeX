\section{Factoring Differences of Squares}

\begin{tcolorbox}[center,width=5.75in]
    When a polynomial is a 
    \gap{difference} \gap{of} \gap{squares},
    it may be rewritten like this:
    {
    \Large
    \begin{align*}{}
        (A^2 - B^2) = (A+B)(A-B)
    \end{align*}
    }
    where $A$ is the \gap{square} \gap{root} of the first term 
    and $B$ is the \gap{square} \gap{root} of the second term.
\end{tcolorbox}

\begin{minipage}{0.8\textwidth}
\begin{myConceptSteps}{To factor a difference of two squares\dots}
    \myStep{Find $A$}{
        Look at the \myEmph{first} term.
        $A$ is the \gap{product} of:
        \begin{itemize}
            \item the \gap{square} \gap{root} of the coefficient
            \item the variable with \gap{one} \gap{half} the exponent in the first term.
        \end{itemize}
    }
    \myStep{Find $B$}{
        Look at the \myEmph{second} term.
        $B$ is the \gap{product} of:
        \begin{itemize}
            \item the \gap{square} \gap{root} of the coefficient
            \item the variable with \gap{one} \gap{half} the exponent in the second term.
        \end{itemize}
    }
    \myStep{fill in}{Substitute \gap{$A$} and \gap{$B$} into $(A+B)(A-B)$.}
\end{myConceptSteps}
\end{minipage}
