\section{Word Problems}

\begin{myConceptSteps}{To write a system of equations from a word problem\dots}
    \myStep{read it}[!]{Read the word problem closely.}
    \myStep{variables}{Define the variables.%
      \begin{itemize}
        \item Name them (usually \gap{$x$}, \gap{$y$}, \gap{$z$}).
        \item Write down what each one represents.
      \end{itemize}
    }
    \myStep{equations}{
      \begin{itemize}
        \item Translate the word problem into \gap{equations} involving your variables.
        \item Write them in \gap{standard} \gap{form}.
      \end{itemize}
    }
    \myStep{matrix}{Write the \gap{augmented} \gap{matrix} for the equations.}
    \myStep{RREF}{Use a matrix calculator (TI-84) to find the \gap{reduced} \gap{row-echelon} \gap{form}.}
    \myStep{solution}{Read the solutions for the variables from the \gap{last} \gap{column} of the RREF.}
    \end{myConceptSteps}
