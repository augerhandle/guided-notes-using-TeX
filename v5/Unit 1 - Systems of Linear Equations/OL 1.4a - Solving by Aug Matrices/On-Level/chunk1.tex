\section{Augmented Matrices}

An \gap{augmented} \gap{matrix} is a simpler way to write systems of equations.
It has variable \gap{coefficients} in the first several columns 
and the \gap{constant} terms in the last column.

\hfil
{
  \sysdelim..\systeme{
    2x+3y+4z=1,
    5x+6y+7z=2,
    8x+9y+10z=3
}
{
  \LARGE
  $\Longleftrightarrow$
}
{
    \large
  \(
    \begin{bmatrix}[ccc|c]
      2 & 3 & 4 & 1\\
      5 & 6 & 7 & 2\\
      8 & 9 & 10 & 3
    \end{bmatrix}
  \)
}
}
{\scshape Dimensions:} \gap{$3\times4$}
\hfil 

\begin{tcolorbox}[center,width=6.5in]
    \large
    Before you can find the augmented matrix,
    write the equations in \gap{standard} \gap{form}.
    \begin{itemize}[nosep]
      \item $x$'s first, $y$'s second, $z$'s third 
      \item constant numbers on the right
    \end{itemize}
\end{tcolorbox}

\myProblemsWithContent[Write the augmented matrix for these systems of equations and state its dimensions.]

    2x-9y-z=41,
    x+2y+3z=2,
    4x+y-6z=11
  }
  \vspace{0.45in}
}
{
    \syslineskipcoeff{1}\sysalign{r,r}\sysdelim..\systeme{
        2x-z=13,
        x+y+z=-7,
        y-3z=22
    }
}
