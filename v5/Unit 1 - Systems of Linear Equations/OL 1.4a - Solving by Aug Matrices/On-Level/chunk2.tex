\section{Solving Systems with a {TI-84}}

\begin{myConceptSteps}{To solve a $3\times3$ system using a TI-84 calculator\dots}
  \myStep{augmented matrix}{%
    \begin{itemize}
      \item enter the matrix dimensions 
      \item enter the matrix entries 
    \end{itemize}
  }
  \myStep{RREF}{%
    Use the {\ttfamily rref()} function to convert the matrix to {\bfseries\itshape reduced row echelon form}
  }
  \myStep{solution}{%
    Read the \gap{values} for the solution from the \gap{last} column of the result.
    Make sure to recognize \gap{no solution} and \gap{infinite solutions} cases.
    }
\end{myConceptSteps}
