\section{Linear Equations with One Variable}

\begin{myConceptSteps}{To sketch the graph of a linear equation with {one} variable\dots}
    \myStep{isolate}{Get the variable by \gap{itself}.}
    \myStep{line}{%
        \begin{itemize}[nosep]
            \item $y = c$ {\LARGE \,$\Rightarrow$\,} \gap{horizontal} line at $c$ on the \gap{$y-axis$}.
            \item $x = c$ {\LARGE \,$\Rightarrow$\,} \gap{vertical} line at $c$ on the \gap{$x-axis$}.
        \end{itemize}
    }
\end{myConceptSteps}

\myProblems[Sketch the graphs of these equations.]
{
        $2x - 8 = 0$
        % 2x = 8
        % x = 4 (vert. line)
}
{
        $y + 1  = 7 - y$
        % y = 6 - y
        % 2y = 6 
        % y = 3 (hor. line)
}
{1.2in}
