\section{Solving Simple Systems Algebraically}

\begin{myConceptSteps}{To solve a simple system by substitution\dots}
    \myStep{$y=$}{Pick an equation that is already solved for $y$.}
    \myStep{substitute}{Replace \gap{$y$} in the other equation.}
    \myStep{solve for $x$}{Solve that equation for \gap{$x$}. You will get a number.}
    \myStep{solve for $y$}{Substitute the $x$-value into the equation in Step-1 and simplify.}
\end{myConceptSteps}

\myProblems[Solve these simple systems by substitution.]
{
    \vspace{-\baselineskip}
    \begin{align*}
        y &= x - 3 \\
        y &= 2x - 4
    \end{align*}
}
{
    \vspace{-\baselineskip}
    \begin{align*}
        2x + 2y = 6 \\
        y = x-5
    \end{align*}
}
