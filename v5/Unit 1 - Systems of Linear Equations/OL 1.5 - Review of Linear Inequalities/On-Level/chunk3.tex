\section{Is a point a solution?}

\begin{myConcept}{To determine if a point is a solution to a linear inequality\dots}
  \begin{itemize}[fullwidth]
    \item Sketch the graph of the solution.
    \item Plot the point as a big \gap{dot}.
    \item The point {is} a solution \myEmph{only if}
    \begin{itemize}
      \item it is \gap{inside} the shading, or 
      \item it is \gap{on} a solid line (\gap{not} \gap{on} a dashed line).
    \end{itemize}
  \end{itemize}
\end{myConcept}
