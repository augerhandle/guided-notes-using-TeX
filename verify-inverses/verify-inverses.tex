\begin{myKeyConcepts}[To verify that $f(x)$ and $g(x)$ are inverses\dots]
    \setlist{labelindent=\parindent,leftmargin=*,itemsep=1em}
    \begin{enumerate}
        \item Compose $f$ with $g$: find $f \circ g = f(g(x))$.
        \item Compose $g$ with $f$: find $g \circ f = g(f(x))$.
        \item Verify that the results of {\bfseries both} compositions is $x$.
    \end{enumerate}
\end{myKeyConcepts}
