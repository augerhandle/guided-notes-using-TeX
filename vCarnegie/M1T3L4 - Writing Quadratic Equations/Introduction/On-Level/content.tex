    \myObjective{understand}{the various ``parts'' of a \gap{parabola}}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{focus}{a \gap{point} used to construct a parabola}
    \myDefinition{directrix}{a \gap{line} used to construct a parabola}
    \myDefinition{vertex}{the part of a parabola \gap{closest} to the focus and directrix}
    \myDefinition{axis of symmetry}{the \gap{line} that goes down the middle of a parabola (AOS)}
    \myDefinition{$p$}{the \gap{distance} from the vertex to the focus or directrix}
\end{myVocabulary}
