\myProblemsWithContent[In these problems,
the \myEmph{vertex}, \myEmph{directrix}, \myEmph{axis of symmetry} and parabola are already shown,
along with the \myEmph{distance} from the directrix to the vertex.
Do the following.
\begin{itemize}[nosep]
    \item Plot the focus as a BIG DOT.
    \item Show the distance from the vertex to focus on the graph.
\end{itemize}
{
    \tiny
    Hints: 
    \begin{itemize}[nosep]
        \item The focus is on the axis of symmetry. 
        \item The vertex is halfway between the focus and directrix.
    \end{itemize}
}
]
{
    \begin{center}
    \begin{myTikzpictureGrid}{0.35} {10}{10} [solid]
        % directrix
        \draw[ultra thick, dashed, black] (-11,2) -- (11,2); 
        \node[] at (11.5,2.25) {$\mathscr{D}$};
        % vertex
        \draw[color=black,thick,fill=black] (4,4) circle [radius=0.25];
        \node[] at (3,3.4) {\small $V$};
        %parabola
        \tkzFct[ black, solid,  thick, samples=500, domain=-10:15,]{((x-4)**2)/8 + 4}
        % AOS
        \draw[ultra thick, dotted, black] (4,-11) -- (4,11);
        \node[] at (4,-11.5) {\small AOS};
        % directrix-to-vertex distance
        \draw[solid, black] (4,4) -- (7,3);
        \draw[solid, black] (4,2) -- (7,3);
        \node[] at (7.25,3) {\small $2$};
        %
        \whenTEACHER{
            % focus
            \draw[color=red, thick, fill=white] (4,6) circle [radius=0.35];
            \draw[color=red, thick, fill=red, opacity=1.0] (4,6) circle [radius=0.15];
            \node[red] at (3,7) {\small $F$};
            % vertex-to-focus distance
            \draw[solid, red] (4,6) -- (-1,5);
            \draw[solid, red] (4,4) -- (-1,5);
            \node[red] at (-1,5) {\small $2$};
        }
    \end{myTikzpictureGrid}\\[0.5\onelineskip]
\end{center}
}
