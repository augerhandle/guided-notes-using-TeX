\myWideProblemWithContent[%
Given the focus and vertex, find the equation of the parabola.
]
{
    Focus $F$ : $(-3,-5))$
    \hspace{3em} 
    Vertex $V$ : $(2,-5)$
    \tcblower
    \begin{minipage}{0.49\textwidth}
    \begin{myTikzpictureGrid}{0.4} {10}{10} [solid]
        \whenTEACHER{
            % directrix
            \draw[ultra thick, dashed, red] (7,11) -- (7,-11); 
            \node[] at (7,11) {$\mathscr{D}$};
            % AOS
            \draw[ultra thick, dashed, red] (-11,-5) -- (11,-5); 
            \node[] at (-8,-5.5) {\tiny AOS};
            % focus
            \draw[color=red,thick,fill=white] (-3,-5) circle [radius=0.35];
            \draw[color=red,thick,fill=red,opacity=1.0] (-3,-5) circle [radius=0.15];
            \node[red] at (-4,-5) {\small $F$};
            % vertex
            \draw[color=red,thick,fill=red] (2,-5) circle [radius=0.25];
            \node[red] at (3,-5) {\small $V$};
            % p-arrow
            \draw[arrows = {-Stealth[scale=1.5]}, very thick, solid, blue] (2,-5) -- (-3,-5); 
            \node[blue] at (0.1,-4.2) {$p=-5$};
            %parabola
            % \tkzFct[ red, dotted, very thick, samples=500, domain=-10:15,]{ ((x-2)**2)/(-16) + 2}
            \tkzFct[ red, dotted, very thick, samples=500, domain=-10:15,]{sqrt(-20*(x-2))-5 }
            \tkzFct[ red, dotted, very thick, samples=500, domain=-10:15,]{-sqrt(-20*(x-2))-5 }
        }
    \end{myTikzpictureGrid}
    \end{minipage}
    %
    \begin{minipage}{0.5\textwidth} 
        \whenTEACHER{
            \tiny
            \begin{enumerate}[nosep]
                \item Draw $F$ and $V$.
                \item Draw the AOS thru $F$ and $V$.
                \item Draw $\mathscr{D}$ on other side of the vertex, $\bm{\perp}$ to the AOS
                \item $h=2$, $k=-5$ (from coords of vertex)
                \item Draw arrow from $V$ to $F$. Since it goes LEFT, $p=-5$ (NEG).
                \item Sketch parabola (bend away from $\mathscr{D}$)
                \item since it's HORIZONTAL, pick : $(y-k)^2 = 4p (x-h)$
                \item substitute : \fbox{$(y+5)^2 = -20(x-2)$}
            \end{enumerate}
            \whenHONORS{
                \begin{itemize}[nosep]
                    \item solve for y : \fbox{$y = \pm \myRoot{-20 (x-2)} - 5$}
                \end{itemize}
                }
        }
    \end{minipage}
}
