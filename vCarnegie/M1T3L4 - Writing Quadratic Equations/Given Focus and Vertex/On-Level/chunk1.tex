\begin{myConceptSteps}{To write the equation of a parabola given its focus and vertex\dots}
    \myStep{Sketch $F$ and $V$}{Put the \gap{focus} and \gap{vertex} on a graph.}
    \myStep{AOS}{Sketch the \gap{axis} \gap{of} \gap{symmetry} through $V$ and $F$. }
    \myStep{Sketch $\mathscr{D}$}{Put the \gap{directrix} on the graph.
        \begin{itemize}[nosep]
            \item \gap{perpendicular} to the AOS 
            \item on the \gap{other} \gap{side} of the vertex
        \end{itemize}
    }
    \myStep{Find $h$ and $k$}{The vertex coordinates are \gap{$(h,k)$}.}
    \myStep{Find $p$}{
        Find $p$ as the distance between $V$ and $F$.
        \begin{itemize}[nosep]
            \item Draw an arrow from $V \longrightarrow F$.
            \item $p$ is the \gap{length} of that arrow.
            \item It is \gap{positive} if it points in the {\large $+x$} or {\large $+y$} direction.
            \item It is \gap{negative} if it points in the {\large $-x$} or {\large $-y$} direction.
        \end{itemize}
    }
    \myStep{Parabola}{Sketch the parabola.
        \begin{itemize}[nosep]
            \item Start at the \gap{vertex}.
            \item Bend \gap{away} from the directrix \gap{towards} the {focus}.
        \end{itemize}
    }
    \myStep{Pick which equation}{
        % Choose the general form of the equation:\\
        \begin{minipage}{0.32\textwidth} 
            \begin{tcolorbox}[colback=white] 
                \small\centering 
                \myEmph{Vertical Parabola}
                \tcblower
                \centering
                {$(x-\bm{h})^2 = 4\bm{p}(y-\bm{k})$}
            \end{tcolorbox}
        \end{minipage}
        % \hspace{10em}
        \begin{minipage}{0.32\textwidth}
            \begin{tcolorbox}[colback=white] 
                \small\centering
                \myEmph{Horizontal Parabola}
                \tcblower
                \centering
                {$(y-\bm{k})^2 = 4\bm{p}(x-\bm{h})$}
            \end{tcolorbox} 
        \end{minipage}
    }
    \myStep{Substitute}{
        Substitute $h$, $k$, and $p$ into the equation.
        Simplify the $4p$ but leave the $(x-h)$ and $(y-k)$ expressions alone.
        \whenHONORS{
            \begin{tcolorbox}[width = 5in, center, colback=white,]
                Be prepared to simplify the equation all the way to $y = \dots$.
            \end{tcolorbox}
        }
    }
\end{myConceptSteps}
