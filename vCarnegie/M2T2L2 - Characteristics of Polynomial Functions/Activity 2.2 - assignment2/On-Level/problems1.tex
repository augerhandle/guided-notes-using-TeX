\myProblemsWithContent[%
\begin{itemize}[nosep,fullwidth]
    \item Find the value of $A$, $B$, $C$, or $D$ that is involved in the transformed function.
    \item Describe the transformation.
    \item Complete the table by finding the coordinates of the transformed reference points.
    \item Sketch a graph of the transformed function.
\end{itemize}
]

    $m(x) = x^3$\\
    $n(x) = m(x-3) $
    \tcblower
    A B C D {\tiny\itshape (circle one)} = \gap{$C=3$}\\[0.5em]
    {\small\itshape transf. description:} \gap{right by 3}
    \vspace{0.5\onelineskip}
    \begin{center}
        \fbox{
            \begin{tabular}{ccc}
                \thead{{\itshape reference points}\\{\itshape on $m(x)$}} &  & \thead{{\itshape transformed points}\\{\itshape on $n(x)$}} \\ 
                \midrule 
                $(-2,-8)$ & $\rightarrow$ & \gap{$(1,-8)$} \\
                $(-1,-1)$ & $\rightarrow$ & \gap{$(2,-1)$} \\
                $(0,0)$   & $\rightarrow$ & \gap{$(3, 0)$} \\
                $(1,1)$   & $\rightarrow$ & \gap{$(4, 1)$} \\
                $(2,8)$   & $\rightarrow$ & \gap{$(5,8)$} \\
            \end{tabular}
        }
        \\[0.75em]
        \begin{myTikzpictureGrid}{0.24} {12}{12} [solid]

                \myDot{(3,0)}[red]

                }
