\myProblems[%
    \begin{itemize}[nosep,]
        \item Write the equation for the transformed function $g(x)$
        given these values of $A$, $B$, $C$, and $D$.
        Use this as your starting point:
        \[g(x) = \bm{A} f(\bm{B}(x-\bm{C}))+ \bm{D}\]
        {
            \tiny\itshape 
            If missing, $A$ and $B$ are 1, and you don't write them.
            If missing, $C$ and $D$  are 0, and you don't write them.
        }
        \item Describe all the transformations.
    \end{itemize}
    ]
{
    $A = 2$, $B = 3$
    \\\myAnswer{
        \tiny 
        g(x) = 2 f(3x)
        \\ v str by 2, h comp by 1/3
    }
}
{
    $C = 5$, $D = 13.5$
    \\\myAnswer{
        \tiny 
        g(x) = f(x-5) + 13.5
        \\ right 5, up by 13.5
        }
}
