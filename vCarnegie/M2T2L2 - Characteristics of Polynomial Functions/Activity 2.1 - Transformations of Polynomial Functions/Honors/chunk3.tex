\myProblems[
    In the following problems, 
    \vspace{-1\onelineskip}
    \begin{itemize}[nosep]
        \item Which constant is involved, $A$, $B$, $C$, $D$?
        \item Describe the transformation in words.
        \item Complete this: $(x,y) \rightarrow (\text{\underline{\phantom{xxx}}},\text{\underline{\phantom{xxx}}})$
    \end{itemize}
]
{
    $y = 2f(x)$
    \myAnswer{\hfill\tiny A=2, v stretch by 2, $(x,y)\rightarrow(x,2y)$}
}
{
    $y = \frac{1}{3}f(x)$
    \myAnswer{\hfill\tiny A=1/3, v compress by 1/3, $(x,y)\rightarrow(x,\frac{1}{3}y)$}
}
{0.9in}


\myProblems
{
    $y = f(\frac{1}{4}x)$
    \myAnswer{\hfill\tiny B=1/4, h stretch by 4, $(x,y)\rightarrow(4x,y)$}
}
{
    $y = f(5x)$
    \myAnswer{\hfill\tiny B=5, h compress by 1/5, $(x,y)\rightarrow(\frac{1}{5}x,y)$}
}
{0.9in}


\myProblems
{
    $y = -f(x)$
    \myAnswer{\hfill\tiny A=-1, refl across x, $(x,y)\rightarrow(x,-y)$}
}
{
    $y = f(-x)$
    \myAnswer{\hfill\tiny B=-1, refl across y, $(x,y)\rightarrow(-x,y)$}
}
{0.9in}


\myProblems
{
    $y = f(x-6)$
    \myAnswer{\hfill\tiny C=6, right by 6, $(x,y)\rightarrow(x+6,y)$}
}
{
    $y = f(x+7)$
    \myAnswer{\hfill\tiny C=-7, left by 7, $(x,y)\rightarrow(x-7,y)$}
}
{0.9in}


\myProblems
{
    $y = f(x) + 8$
    \myAnswer{\hfill\tiny D=8, up by 8, $(x,y)\rightarrow(x,y+8)$}
}
{
    $y = f(x)-9$
    \myAnswer{\hfill\tiny D=-9, down by 9, $(x,y)\rightarrow(x,y-9)$}
}
{0.9in}
