\section{Sketching the Graphs}

You can use Desmos 
to check that you multiplied correctly.
\begin{itemize}[nosep]
    \item The curves should be the same.
    \item The $x$-intercepts should be the same.
    \item The maxima/minima should be the same.
\end{itemize}

\vspace{-2\onelineskip}
\myProblemsWithContent[]
{
    Multiply these linear factors: $(x+2)$, $(x-3)$ to get a new polynomial, $g(x)$.
    \begin{center}
        \vspace{0.75in}
        $g(x) = $ \gap{$x^2 - x -6$}
    \end{center}
    Sketch the graphs of 
    \begin{itemize}[nosep]
        \item $y = (x+2)(x-3)$ \myAnswer{\hfill\tiny before mult.}
        \item $y = g(x)$ \myAnswer{\hfill\tiny after mult.}
    \end{itemize}
    Sketch dots at the $x$-intercepts and maximum/minimum.\\
    \begin{myTikzpictureGrid}{0.3} {8}{8} [solid]
        \whenTEACHER{
            \tkzFct[ red, dotted, very thick, samples=500, domain=-10:15,]{(x+2)*(x-3)}
            \tkzFct[ red, dotted, very thick, samples=500, domain=-10:15,]{x**2 - x - 6}
            \draw[color=red,thick,fill=red] (-2,0) circle [radius=0.25];
            \draw[color=red,thick,fill=red] (3,0) circle [radius=0.25];
            %min
            \draw[color=red,thick,fill=red] (1/2,-6.25) circle [radius=0.25];
            \node[red] at (0.5,-7) {\tiny min};
        }
    \end{myTikzpictureGrid}
    \myAnswer{\hfill\tiny same graphs!}
}
