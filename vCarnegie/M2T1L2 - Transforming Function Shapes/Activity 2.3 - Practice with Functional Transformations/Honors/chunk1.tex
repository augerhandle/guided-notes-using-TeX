Use the \gap{box} \gap{method} to multiply polynomials.
In some problems, you might need to use it \gap{twice}.

\vspace{-0.5em}
\myProblemsWithContent
{
    Determine the product, $h(x)$, 
    of the linear and quadratic functions, $f(x)$ and $g(x)$.
    \begin{itemize}[nosep]
        \item $f(x) = 2x-5$
        \item $g(x) = 2x^2 - 3x + 4$
        \item $h(x) = f(x) \cdot g(x)$
    \end{itemize}
    \tcblower
    \vspace{1.5in}
    $h(x) =$ \gap{\tiny$4 x^3 - 16 x^2 + 23 x - 20$}
}
{
    Determine the product of these three linear factors:
    $3x(4-x)(3-2x)$.
    \myAnswer{\hfill\tiny $6 x^3 - 33 x^2 + 36 x$}
    \tcblower 
}
