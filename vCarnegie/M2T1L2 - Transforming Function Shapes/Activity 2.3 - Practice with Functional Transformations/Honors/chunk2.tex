\section{Domain and Range}

\myProblemsWithContent 
{
    Graph the quadratic parent function.
    Find its domain and range.\\
    \begin{minipage}{0.39\textwidth}
        \begin{myTikzpictureGrid}{0.25} {5}{10} 
            \whenTEACHER{
                \tkzFct[ red, solid, thick, samples=100, domain=-5:5,]{x**2}
            %     \draw[color=red,thick,fill=red] (-7/6,121/12) circle [radius=0.25];
            %     \node[red] at (-1.2,11) {\tiny max};
            }
        \end{myTikzpictureGrid}
    \end{minipage}
    \hfill
    \begin{minipage}{0.55\textwidth}
        \small
        \myEmph{Domain:}
        \begin{itemize}[fullwidth,nosep]
            \item \gap{$-\infty < x < \infty$}
            \item $x \in $ \gap{$(-\infty,\infty)$}
        \end{itemize}
        \myAnswer{\tiny true for ALL quadr fcts\\}
        \vspace{1em}
        \myEmph{Range:}
        \begin{itemize}[fullwidth,nosep]
            \item \gap{$y \ge 0$}
            \item $y \in $ \gap{$[0,\infty)$}
        \end{itemize}
    \end{minipage}
}

\myProblemWithContent 
{
    Find the domain and range of the function 
    $h(x) = x (3-x) (5-2x) $
    \myAnswer{\tiny graph FACTORED form}\\
    \begin{minipage}{0.39\textwidth}
        \begin{myTikzpictureGrid}{0.29} {5}{7} 
            \whenTEACHER{
                \tkzFct[ red, solid, thick, samples=100, domain=-3:5,]{x * (3-x) * (5-2*x)}
            %     \draw[color=red,thick,fill=red] (-7/6,121/12) circle [radius=0.25];
            %     \node[red] at (-1.2,11) {\tiny max};
            }
        \end{myTikzpictureGrid}
    \end{minipage}
    \hfill
    \begin{minipage}{0.55\textwidth}
        \small
        \myEmph{Domain:}
        \begin{itemize}[fullwidth,nosep]
            \item \gap{$-\infty < x < \infty$}
            \item $x \in $ \gap{$(-\infty,\infty)$}
        \end{itemize}
        \myAnswer{\tiny true for ALL cubic fcts\\}
        \vspace{1em}
        \myEmph{Range:}
        \begin{itemize}[fullwidth,nosep]
            \item \gap{$-\infty < y < \infty$}
            \item $y \in $ \gap{$(-\infty,\infty)$}
        \end{itemize}
        \myAnswer{\tiny true for ALL cubic fcts\\}
    \end{minipage}
}
