\section{Adding and Subtracting}

\begin{myConceptSteps}{To add or subtract polynomials\dots}
    \myStep{distribute negatives}{%
        \gap{Distribute} \gap{minus} \gap{signs} in front of parentheses,
        and \gap{get} \gap{rid} \gap{of} the parentheses.
    }
    \myStep{combine}{%
        \gap{Combine} {like} {terms} with 
        \begin{itemize}
            \item the same \gap{variables}
            \item the same \gap{exponents}.
        \end{itemize}
        }
\end{myConceptSteps}


\myWideProblem[Simplify these expressions by adding or subtracting.]
{
    \centering 
    $ (x^2 - 2x + 3) + (4x^2 - 5x + 6)$
}
{2in}
