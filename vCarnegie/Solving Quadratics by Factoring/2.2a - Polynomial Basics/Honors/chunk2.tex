\section{Adding and Subtracting}

\begin{myConceptSteps}{To add or subtract polynomials\dots}
    \myStep{distribute negatives}{\gap{Distribute} any \gap{minus} \gap{signs} in front of parentheses.}
    \myStep{simplify}{\gap{Get rid} of the parentheses.}
    \myStep{combine}{%
        \gap{Combine} {like} {terms}.
        {\bfseries\itshape Only} combine terms with 
        \begin{itemize}
            \item the same \gap{variables}
            \item the same \gap{exponents} on each variable.
        \end{itemize}
        }
\end{myConceptSteps}
