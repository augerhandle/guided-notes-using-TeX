    \myEmph{Quadratic form} trinomials look like this: 
    {
        \Large
        \begin{equation*}
            a x^m + b x^n + c 
        \end{equation*} 
    }
    where \gap{$m = 2n$}.
\end{tcolorbox}
\end{minipage}

    \renewcommand{\arraystretch}{1}
    \Large
    \begin{center}
    \begin{tabular}{c|c}
        quadratic form & \myEmph{not} \\
        \toprule 
        $x^4 + 5x^2 + 6$    & $x^4 + 5x + 6$ \\
        $3t^8 - 8t^4 - 16$  & $3t^5 - 8t^4 - 16$ \\
    \end{tabular} 
    \end{center}
