\begin{tcolorbox}[center,width=5in]
    We will factor \myEmph{quadratic form} trinomials that look like this: 
    {
        \Large
        \begin{equation*}
            a x^4 + b x^2 + c 
        \end{equation*} 
    }
\end{tcolorbox}

\begin{myConceptSteps}{To factor a \myEmph{quadratic form} trinomial\dots}
    \myStep{rewrite}{%
        Rewrite the polynomial in \gap{descending} \gap{order}.
    }
    \myStep{constants}{Write down $a$, $b$, $c$. Calculate \gap{$ac$}.}
    \myStep{T-chart}{Find the \gap{factors} of $ac$ that add to $b$. Use a \gap{T-chart}.}
    \myStep{inside the box}{Fill in the $a$ and $c$ boxes. Put the factors in the two blanks (any order).}
    \myStep{outside the box}{
        \begin{itemize}[nosep]
            \item Find the GCF of the \gap{top} \gap{row}. Write it outside on the left.
            \item Fill in the rest of the outside by thinking about 
                a \gap{times} \gap{table}.
        \end{itemize}
    }
    \myStep{result}{%
        The result is the product of 
        \begin{itemize}[nosep]
            \item the \gap{left} outside column and 
            \item the \gap{top} outside row.
        \end{itemize}
    }
\end{myConceptSteps}
