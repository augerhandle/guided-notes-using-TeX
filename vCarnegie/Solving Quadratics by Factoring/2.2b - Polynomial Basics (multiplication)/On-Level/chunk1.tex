\section{Monomials $\times$ Polynomials}

\vfil 

\begin{tcolorbox}[center,width=5.5in]
    \large
    To multiply a {\bfseries\itshape monomial} (one term) 
    times a 
    {\bfseries\itshape polynomial} (many terms),
    you \gap{distribute}.
\end{tcolorbox}

\vfil

        $ (7x^2)  (x + 3) $
    }
    {
        $ (2z^2)  (4z^3 - z) $
    }
    [\large]{3in}
