\section{Monomials $\times$ Monomials}

\begin{tcolorbox}[center,width=6in]
    To multiply \gap{monomials} \gap{times} \gap{monomials},
    use the \gap{product} \gap{property} exponent rule.
    {
        \Large
        \[
            B^n B^m = B^{n+m}
        \]
    }
    Remember to move any coefficients \gap{out} \gap{front} first.
\end{tcolorbox}


\myProblems[Multiply the following monomials.]
    {
        $ (2x) (3x^4) $
    }
    {
        $ (-4x^2)(2x) $
    }
    {1.5in}
