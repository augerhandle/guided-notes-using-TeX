\section{Recognizing Differences of Squares}

\begin{myConceptSteps}{To recognize a difference of squares\dots}
    [A polynomial is a \gap{difference} \gap{of} \gap{squares} 
    if all of the following are true:
    ]
    \myStep{2 terms}{The polynomial has exactly \gap{two} \gap{terms}}
    \myStep{coefficients}{The integer \gap{coefficients} of both terms are perfect squares.}
    \myStep{subtraction}{The two terms are being \gap{subtracted}.}
    \myStep{degree}{The degree of the terms is \gap{even} or \gap{zero}.}
\end{myConceptSteps}

These are differences of squares:

{
    \normalsize
    \hfill
    \begin{minipage}{0.25\textwidth}
        \begin{itemize}[nosep]
            \item $x^2 - 25$
            \item $25 - x^2$
        \end{itemize}
    \end{minipage}
    \hfill
    \begin{minipage}{0.25\textwidth}
        \begin{itemize}[nosep]
            \item $9x^2 - 25$ 
            \item $25 - 9x^2$
        \end{itemize}
    \end{minipage}
    \hfill
    \begin{minipage}{0.25\textwidth}
        \begin{itemize}[nosep]
            \item $9x^2 - 1$
            \item $1 - 9x^2$
        \end{itemize}
    \end{minipage}
    \hfill{}
}
