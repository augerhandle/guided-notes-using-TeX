\begin{myConcept}{To convert between factors, zeros, roots, solutions, and $x$-intercepts related to $f(x)$\dots\dots}
\begin{itemize}[fullwidth]
    \item If \gap{$(x-a)$} is a linear factor, then 
    \begin{itemize}
        \item[$\circ$] $a$ is the \gap{$x$-coordinate} of an $x$ intercept.
        \item[$\circ$] $a$ is a \gap{zero} of $f(x)$.
        \item[$\circ$] $a$ is a \gap{root}/\gap{solution} of the equation $f(x)=0$.
    \end{itemize}
\end{itemize}
\end{myConcept}

\begin{tcolorbox}[center,width=6in]
    \begin{itemize}
        \item \myEmph{Zeros}, \myEmph{roots}, and \myEmph{solutions} all have \
        the \gap{same} \gap{value}.
        \item (I know. It's crazy. But you'll bump into all these words.)
    \end{itemize}
\end{tcolorbox}
