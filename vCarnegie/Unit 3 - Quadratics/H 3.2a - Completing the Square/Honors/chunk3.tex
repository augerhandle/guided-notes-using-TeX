

\myProblemsWithContent[Rewrite these expressions by completing the square.]
{
    $x^2 + 14x$
    \tcblower 
    $B = \text{\gap{14}}$
    \hfill
    $\frac{B}{2} = \text{\gap{7}}$
    \hfill
    $\left(\frac{B}{2}\right)^2 = \text{\gap{49}}$
    \whenSTUDENT{\vspace{1.25in}}
    \whenTEACHER{\vspace{0.75in}}
}
{
    $x^2 -12x$
    \tcblower
    $B = \text{\gap{-12}}$
    \hfill
    $\frac{B}{2} = \text{\gap{-6}}$
    \hfill
    $\left(\frac{B}{2}\right)^2 = \text{\gap{36}}$
}
