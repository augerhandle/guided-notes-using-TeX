\section{Evaluating}

\begin{myConceptSteps}{To evaluate a quadratic function at \gap{$x=a$}\dots}
    \myStep{rewrite}{Replace $x$ by \gap{parentheses} with blanks.}
    \myStep{substitute}{Write $a$ between \gap{all} the parentheses.}
    \myStep{simplify}{Simplify the expression. The answer will be a \gap{number}.}
    \myStep{answer}{Write the answer as $f(a) = \boxed{\phantom{X}}$.}
\end{myConceptSteps}

\myWideProblem[Evaluate this quadratic function at the given value of $x$.]
{
    $f(x) = -x^2 - 10x - 17 $
    \quad at \quad
    $x = -3$
    }
{1.5in}
[\myAnswer{$f(-3) = 4$}]
