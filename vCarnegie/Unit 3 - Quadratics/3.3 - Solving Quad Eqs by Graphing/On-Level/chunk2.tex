\myProblemsWithContent[%
Solve these quadratic equations.
Put \myEmph{dots} at the $x$-intercepts.
Write the solutions as \fbox{$x=$\dots}.
]
{
    $ x^2 + 3x - 12 = 2x $\\[2\baselineskip]
    \begin{myTikzpictureGrid}{0.25} {10}{10}
        % \tkzText(13,8){$x - y = 0$}
        % \tkzFct[ solid, ultra thick, samples=100, domain=-6:6,]{x**2 - 6*x +8}
        % \tkzText(-13,7.25){$x + 2y = 6$}
        \whenTEACHER{\tkzFct[ solid, red, ultra thick, samples=100, domain=-10:10,]{x**2 + x -12 }}
    \end{myTikzpictureGrid}\\
    %
    \vspace{0.5in}
    \myAnswer{$x = -4, 3$}
}
{
    $ x^2 + 25= 10x $\\[2\baselineskip]
    \begin{myTikzpictureGrid}{0.25} {10}{10}
        % \tkzText(13,8){$x - y = 0$}
        \whenTEACHER{\tkzFct[ solid, red, ultra thick, samples=100, domain=-10:10,]{x**2 - 10*x + 25 }}
        % \tkzText(-13,7.25){$x + 2y = 6$}
    \end{myTikzpictureGrid}\\
    %
    \vspace{0.5in}
    \myAnswer{$x=5$}
}
