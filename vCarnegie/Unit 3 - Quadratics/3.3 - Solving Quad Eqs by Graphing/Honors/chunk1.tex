\begin{myConceptSteps}{To solve a quadratic equation by graphing\dots}
    \myStep{standard form}{%
        Rewrite the equation as \gap{$ax^2 + bx + c = 0$}.
        }
    \myStep{graph}{%
        Graph the function $f(x) = ax^2 + bx + c$
        using \myKey{Y=} on a {\bfseries TI-84}.
    }
    \myStep{solution}{%
        The \myEmph{solutions}  
        are the \gap{$x$-intercepts} of the function's graph.
    }
\end{myConceptSteps}
