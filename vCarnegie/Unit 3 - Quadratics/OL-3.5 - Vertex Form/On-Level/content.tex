\begin{myVocabulary}
    \myDefinition{quadratic}{degree 2 (largest exponent is 2)}
    \myDefinition{parabola}{the \gap{shape} of the graph of a quadratic function}
    \myDefinition{vertex}{the \gap{highest} or \gap{lowest} point on a parabola}
    \myDefinition{symmetric}{\gap{mirror} images}
    \myDefinition{standard form}{\gap{$f(x) = ax^2 + bx + c $}}
    \myDefinition{vertex form}{\gap{$f(x) = a(x-h)^2 + k $}}
\end{myVocabulary}
