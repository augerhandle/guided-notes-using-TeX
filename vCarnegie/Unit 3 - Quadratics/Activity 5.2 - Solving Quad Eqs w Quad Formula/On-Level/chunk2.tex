\section{Solving quadratic equations}

\begin{myConceptSteps}{
    To solve a \gap{quadratic} equation using the quadratic formula\dots
}
\myStep{rewrite}{%
Write the equation in \gap{standard} \gap{form}:  $ax^2 + bx + c = 0$
\begin{itemize}
    \item Write down $a$, $b$, $c$. 
\end{itemize}
}
\myStep{discriminant}{Calculate \gap{$ D = b^2 - 4ac $}.}
\myStep{quadratic formula}{
    Substitute $a$, $b$, $c$, and $D$ into \hspace{1em}
        \small
        $\displaystyle x = \frac{-b \pm \sqrt{D}}{2a}$
}
\myStep{evaluate}{\gap{Simplify} the right-hand-side of the equations.
    \begin{itemize}
        \item When $D$ is \gap{positive}, there are \gap{two} real solutions.
        \item When $D =$ \gap{$0$}, there is \gap{one} \gap{solution}.
        \item When $D$ is \gap{negative}, there are \gap{two} solutions with an $\bm{i}$ in them.\myAnswer{$\Leftarrow$ NEW!!}
    \end{itemize}
}
\end{myConceptSteps}
