\section*{Imaginary Numbers}
\myProblemsWithContent[Evaluate these square roots. Why is your answer correct?]
    {
        $\sqrt{25}$
        \tcblower
        \myAnswer{$5$, because $5^2=25$.}
    }
    {
        $\sqrt{100}$
        \tcblower
        \myAnswer{$10$, because $10^2=100$.}
    }

\vspace{1\onelineskip}
What happens when we take the square root of a \myEmph{negative} number?
\myWideProblem[]
    {
        $\sqrt{-49}$
        \myAnswer{
            \tiny
            not $7$ or $-7$, b/c both are POSITIVE when squared
        }
    }
    {0.5in}

    \noindent\vspace{1\onelineskip}
    \myEmph{Imaginary numbers} 
    are a \gap{new} \gap{kind} \gap{of} \gap{number}.

    \vspace{-1\onelineskip}
    \begin{center}
        \small
        \centering
        \begin{tcolorbox}[width=6in,colback=white]
            \begin{minipage}{0.4\textwidth}
                Imaginary numbers are written with the letter {\large $\bm{i}$}.
            \end{minipage}
            \hfill
            \begin{minipage}{0.25\textwidth}
                $i =$ \gap{$\sqrt{-1}$}
            \end{minipage}
            \hfill
            \begin{minipage}{0.25\textwidth}
                $i^2 = $ \gap{$-1$}
            \end{minipage}
        \end{tcolorbox}
    \end{center}
    \vspace{-1\onelineskip}
    Using $i$,
    we can find the square root of negative numbers, too.
    
    \myProblems[Evaluate these square roots. ]
    {
        $\sqrt{-1}$
        \myAnswer{
            \hfill\tiny
            $i$,
            b/c $i^2=-1$.
        }
    }
    {
        $\sqrt{-81}$
        \myAnswer{
            \hfill\tiny
            $9i$,
            b/c $(9i)^2= -81$.
        }
    }
    {0.5in}
