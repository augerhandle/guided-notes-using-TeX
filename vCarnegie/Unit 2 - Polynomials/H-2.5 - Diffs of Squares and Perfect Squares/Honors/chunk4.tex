\section{Factoring Perfect Square Trinomials}

\begin{tcolorbox}[center,width=6in]
    When a polynomial is a 
    \gap{oerfect} \gap{square} \gap{trinomial},
    it may be rewritten like this:
    {
    \Large
    \begin{align*}{}
        (A^2 + 2AB + B^2) &= (A+B)^2 \\
        (A^2 - 2AB + B^2) &= (A-B)^2
    \end{align*}
    }
    where $A$ is the \gap{square} \gap{root} of the first term 
    and $B$ is the \gap{square} \gap{root} of the last term.
\end{tcolorbox}


\begin{myConceptSteps}{To factor perfect square trinomials\dots}
    \myStep{$A$}{Look at the \gap{coefficient} and \gap{variables} in the \gap{first} {term}:
        \begin{itemize}
            \item Start $A$ with the \gap{square} \gap{root} of the coefficient.
            \item Split any variables into \gap{two} \gap{equal} \gap{groups}.
                \begin{itemize}
                    \item[$\circ$] $x^2$ {\Large $\Rightarrow$} \gap{$x$} \gap{$x$}
                    \item[$\circ$] $x^6$ {\Large $\Rightarrow$} \gap{$x^3$} \gap{$x^3$}
                \end{itemize}
                Add \gap{one} of those groups at the end of $A$.
        \end{itemize}
    }
    \myStep{$B$}{Do the same thing but for the \gap{second} term.}
    \myStep{fill in}{%
        Substitute \gap{$A$} and \gap{$B$} into $(A\pm B)$.
        Replace $\pm$ by $+$ or $-$ depending on the \gap{sign} of the middle term.
    }
    \myStep{check}{Verify that the \gap{middle} \gap{term} in indeed $\pm 2AB$.}
\end{myConceptSteps}

\vspace{-1\onelineskip}
\myProblems[Factor these perfect square trinomials.]
{
    \centering $x^2 + 10x + 25$
    % A = x 
    % B = 5
    % 2AB = 10X
}
{
    \centering $9x^2 -12x + 4$
    %A = 3X 
    %B = 2 
    %2AB = 12X
}
{1.25in}

\begin{tcolorbox}[center,width=4in]
    \begin{itemize}[nosep]
        \item Notice that $(A+B)^2$ is \myEmph{not} $(A^2 + B^2)$. 
        \item Notice that $(A-B)^2$ is \myEmph{not} $(A^2 - B^2)$.
    \end{itemize} 
\end{tcolorbox}
