\noindent Logarithms are \gap{inverses} of exponential functions with the \gap{same} \gap{base}. 
\begin{tcolorbox}[center,width=4in,colback=white,]
    \begin{center}
        \normalsize
        \renewcommand{\arraystretch}{1.75}
        \begin{tabular}{ccc} 
            \itshape{exp function} & & \itshape{log function} \\
            \hline
            $f(x) = $\gap{$b^x$} & {\Large $\Leftrightarrow$} & $f^{-1}(x) = $\gap{$\myLog[b](x)$} \\
        \end{tabular}
    \end{center}
    where $b$ is the \gap{base} 
\end{tcolorbox}
