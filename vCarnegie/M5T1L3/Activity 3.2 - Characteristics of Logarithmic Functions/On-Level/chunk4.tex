
\myProblemsWithContent[Find the inverse of these logarithmic functions.]
{
    $f(x) = \myLog[5](x)$      
    \hfill\myAnswer{I DO}
    \tcblower
    \begin{itemize}[fullwidth,nosep]
        \setlength\itemsep{1em}
        \item base : $b=$\gap{5}
        \item exponential base $b$ : \gap{$5^x$}
        \item inverse function : \gap{$f^{-1}(x) = 5^x$}
    \end{itemize}
}
