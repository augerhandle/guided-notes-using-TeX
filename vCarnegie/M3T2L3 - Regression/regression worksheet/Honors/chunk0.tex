\small
\noindent
In these problems, you will use \myDesmos to do 
\myEmph{linear}, \myEmph{quadratic}, and \myEmph{cubic} regression.

\begin{center}
\small
\begin{tabular}{l|l|cr}
    \toprule
    {\itshape model} 
        & {type this into \myDesmos} 
        &
        & {\itshape regression equation} \\
    \midrule 
    {\scshape Linear} 
        & {\ttfamily
            y1 $\sim$ m x1 + b
        }
        & $y = $& $\bm{m}x + \bm{b}$
        \\
    {\scshape Quadratic} 
        & {\ttfamily
            y1 $\sim$ a x1$\wedge$2 + bx1 + c
        }
        & $y = $& $\bm{a}x^2 + \bm{b}x + \bm{c}$
        \\
    {\scshape Cubic} 
        & {\ttfamily
            y1 $\sim$ a x1$\wedge$3 + bx1$\wedge$2 + cx1 + d
        }
        & $y = $& $\bm{a}x^3 + \bm{b}x^2 + \bm{c}x + \bm{d}$
        \\
    \bottomrule
\end{tabular}
\end{center}
