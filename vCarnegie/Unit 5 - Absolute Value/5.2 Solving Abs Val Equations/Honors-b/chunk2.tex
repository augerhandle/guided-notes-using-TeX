\section{One Absolute Value but Variables on Both Sides}


\begin{myConceptSteps}{To solve absolute value equations with variables on ``both sides''\dots}
    \myStep{isolate}{Get the absolute value expression \gap{by} \gap{itself}.}
    \myStep{split}{%
        Write two equations for the \gap{negative} and \gap{positive} cases.
        \begin{itemize}
            \item These equations do \gap{not} have absolute values in them!
        \end{itemize}
    }
    \myStep{solve}{Solve each of the \gap{simpler} equations.}
    \myStep{check}{Reject any extraneous solutions.
        \begin{itemize}[nosep]
            \item You find extraneous solutions by \gap{substituting} the 
            solutions from the previous step back into the \gap{original} equation.
        \end{itemize}
    }
\end{myConceptSteps}

\myProblems[Solve these equations. Make sure to check for extraneous solutions.]
{
    $
        \left| 5 - \frac{3}{4}x \right| = \frac{1}{4}x +2
    $
}
{
    $
        6|c-3| = 12c
    $
}
{4.5in}
[\myAnswer{3, 14}]
[\myAnswer{$c=1$ \quad ($-3$ is extraneous)}]
