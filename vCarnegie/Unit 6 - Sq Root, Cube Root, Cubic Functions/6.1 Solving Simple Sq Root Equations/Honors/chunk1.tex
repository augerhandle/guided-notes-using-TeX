\section{Introduction}

You are learning to solve different kinds of equations.

\newcommand{\myExtraVSpace}{\vspace{0\onelineskip}}

\begin{tcbraster}[
    raster equal height, 
    raster columns = 2,
    raster column skip = 0.5in,
]
    \begin{tcolorbox}[colback=white,boxrule=0.5pt,]
        \small
        solving \myEmph{linear} equations\dots
        \begin{center}
            $5x +3 = 13$
        \end{center}
        \whenTEACHER{
            \tiny Get $x$ by itself.
        }
        \whenSTUDENT{
            \myExtraVSpace
        }
    \end{tcolorbox}
    \begin{tcolorbox}[colback=white,boxrule=0.5pt,]
        \small
        solving \myEmph{quadratic} equations\dots
        \begin{center}
            $x^2 + 5x + 6 = 0$
        \end{center}
        \whenTEACHER{
            \tiny use factoring, quad. formula
        }
        \whenSTUDENT{
            \myExtraVSpace
        }
    \end{tcolorbox}
    \begin{tcolorbox}[colback=white,boxrule=0.5pt,]
        \small 
        solving \myEmph{absolute value} equations\dots
        \begin{center}
            $|4x -2 | = 10$
        \end{center}
        \whenTEACHER{
            \tiny split into pos./neg. cases
        }
        \whenSTUDENT{
            \myExtraVSpace
        }
    \end{tcolorbox}
    \begin{tcolorbox}[colback=white,boxrule=2pt,]
        \small 
        solving \myEmph{square root} equations\dots
        \begin{center}
            \hfill
            $\myRoot{5x+1} = 4$
            \hfill 
            $\left({5x+1}\right)^\frac{1}{2} = 4$
            \hfill\,
        \end{center}
        \whenTEACHER{
            \tiny SQUARE BOTH SIDES
        }
        \whenSTUDENT{
            \myExtraVSpace
        }
    \end{tcolorbox}
\end{tcbraster}
