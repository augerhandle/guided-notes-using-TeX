\section{Horizontal and Vertical Lines}

\begin{minipage}[t]{0.49\textwidth}
    The equation of a \gap{horizontal} line is
    \begin{center}
    \begin{tcolorbox}[center,colback=white,width=1.2in]
        \myAnswer{$y=k$}
    \end{tcolorbox}
    \begin{myTikzpictureGrid}{0.4} {6}{6}
        % \tkzText(13,8){$x - y = 0$}
        \draw [ultra thick, dashed] (-7,-4) -- (7,-4);
        \whenTEACHER{
            \node[draw,red,fill=white] at (9,-4) {$y=-4$};
        }
        %
        \draw [ultra thick, dashed] (-7,1) -- (7,1);
        \whenTEACHER{
            \node[draw,red,fill=white] at (9,1) {$y=-1$};
        }
        %
        \draw [ultra thick, dashed] (-7,5) -- (7,5);
        \whenTEACHER{
            \node[draw,red,fill=white] at (8.5,5) {$y=5$};
        }
    \end{myTikzpictureGrid}
    \end{center}
\end{minipage}
