\begin{myConceptSteps}{To solve an equation with rational expressions\dots}
    \myStep{factor}{Factor all the \gap{denominators} (not the numerators).}
    \myStep{LCD}{%
        Find the \gap{LCD} of all the terms in the equation.
    }
    \myStep{excluded values}{%
        Find the \gap{excluded} values by finding the \gap{zeros} 
        of the LCD.
    }
    \myStep{clear the fractions}{%
        Multiply \gap{both} \gap{sides} of the equation by the \gap{LCD}.
        \begin{itemize}[nosep]
            \item There won't be any more fractions!
        \end{itemize}
    }
    \myStep{solve}{%
        Solve the equation.
        \begin{itemize}[nosep]
            \item \gap{Drop} any \gap{solutions} that are \gap{excluded}.
        \end{itemize}
    }
\end{myConceptSteps}


\myProblems[Solve these rational equations.]
{
    $
    \frac{1}{4x^2} + \frac{x+1}{8x^2} = \frac{1}{2x^2}
    $
}
{
    $
    \frac{1}{4} = \frac{n+1}{8n} - \frac{1}{2}
    $
}
{2.3in}
[\myAnswer{$x=1$}]
[\myAnswer{$n=\frac{1}{5}$}]
