\section{Simplifying rational expressions}

\begin{myConceptSteps}{To \myEmph{simplify} a rational expression\dots}
    \myStep{factor}{Completely \gap{factor} the numerator and denominator.}
    \myStep{domain}{The domain is \gap{$x \ne \text{excluded values}$}.
    }
    \myStep{simplified function}{\gap{Cancel} common factors in the numerator and denominator.
        \begin{itemize} 
            \item You can only cancel terms that are \gap{multiplied} together.
        \end{itemize}
    }
    \myStep{result}{%
        The result has \gap{two} parts:
        \begin{itemize}
            \item the \gap{simplified} \gap{function} 
            \item the \gap{domain}        
        \end{itemize}
    }
\end{myConceptSteps}

\myProblems[Simplify these rational expressions. State the domain as well as the simplified function.]
    {
        \large
        $\frac{2x+4}{(x+2)(x+8)}$
    }
    {
        \large
        $\frac{x^2 - 7x + 12}{2x - 6}$
    }
{4.25in}
[\myAnswer{sf: $\frac{2}{x+8}$, dom: $x \ne -2, -8$}]
[\myAnswer{sf: $\frac{x-4}{2}$, dom: $x \ne 3$}]
