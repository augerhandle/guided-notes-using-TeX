
\begin{myConceptSteps}{To solve a raational by graphing\dots}
    \myStep{two equations}{
        Write the \gap{left} and \gap{right} sides as 
        \gap{two} equations using {\large$y=$}.
    }
    \myStep{TI-84}{Enter each equation into the \myTi calculator.
        Use \myKey{ALPHA} \myKey{Y=} for fractions.
        \begin{itemize}
            \item Enter the \gap{left} side using \myKey{Y=}.
            \item Enter the \gap{right} side using \myKey{Y=}.
        \end{itemize}
        \begin{tcolorbox}[center,width=4in,colback=white]
            You \myEmph{must} use \myKey{ALPHA} \myKey{Y=} for the fractions.
        \end{tcolorbox}
    }
    \myStep{intersect}{%
        Find the points where the two curves \gap{intersect}.
        \begin{itemize}
            \item Use \myKey{2ND} \myKey{TRACE} \myKey{Intersect} 
             to find the coordinates of the points of intersection.
            \item There might be more than one.
        \end{itemize}
    }
    \myStep{solution}{%
        The solution to the problem is the
        \gap{$x$-coordinates} of the points.
    }
\end{myConceptSteps}
