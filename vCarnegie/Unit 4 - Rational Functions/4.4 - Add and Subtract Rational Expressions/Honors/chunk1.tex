\section{Least Common Denominators}

\begin{myConceptSteps}{To find the LCD of rational numbers (\gap{fractions})\dots}
    \myStep{factor}{%
        Factor the \gap{denominators} with 
        \gap{factor} \gap{trees}.
    }
    \myStep{list}{%
        Write a list of the factors.
        \begin{itemize}
            \item For each different factor in any tree,
            \item find where it occurs \gap{most} \gap{often}, and 
            \item put it in the list that many times.
        \end{itemize}
    }
    \myStep{multiply}{%
        Multiply the list together -- that's the LCD.
    }
\end{myConceptSteps}


\myProblems[Find the LCD of these fractions.]
    {
        $
        \frac{1}{8}
        \qquad\qquad
        \frac{1}{30}
        $
    }
    {
        $
        \frac{1}{12}
        \qquad\qquad
        \frac{2}{18}
        \qquad\qquad
        \frac{1}{20}
        $
    }
    {1.5in}
    [\myAnswer{120}]
    [\myAnswer{180}]
