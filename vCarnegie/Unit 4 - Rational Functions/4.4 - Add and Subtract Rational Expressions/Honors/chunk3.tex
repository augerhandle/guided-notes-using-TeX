\section{Adding and subtracting rational expressions}


\begin{myConceptSteps}{To add or subtract two rational expressions\dots}
    \myStep{factor}{Factor all the \gap{denominators}.}
    \myStep{LCD}{%
        Find the \gap{LCD} of the fractions.
    }
    \myStep{multiply}{%
        Multiply \gap{both} the numerator and denominator 
        of the fractions by the factors needed to make their \gap{denominators}
        equal to the \gap{LCD}.
    }
    \myStep{add/subtract}{%
        Add (or subtract) the \gap{numerators}.
        Write it over the LCD. 
    }
    \myStep{excluded values}{The excluded values come from \gap{LCD $\ne 0$}}
    \myStep{simplify}{%
        Simplify 
        and write the domain as \gap{$x \ne$ excluded values}.
    }
\end{myConceptSteps}

\begin{tcolorbox}[center,colback=white,width=6.5in,]
    Subtraction is similar, 
    but remember to \gap{distribute} any minus signs.
    This is easy to forget!
\end{tcolorbox}
