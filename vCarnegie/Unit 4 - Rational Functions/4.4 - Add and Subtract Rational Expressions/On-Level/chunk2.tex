\section{Adding and subtracting rational expressions}


\begin{myConceptSteps}{To add or subtract two rational expressions\dots}
    \myStep{factor}{Factor all the \gap{denominators}.}
    \myStep{multiply}{%
        Multiply the \gap{numerator} and \gap{denominator}
        of each fraction by the denominator 
        of the \gap{other} \gap{fraction}.
        \begin{itemize}
            \item This will give you two fractions with a \gap{common} denominator.
        \end{itemize}
    }
    \myStep{combine numerators}{%
        Add (or subtract) the \gap{numerators} 
        as a \gap{single} \gap{fraction}
        over the common denominator. 
    }
    \myStep{excluded values}{The excluded values come from \gap{denominator $\ne 0$}}
    \myStep{simplify}{%
        Simplify 
        and write the domain as \gap{$x \ne$ excluded values}.
    }
\end{myConceptSteps}

\myProblems[Add or subtract these rational expressions.]
    {
        $
        \frac{2}{x} % 10/5x
        +
        \frac{3}{5} % 3x/5x = (10+3x)/5x
        $
    }
    {
        $
        \frac{5}{x+1} % 15/3(x+1)
        -
        \frac{2}{3}   % 2(x+1)/3(x+1) = (15-2x-2)/3(x+1) = (13-2x)/3(x+1)
        $
    }
    {1.75in}
    [\myAnswer{$\frac{10+3x}{10x} ; x \ne 0$}]
    [\myAnswer{$\frac{13-2x}{3(x+1)} ; x \ne -1$}]

    \myProblems
    {
        $
        \frac{2}{(3 x + 6)}  
        +
        \frac{5}{(x - 4)} 
        %
        % 2(x-4) + 5(3)(x+2) = 2x-8+15x+30 = 17x + 22
        $
    }
    {
        $
        \frac{2x}{x-5} 
        -
        \frac{1}{x+3}  
        $
    }
    {1.75in}
    [\myAnswer{$\frac{17 x + 22}{3 (x - 4) (x + 2)} ; x \ne -2, 4$}]
    [\myAnswer{$\frac{2 x^2 + 5 x + 5}{(x - 5) (x + 3)} ; x \ne -3, 5$}]
