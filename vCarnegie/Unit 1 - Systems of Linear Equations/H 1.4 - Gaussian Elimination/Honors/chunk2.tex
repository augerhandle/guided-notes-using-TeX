\section{Gaussian Elimination}

\begin{myConceptSteps}{To transform a system to \gap{row} \gap{echelon} \gap{form}\dots}
  \myStep{left-to-right}{%
      Transform \gap{columns} starting at the
      \gap{left} and moving \gap{right}.
      }
  \myStep{ones}{%
      Use {\bfseries\itshape row operations} to get \gap{$1$s} on the diagonal.
      }
  \myStep{zeros}{%
      Use {\bfseries\itshape row operations} to get \gap{$0$s} below the diagonal.
    }
  \myStep{column-by-column}{%
    Finish steps 2 and 3
    \gap{before} moving to the next column.
}
\end{myConceptSteps}
