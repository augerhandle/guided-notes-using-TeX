\section{Graphing a system of 2-variable inequalities}

\begin{myConceptSteps}{To sketch the solution of a system of 2-variable inequalities\dots}
  \myStep{each inequality}{%
    Sketch the solution of \gap{each} inequality.
  }
  \myStep{overlap}{%
    The solution is where 
    the shaded regions \gap{overlap}.
    If there's \gap{no} overlap,
    there is \gap{no solution}.
}
\end{myConceptSteps}
