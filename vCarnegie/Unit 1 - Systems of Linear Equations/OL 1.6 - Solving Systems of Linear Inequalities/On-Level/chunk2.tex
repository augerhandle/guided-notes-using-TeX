\section{Graphing a system of 2-variable inequalities}

\begin{myConceptSteps}{To sketch the solution of a system of 2-variable inequalities\dots}
  \myStep{each inequality}{%
    Sketch the solution of \gap{each} inequality:
    \renewcommand{\arraystretch}{1.5}
    \begin{center}
      \begin{tabular}{p{1in}cc}
        {\itshape inequality} & {\itshape kind of line?} & {\itshape shade where?}\\
        \toprule
        {\large $y\ge\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{above} \\
        {\large $y>\cdots$} & \gap{dashed} & \gap{above} \\
        {\large $y\le\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{below\,} \\
        {\large $y<\cdots$} & \gap{dashed} & \gap{below\,} \\
      \end{tabular}
    \end{center}
  }
  \myStep{overlap}{%
    The solution is where 
    the shaded regions \gap{overlap}.
    \begin{itemize}
      \item If there's \gap{no} overlap, there is \gap{no solution}.
    \end{itemize}
}
\end{myConceptSteps}
