\section{Graphing a 2-variable inequality}

\begin{myConceptSteps}{To sketch the solution of a 2-variable inequality\dots}
  \myStep{rewrite}{%
    Write the inequality in \gap{slope}-\gap{intercept} form
    (unless the boundary is a \gap{vertical} line).
    \gap{Flip} the inequality if you 
    multiply or divide by \gap{negatives}.
  }
  \myStep{Boundary line and shading}{%
    Sketch the boundary \gap{line} and add \gap{shading}:
    \begin{center}
      \renewcommand{\arraystretch}{1.45}
        \begin{tabular}{p{1in}cc}
        \multicolumn{3}{c}{\myEmph{inequalities with $y$}}\\[0.5\onelineskip]
        {\small\itshape inequality} & {\small\itshape kind of line?} & {\small\itshape shade where?}\\
        \midrule
        {\large $y\ge\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{above} \\
        {\large $y>\cdots$} & \gap{dashed} & \gap{above} \\
        {\large $y\le\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{below\,} \\
        {\large $y<\cdots$} & \gap{dashed} & \gap{below\,} \\
      \end{tabular}
    \end{center}
    \vspace{-1\onelineskip}
    \begin{center}
      \renewcommand{\arraystretch}{1.45}
        \begin{tabular}{p{1in}cc}
        \multicolumn{3}{c}{\myEmph{inequalities with only $x$}}\\
        {\small\itshape inequality} & {\small\itshape kind of line?} & {\small\itshape shade where?}\\
        \midrule
        {\large $x\ge\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{right} \\
        {\large $x>\cdots$} & \gap{dashed} & \gap{right} \\
        {\large $x\le\cdots$} & \gap{solid\,\,\,\,\,\,} & \gap{left\,} \\
        {\large $x<\cdots$} & \gap{dashed} & \gap{left\,} \\
      \end{tabular}
    \end{center}
  }
\end{myConceptSteps}
