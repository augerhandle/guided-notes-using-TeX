\begin{myConcept}{To solve $3\times3$ systems of equations on a {\scshape TI-84}\dots}
    \myEmph{Enter the augmented matrix.}
    \begin{enumerate}%[nosep]
        \item Press \myButton{2ND} \myButton{MATRIX}.
        \item Move the cursor right to {\ttfamily [EDIT]}.
        \item Pick one of the matrix names (usually {\ttfamily [A]}). 
            Then press \myButton{ENTER}.
        \item Enter the augmented matrix dimensions, replacing any old dimensions.
            \begin{itemize}
                \item Enter \#rows (usually 3) \myButton{ENTER}.
                \item Enter \#columns (usually 4) \myButton{ENTER}.
            \end{itemize}
        \item Enter the numbers in the augmented matrix,
            row by row.
            \begin{itemize}
                \item Press \myButton{ENTER} after each number.
                \item Use the arrow keys to move around.
            \end{itemize}
        \item When the matrix is full, press \myButton{2ND} \myButton{QUIT}.
        \item Check the matrix.
            \begin{itemize}
                \item Press \myButton{2ND} \myButton{MATRIX}.
                \item Select the matrix you just created (usually {\ttfamily [A]}),
                    and then press \myButton{ENTER}.
                \item Make sure all the numbers are what you entered.
            \end{itemize}
    \end{enumerate}
    %
    \myEmph{Convert the matrix to reduced row-echelon form (RREF).}
    \begin{enumerate}
        \item Press \myButton{ENTER} \myButton{MATRIX}.
        \item Use the arrow keys to move the cursor right to {\ttfamily [MATH]}.
        \item Use the arrow keys to move the cursor down to {\ttfamily B:rref(},
            and press \myButton{ENTER}.
        \item Choose your matrix.
            \begin{itemize}
                \item Press \myButton{2ND} \myButton{MATRIX}.
                \item Select your matrix (usually {\ttfamily [A]}), 
                    and then press \myButton{ENTER}.
                \item Press \myButton{)}, and then press \myButton{ENTER} again.
            \end{itemize}
        \item The calculator will display the reduced row-echelon form 
            of the system.
    \end{enumerate}
    %
    \myEmph{The answer.}
    \begin{itemize}
        \item If there are three {\ttfamily 1}s on the diagonal of the matrix, 
            the variables equal (in order) the 
            \myEmph{numbers in the last column}.
        \item If the last row is {\ttfamily 0 0 0 0} (all {\ttfamily 0}s), 
            there are an \myEmph{''infinite number of solutions''}.
        \item If the last row is {\ttfamily 0 0 0 $\blacksquare$}, 
            there is \myEmph{''no solution''}.
    \end{itemize}
\end{myConcept}
