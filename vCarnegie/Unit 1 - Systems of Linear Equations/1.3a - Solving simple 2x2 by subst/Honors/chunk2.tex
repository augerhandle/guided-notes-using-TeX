\section{Solving Simple Systems by Substitution}

\begin{myConceptSteps}{To solve a simple system by substitution\dots}
    \myStep{first equation}{Pick an equation that is \gap{already} \gap{solved} for one of the variables.}
    \myStep{substitute \#1}{Replace that variable in the \gap{other} equation.}
    \myStep{solve}{Solve that equation for the \gap{second} variable. You will get a number.}
    \myStep{substitute \#2}{Substitute that number into the equation in Step-1 and simplify.}
\end{myConceptSteps}

\myProblems[Solve these simple systems by substitution.]
{
    \vspace{-\baselineskip}
    \begin{align*}
        2x + 2y = 6 \\
        y = x-5
    \end{align*}
}
{
    \vspace{-\baselineskip}
    \begin{align*}
        x = y-5 \\
        2y-6 = -2x
    \end{align*}
}
