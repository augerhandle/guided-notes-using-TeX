\myProblemsWithContent
{
    \small
    $
        f(x) 
            =  
            \cfrac {x^3 -3x^2 + 2x} {x^2 + x} 
            =
            \cfrac {x(x-1)(x-2)} {x(x+1)}
    $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{0}{0} =  $
            \fbox{ no y-int. }
        }
        \\[\myStepSkip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n=d+1$ 
            \myAndThen 
            \fbox{slant asymp.: $y = x-4$}
        }
        \vspace{10em}
        \\[\myStepSkip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ 
                f_{\scalebox{0.5}{simp}}(x) 
                = \cfrac {\cancel{x}(x-1)(x-2)} {\cancel{x}(x+1)}
                = \cfrac {(x-1)(x-2)} {(x+1)}
            $ 
        }
        \\[\myStepSkip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            \cancel{$(x)$} 
            \myAndThen 
            \fbox{hole: $(0,2)$}
        }
        \\[\myStepSkip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simp}}(x)$ denom $=0$   
            \myAndThen 
            VA @ $\fbox{x=-1}$        
        }
        \\[\myStepSkip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            $x \ne$ holes \& VA 
            \myAndThen
            \fbox{$x \ne -1, 0$}
        }
        \\[\myStepSkip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            $ f_{\scalebox{0.5}{simp}}(x) $  numer $=0$
            \myAndThen 
            \fbox{$x=1, 2$}
        }
        \\[\myStepSkip]
}
