\myWideProblemWithContent[%
    Use the \myEmph{Lesson 7.3a} procedure 
    to find the attributes of these rational functions.
    ]
{
    $ f(x) =\cfrac {2x^2+6x+4} {x^2+x-2} $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{4}{-2} = \fbox{-2} $
        }
        \\[2\onelineskip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n = 2$, $d = 2$, $a=2$, $b=1$
            \myAndThen 
            $n=d$ 
            \myAndThen 
            $y = \frac{a}{b}= \frac{2}{1} $
            \myAndThen 
            \fbox{$y = 2$}
        }
        \\[2\onelineskip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ f_{\scalebox{0.5}{factored}}(x) = \frac {2(x+1)(x+2))} {(x+2)(x-1)} $ 
            \\[0.5\onelineskip]
            $ 
                f_{\scalebox{0.5}{simplified}}(x) 
                = \frac {2(x+1)\cancel{(x+2)})} {\cancel{(x+2)}(x-1)}
                = \frac {2(x+1)} {(x-1)} 
            $ 
            \myAndThen[1em]
            canc. factors: $(x+2)$
        }
        \\[2\onelineskip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            \cancel{$(x+2)$} for $f_{\scalebox{0.5}{simplified}}(x)$
            \myAndThen 
            hole @ $\fbox{x=-2}$
        }
        \\[2\onelineskip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simplified}}(x)$ denom $=0$   
            \myAndThen 
            VA @ $\fbox{x=1}$        
        }
        \\[2\onelineskip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            holes \& VA 
            \myAndThen
            \fbox{$x \ne 1, -2$}
        }
        \\[2\onelineskip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            $ f_{simp}(x) $  numer $=0$
            \myAndThen 
            \fbox{$x=-1$}
        }
        \\[2\onelineskip]
}
