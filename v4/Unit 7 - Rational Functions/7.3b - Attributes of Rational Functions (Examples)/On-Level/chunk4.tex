\myWideProblemWithContent[%
    Use the \myEmph{Lesson 7.3a} procedure 
    to find the attributes of these rational functions.
    ]
{
    $ f(x) =\cfrac {-2x+4} {x^2-x-6} $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{4}{-6} =$ \fbox{$-\frac{2}{3}$}
        }
        \\[2\onelineskip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n = 1$, $d = 2$,
            \myAndThen 
            $n<d$ 
            \myAndThen 
            \fbox{$y = 0$}
        }
        \\[2\onelineskip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ f_{\scalebox{0.5}{factored}}(x) = \frac {-2(x-2)} {(x+2)(x-3)} $ 
            \\[0.5\onelineskip]
            no factors cancel
        }
        \\[2\onelineskip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            no cancelled factors 
            \myAndThen 
            \fbox{no holes}
        }
        \\[2\onelineskip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simplified}}(x)$ denom $=0$   
            \myAndThen 
            VA @ $\fbox{x=-2,3}$        
        }
        \\[2\onelineskip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            holes \& VA 
            \myAndThen
            \fbox{$x \ne -2, 3$}
        }
        \\[2\onelineskip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            $ f_{simp}(x) $  numer $=0$
            \myAndThen 
            \fbox{$x=2$}
        }
        \\[2\onelineskip]
}
