\myProblemWithContent
{
    \small
    $
        f(x) 
            =  
            \cfrac {x^2-x-2} {2x+4} 
            =
            \cfrac {(x+1)(x-2)} {2(x+2)}
    $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{-2}{4} =  $
            \fbox{y-int: $ -\frac{1}{2} $ }
        }
        \\[2\onelineskip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n>d$ 
            \myAndThen 
            \fbox{no HA}
        }
        \\[2\onelineskip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ 
                f_{\scalebox{0.5}{simp}}(x) 
                = \cfrac {(x+1)(x-2)} {2(x+2)}
            $ 
        }
        \\[2\onelineskip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            no factors cancel
            \myAndThen 
            \fbox{no holes}
        }
        \\[2\onelineskip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simp}}(x)$ denom $=0$   
            \myAndThen 
            VA @ $\fbox{x=-2}$        
        }
        \\[2\onelineskip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            $x \ne$ holes \& VA 
            \myAndThen
            \fbox{$x \ne -2$}
        }
        \\[2\onelineskip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            $ f_{\scalebox{0.5}{simp}}(x) $  numer $=0$
            \myAndThen 
            \fbox{$x=-1, 2$}
        }
        \\[2\onelineskip]
}
