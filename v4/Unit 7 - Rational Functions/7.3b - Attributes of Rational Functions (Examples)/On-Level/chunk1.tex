\myProblemsWithContent[%
    Find the attributes of these already-factored rational functions.
    ]
{
    \small
    $
        f(x) 
            =  
            \cfrac {3x^2-15x+12} {x^3 - 2x^2 - 8x} 
            =
            \cfrac {3(x-4)(x-1)} {x(x+2)(x-4)}
    $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{12}{0} = \fbox{y-int.: none} $
        }
        \\[\myStepSkip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n<d$ 
            \myAndThen 
            \fbox{HA: $y=0$}
        }
        \\[\myStepSkip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ 
                f_{\scalebox{0.5}{simp}}(x) 
                = \cfrac {3\cancel{(x-4)}(x-1)} {x(x+2)\cancel{(x-4)}} 
                = \cfrac {3(x-1)} {x(x+2)} 
            $ 
        }
        \\[\myStepSkip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            \cancel{$(x-4)$} 
            \myAndThen 
            hole @ $\fbox{x=4}$
        }
        \\[\myStepSkip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            simp denom $=0$   
            \myAndThen 
            2 VAs @ \fbox{$x=0$, $x=-2$}        
        }
        \\[\myStepSkip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            $x \ne$ holes \& VA 
            \myAndThen
            \fbox{$x \ne -2, 0, 4$}
        }
        \\[\myStepSkip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            simp  numer $=0$
            \myAndThen 
            \fbox{$x=1$}
        }
        \\[\myStepSkip]
}
{
    \small
    $
        f(x) 
        =  
        \cfrac {2x^2+4x-6} {x^2+x-2} 
        =  
        \cfrac {2(x+3)(x-1)} {(x-1)(x+2)}
    $
\tcblower
    %
    %
    \myEmph{Step 1. $y$--intercept}\\
        \whenTEACHER{
            \small
            $ x = 0 $ 
            \myAndThen
            $ y = f(0) = \frac{-6}{-2} = \fbox{y-int.: y=3} $
        }
        \\[\myStepSkip]
    \myEmph{Step 2. horizontal\whenHONORS{ or slope} asymptote}\\
        \whenTEACHER{
            \small
            $n=d$ 
            \myAndThen 
            $y = \frac{2}{1} $
            \myAndThen
            \fbox{HA: $y=2$}
        }
        \\[\myStepSkip]
    \myEmph{Step 3. factor \& simplify}\\
        \whenTEACHER{
            \small
            $ 
                f_{\scalebox{0.5}{simp}}(x) 
                = \cfrac {2(x+3)\cancel{(x-1)}} {\cancel{(x-1)}(x+2)} 
                = \cfrac {2(x+3)} {(x+2)} 
            $ 
        }
        \\[\myStepSkip]
    \myEmph{Step 4. holes}\\
        \whenTEACHER{
            \small
            \cancel{$(x-1)$} 
            \myAndThen 
            hole @ $\fbox{x=1}$
        }
        \\[\myStepSkip]
    \myEmph{Step 5. vertical asymptotes}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simp}}(x)$ denom $=0$   
            \myAndThen 
            1 VA @ \fbox{$x=-2$}        
        }
        \\[\myStepSkip]
    \myEmph{Step 6. domain}\\
        \whenTEACHER{
            \small 
            $x \ne$ holes \& VA 
            \myAndThen
            \fbox{$x \ne -2, 1$}
        }
        \\[\myStepSkip]
    \myEmph{Step 7. $x$--intercepts}\\
        \whenTEACHER{
            \small
            $f_{\scalebox{0.5}{simp}}(x)$  numer $=0$
            \myAndThen 
            \fbox{$x=-3$}
        }
        \\[\myStepSkip]
}
