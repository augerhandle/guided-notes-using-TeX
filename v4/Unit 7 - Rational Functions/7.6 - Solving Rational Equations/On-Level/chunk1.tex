\begin{myConceptSteps}{To solve an equation with rational expressions\dots}
    \myStep{LCD}{%
        Find the {\bfseries\itshape least common denominator}.
    }
    \myStep{excluded values}{%
        Find the \gap{excluded} values by finding the \gap{zeros} 
        of the LCD.
    }
    \myStep{clear the fractions}{%
        \myEmph{Multiply} \gap{both} sides of the equation by the \gap{LCD}.
        \begin{itemize}[nosep]
            \item There won't be any more fractions!
        \end{itemize}
    }
    \myStep{solve}{%
        Solve the resulting equation.
        \begin{itemize}[nosep]
            \item Ignore \cancel{cross out} any \gap{values} that are \gap{excluded}.
        \end{itemize}
    }
\end{myConceptSteps}

    $
        \frac{4}{x+2} = \frac{3}{x+6}
    $
}
{
    $
        \frac{1}{2} + \frac{1}{2(x-1)} = \frac{x-3}{x-1}
    $
}
