\section{Adding rational expressions}


\begin{myConceptSteps}{~to {\bfseries\itshape add} two rational expressions\dots}
    \myStep{common denominator}{
        If the fractions have \gap{different} denominators\dots
        \begin{itemize}[nosep]
            \item Factor the denominators.
            \item Find the \gap{LCD} of the two rational expressions.
            \item \gap{Multiply} the numerators and denominators 
            by factors needed to make their denominators equal to the \gap{LCD}.
        \end{itemize}
    }
    \myStep{add}{%
        Add the \gap{numerators}.
        \begin{itemize}[nosep]
            \item[$\circ$]{\bfseries\itshape Do not} add the denominators.
            The denominator becomes the \gap{LCD}. 
        \end{itemize}
    }
    \myStep{simplify}{%
        Simplify the expression. Write the \gap{domain} as part of the answer.
    }
\end{myConceptSteps}
