\section{Finding Attributes}

\begin{myConceptSteps}{To find the attributes of a rational function\dots}
    \myStep{$y$-intercept}{Evalutate $y=f(0)$ by substituting $x=0$.}
    \myStep{horizontal\whenHONORS{ or slant }asymptote}{%
        \begin{itemize}
            \item Find these numbers:
                \begin{itemize}
                    \item the \myEmph{degree} of the numerator \gap{$n$} and denominator \gap{$d$}.
                    \item the \myEmph{leading coefficient} of the numerator \gap{$a$} and denominator \gap{$b$}.
                \end{itemize}
            \item Find the equation of the asymptote as follows:
                \begin{itemize}
                    \whenHONORS{
                        \item when \gap{$ \bm{n < d} $}, \hspace{4em}\myEmph{horizontal} asymptote: \gap{$y=0$}
                        \item when \gap{$ \bm{n = d} $}, \hspace{4em}\myEmph{horizontal} asymptote: \gap{$y = \frac{a}{b}$}
                        \item when \gap{$ \bm{n = d+1} $}, \hspace{0.55em}\myEmph{slant} asymptote: \gap{long} \gap{division} {\itshape quotient only}
                        \item when \gap{$ \bm{n > d+1} $}, \hspace{0.5em}\myEmph{none}
                    % \whenONLEVEL{%
                    %     \item when \gap{$ \bm{n > d} $}, \hspace{2.5em}no horizontal asymptote
                    % }
                    }
                    \whenONLEVEL{
                        \item when \gap{$ \bm{n < d} $}, \hspace{1em}\myEmph{horizontal} asymptote: $y=0$
                        \item when \gap{$ \bm{n = d} $}, \hspace{1em}\myEmph{horizontal} asymptote: $y = \frac{a}{b}$
                        \item when \gap{$ \bm{n > d} $}, \hspace{1em}n\myEmph{none}
                    }
                \end{itemize}
        \end{itemize}
    }
    \myStep{factor \& simplify}{
        \begin{itemize}
            \item {\itshape factored form}: \gap{Factor} the {numerator} and {denominator}.
            \item {\itshape simplified function}: \gap{Cancel} all factors that are the same.
            \item {\itshape cancelled factors}: \gap{write} down the cancelled factors.
        \end{itemize}
    }
    \myStep{holes}{%
        \begin{itemize}
            \whenONLEVEL{
                \item Set each \gap{cancelled} factor to zero. 
                \item \gap{Solve} for {$x$} in each of those equations.
            }
            \whenHONORS{
                \item You get one $(x,y)$ point for each cancelled factor. These are the \gap{holes}.
                \begin{itemize}
                    \item Set each \gap{cancelled} factor to zero, and \gap{solve} for {$x$} to get the $x$-coordinate.
                    \item \gap{Substitute} $x$ into the {\itshape simplified function} to get the  $y$-coordinate.
                \end{itemize}
            }
        \end{itemize}
    }
    \myStep{vertical asymptotes}{%
        \begin{itemize}
            \item Use the {\itshape simplified function}.
            \item Set \gap{denominator} factors to zero. 
            \item \gap{Solve} for {$x$} in each of those equations.
        \end{itemize}
    }
    \myStep{domain}{$\bm{x \ne}$ \gap{holes} and \gap{vertical} asymptotes}
    \myStep{$x$-intercepts}{%
        \begin{itemize}
            \item Use the \gap{simplified} function.
            \item Set \gap{numerator} factors to zero. 
            \item \gap{Solve} for {$x$} in each of those equations.
        \end{itemize}
    }
\end{myConceptSteps}
