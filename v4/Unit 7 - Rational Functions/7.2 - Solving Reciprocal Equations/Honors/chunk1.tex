\begin{myConceptSteps}{To solve reciprocal equations\dots}
    \myStep{isolate}{Get the reciprocal expression \gap{by} \gap{itself}.}
    \myStep{denominator}{Multiply \gap{both} \gap{sides} of the equation by the denominator.
        \begin{itemize}[nosep]
            \item[$\circ$] You might have to \gap{distribute}.
        \end{itemize}
    }
    \myStep{Solve}{Solve the \gap{simpler} equation.}
\end{myConceptSteps}

\myProblems[Solve these reciprocal equations.]
{
    $\frac{4}{x} = 12$
    % 4 = 12x 
    % 4/12 = x 
    % 1/3 = x
}
{
    $\frac{24}{x+1} = 4$
    % 24 = 4(x+1) 
    % 24 = 4x + 4 
    % 20 = 4x 
    % 5 = x
}
{1in}

\myProblems
{
    $\frac{2}{x-3} + 8  = 10$
    % 2/(x-3) = 2 
    % 2 = 2(x-3) 
    % 2 = 2x - 6 
    % 8 = 2x 
    % 4 = x
}
{
    $\frac{10}{x-5} - 4 = -9$
    % 10/(x-5) = -9 + 4 = -5 
    % 10 = -5(x-5) 
    % 10 = -5x + 25
    % -15 = -5x 
    % 3 = x
}
{2in}
