\section{How to solve absolute value equations}

\begin{myConceptSteps}{To solve square root equations\dots}
    \myStep{isolate}{Get the square root expression \gap{by} \gap{itself}.}
    \myStep{square}{Square both sides of the equation.}
    \myStep{solve}{solve the \gap{simpler} equation}
    \myStep{check}{Make sure you don't get extraneous solutions.}
\end{myConceptSteps}
