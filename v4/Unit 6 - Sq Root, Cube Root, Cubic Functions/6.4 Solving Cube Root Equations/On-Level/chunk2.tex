\myProblems[Solve these cube root equations.]
{
    $\myRoot[3]{x} - 2 = 3$
    % root3(x) = 3 + 2 
    % root3(x) = 5 
    % x = (5)^3 
    % x = 125
}
{
    $5 \myRoot[3]{x} + 6 = -9$
    % 5 root3(x) = -9 - 6
    % 5 root3(x) = -15 
    %   root3(x) = -3
    %   x = (-3)^3 
    %   x = -27
}{1.5in}[\raggedleft\gap{$x=125$}][\raggedleft\gap{$x=-27$}]
