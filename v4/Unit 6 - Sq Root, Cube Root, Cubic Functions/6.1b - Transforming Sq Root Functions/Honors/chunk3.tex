\myWideProblemWithContent[%
    Sketch the graphs of these functions and then fill in the blanks.
    ]
{
    $g(x) = \myRoot{x - 3} -4$
    \hfill 
    ${a} =$ \gap{$1$},\quad ${h} =$ \gap{$3$},\quad ${k} =$ \gap{$-4$}
    \tcblower 
    \begin{minipage}{0.34\textwidth}
        \centering
        \begin{myTikzpictureGrid}{0.4} [-1]{10}[-5]{3}
            \tkzFct[ solid, ultra thick, samples=1000, domain=-6:10,]{sqrt(x-3)-4}
        \end{myTikzpictureGrid}
        \end{minipage}
    \begin{minipage}{0.65\textwidth}
        \vspace{0.5\baselineskip}
        end point coordinates:~\dotfill\gap{$(3,-4)$}\\[0.5\baselineskip]
        domain:~\dotfill\gap{$x\ge3$}\\[0.5\baselineskip]
        range:~\dotfill\gap{$y\ge -4$}\\[0.5\baselineskip]
