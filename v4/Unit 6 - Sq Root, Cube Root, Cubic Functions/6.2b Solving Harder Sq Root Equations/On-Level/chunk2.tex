\section{Two Square Roots}

\begin{tcolorbox}[center,width=4.5in,colback=white]
    When there are \gap{two} square roots,
    isolate them to \gap{opposite} sides of the equation.
\end{tcolorbox}



\myProblems[Solve these square root equations.]
{
    $\myRoot{5v + 17} = \myRoot{4v+10}$
}
{
    $\myRoot{\frac{x}{3}} = \myRoot{3x-24}$
}{3in}[\raggedleft\gap{$v=-7$}][\raggedleft\gap{$x=9$}]
