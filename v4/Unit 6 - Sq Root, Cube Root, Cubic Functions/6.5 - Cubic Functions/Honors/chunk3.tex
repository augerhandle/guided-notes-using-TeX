\myProblemsWithContent[%
    Given these transformations,
    find $a$, $h$, $k$, and write the formula for $g(x)$.
]
{
    \begin{itemize}[nosep,fullwidth]
        \item reflection across the $x$-axis 
        \item compression by $\frac{3}{4}$
        \item shift down by 2
        \item shift right by 4
    \end{itemize}
    \tcblower 
    \large
    $a$=\gap{$-\frac{3}{4}$}, $h$=\gap{4}, $k$=\gap{-2}\\[1\baselineskip] 
    $g(x) =$ \gap{$-\frac{3}{4}(x-4)^3-2$}
}
{
    \begin{itemize}[nosep,fullwidth]
        \item stretch by $\frac{4}{3}$ 
        \item shift left by 5
        \item shift up by 9
    \end{itemize}
    \tcblower 
    \large
    $a$=\gap{$\frac{4}{3}$}, $h$=\gap{-5}, $k$=\gap{9}\\[1\baselineskip] 
    $g(x) =$ \gap{$\frac{4}{3}(x+5)^3 + 9$}
}
