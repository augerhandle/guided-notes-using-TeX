\myWideProblemWithContent[%
    Use a graphing calculator for this function. Sketch the graph here. Find the attributes.
]
{
    $ g(x) = \myRoot{x} + 2 $
    \tcblower
    \myTwoMinipages{
        \begin{myTikzpictureGrid}{0.45} [-2]{17}[-1]{7}
            % \tkzText(13,8){$x - y = 0$}
            \whenTEACHER{
                \tkzFct[ color=red, solid, thick, samples=200, domain=0:17,]{(x)**0.5 + 2}
            }
            % \tkzText(-13,7.25){$x + 2y = 6$}
        \end{myTikzpictureGrid}
    }{
        \begin{itemize}[fullwidth,]
            \item coordinates of the end point: \gap{$(0,2)$}
            \item domain: \gap{$x\ge 0$}
            \item range: \gap{$y\ge 2$}
            \item left end behavior: as $x\rightarrow$\gap{$0$}, $y\rightarrow$\gap{$0$}
            \item right end behavior: as $x\rightarrow$\gap{$\infty$}, $y\rightarrow$\gap{$\infty$}
        \end{itemize}
    }
}
