\begin{myVocabulary}
    \myDefinition{slope}{
        change in \gap{$y$} divided by change in \gap{$x$} 
        (rise over run)
        }
    \myDefinition{$y$-intercept}{the point where the line crosses the \gap{$y$-axis}}
    \myDefinition{slope-intercept form}{\Large $y=mx+b$}
\end{myVocabulary}
