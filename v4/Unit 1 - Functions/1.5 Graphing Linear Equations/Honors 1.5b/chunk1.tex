\begin{myConceptSteps}{~To graph any linear equation\dots}
    \myStep{rewrite}{\gap{Rewrite} the equation in slope-intercept form.}
    \myStep{balance}{%
        Keep the $=$ symbols {\bfseries\itshape aligned}. 
        (Draw a \gap{vertical} line.)
        }
    \myStep{graph}{Sketch the graph the line using $m$ and $b$..)}
\end{myConceptSteps}


\myProblems[Rewrite these equations in slope-intercept form, and find the slope and $y$-intercept.]
{
    \centering
    $2x + 3y = 12$
}
{
    \centering
    $10 + 7y + 2x = 2y + 20 - x$
    % 7y + 2x = -x + 2y + 10
    % 5y + 2x = -x + 10
    % 5y = -3x + 10
    % y = -(3/5)x + 2
}{3in}
