\myLesson{Graphing Linear Equations}[5][a]

\begin{myObjectives}
    \myObjective{find}{the {\bfseries\itshape slope} and {\bfseries\itshape $y$-intercept} of an equation in slope-intercept form}
    \myObjective{graph}{an equation in slope-intercept form}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{slope}{
        change in \gap{$y$} divided by change in \gap{$x$} 
        ({\bfseries\itshape rise} over {\bfseries\itshape run})
    }
    \myDefinition{$y$-intercept}{the point where the line crosses the \gap{$y$-axis}}
    \myDefinition{slope-intercept form}{\Large $y=mx+b$}
\end{myVocabulary}
