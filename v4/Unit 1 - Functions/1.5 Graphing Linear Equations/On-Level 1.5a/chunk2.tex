\section{Graphing equations in slope-intercept form}

\begin{myConceptSteps}{To graph $y=mx+b$\dots}
    \myStep{$y$-intercept}{Find the $y$-intercept (\gap{$b$}).}
    \myStep{first point}{Draw the $y$-intercept as a \gap{point} on the \gap{$y$}-axis.}
    \myStep{slope}{Find the slope (\gap{$m$}). Rewrite it as a fraction: \gap{rise} over \gap{run}.}
    \myStep{second point}{
            \begin{itemize}[fullwidth]
                \item If $m$ is \gap{positive}, go \gap{up} by rise and {\bfseries\itshape right} by run. Draw the point there.
                \item If $m$ is \gap{negative}, go \gap{down} and {\bfseries\itshape right}. Draw the point there.
            \end{itemize}
        }
    \myStep{line}{Draw a line that goes through the two points.}
    \end{myConceptSteps}
