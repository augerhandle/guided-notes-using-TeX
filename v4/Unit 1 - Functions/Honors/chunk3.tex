\section{The Vertical Line Test}

\begin{myConceptSteps}{~to determine if a relation is a function\dots}
    \myStep{graph}{\gap{Graph} the relation. }
    \myStep{vertical lines}{Imagine a \gap{vertical line} at every $x$-value. }
    \myStep{$y$-values}{How many places does each line \gap{intersect} the graph?}
    \myStep{test}{For functions, the lines must intersect \gap{only once}.}
\end{myConceptSteps}
