\section{Plotting Points}

\myCenteredBox[width=5in]{
    To plot points you need to keep the \gap{$x$'s} and \gap{$y$'s} straight.
}

\myWideProblemWithContent[\\Plot these points given their $x$- and $y$-coordinates.]
{
    \begin{minipage}{0.35\textwidth}
        \centering
        \begin{tabular}{c}
            \textit{point-1}\\
            \hline
            $x$:  \texttt{3} \\
            $y$:  \texttt{4} \\
        \end{tabular}
        %
        \hfil 
        %
        \begin{tabular}{c}
            \textit{point-2}\\
            \hline
            $x$:  \texttt{-4} \\
            $y$:  \texttt{-2} \\
        \end{tabular}
    \end{minipage}
    %
    \hfil 
    %
    \begin{minipage}{0.59\textwidth}
        \centering
        \begin{myTikzpictureGrid}{0.3} {5}{5}
        \end{myTikzpictureGrid}
    \end{minipage}
}

{
    \begin{minipage}{0.49\textwidth}

            \hline
            -3 &  0 \\
            -3 & -2 \\
            0 &  1 \\
            2 &  2 \\
            4 &  3 \\
        \end{tabular}
