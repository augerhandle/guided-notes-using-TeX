\section{Evaluating a Function From Its Graph}

\begin{myConceptSteps}{To \gap{evaluate} a function given its input\dots} 
    \myStep{input}{Find the \gap{input} value on the $x$-axis.}
    \myStep{first arrow}{Draw a \gap{vertical} arrow from the $x$-axis to the curve.}
    \myStep{dot}{Draw a \gap{dot} where the arrow \gap{intersects} the graph.}
    \myStep{second arrow}{Draw a \gap{horizontal} arrow over to the $y$-axis.}
    \myStep{output}{Find the $y$-value where the arrow \gap{intersects} $y$-axis.}
\end{myConceptSteps}
