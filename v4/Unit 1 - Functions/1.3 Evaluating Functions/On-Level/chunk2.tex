\section{Evaluating a Function From Its Expression}

\begin{myConceptSteps}{To \gap{evaluate} a function given its expression\dots} 
    \myStep{substitute}{replace the \gap{variable} with the input value}
    \myStep{simplify}{simplify the expression (Remember \gap{PEMDAS}.)}
\end{myConceptSteps}

\subsection{Evaluating functions (without a calculator)}

\myProblemsWithContent[Evaluate these functions at the specified input value.]
    {
        function: $f(x) = x^2 - 10$ \\
        input value: $x = -4$
        \vspace{1in}\\
        $f(-4) =$ \gap{6}
    }
    {
        function: $s(x) = 3x^2 - x$ \\
        input value: $x = -4$
        \vspace{1in}\\
        $s(-2) =$ \gap{14} % 3*(-2)**2 - (-2) = 3*(4) + 2 = 12 + 2 = 14
    }



\subsection{Evaluating functions (with a calculator)}

\myCenteredBox[width=6in]{
    When you use a calculator to simplify the expression,
    {\bfseries\itshape always, always}
    use \gap{parentheses} when you enter the input value.
}


\myProblemsWithContent[Evaluate these functions at the specified input value.]
    {
        function: $f(x) = x^2 + x - 6$ \\
        input value: $x = -4$
        \vspace{1in}\\
        $f(-4) =$ \gap{6}
    }
    {
        function: $s(x) = x^4 + 2x^2 - x + 5$ \\
        input value: $x = -4$
        \vspace{1in}\\
        $s(-1) =$ \gap{7} % 1 + 2 - 1 + 5 = 7
    }

\myCenteredBox[width=6in]{
    {\bfseries\itshape Warning:}
    I cannot see your work on a calculator. 
    So if you make a silly mistake, You'll get the problem {\itshape wrong}.
    So, {\bfseries\itshape do it twice when you use a calculator!}
}
