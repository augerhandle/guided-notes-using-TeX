\section{Domain and Range of Discrete Relations}

\myCenteredBox{
    For {\bfseries\itshape discrete} relations (\gap{tables} or sets of \gap{points}), 
    \begin{itemize}
        \item{
            find the \gap{domain} by listing the $x$-values (no repeats!)
        }
        \item{
            find the \gap{range} by listing the $y$-values (no repeats!)
        }
    \end{itemize}
}

\myProblemsWithContent[Find the domain and range of these relations.]
{
    $
    \left\{
        (-1,-1), (0,0), (-1, 3), (1,2), (2,3) , (3,-2)
    \right\}
    $
}
{
    \begin{tabular}{r|r}
        $x$ & $y$ \\
        \hline\hline
        -1 &-1\\
        0&0\\
        1&1\\
        2&2\\
        3&1\\
        4&0\\
        5&-1\\
    \end{tabular}
    \vspace{0.75in}
}
