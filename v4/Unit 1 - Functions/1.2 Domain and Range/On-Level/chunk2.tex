\section{Domain and Range of Continuous Relations}


\myCenteredBox{
    If the relation is given as a \gap{graph}, 
    \begin{itemize}
        \item{
            find the {\bfseries\itshape domain} 
            by drawing the \gap{shadow} of the graph on the \gap{$x$}-axis.
        }
        \item{
            find the {\bfseries\itshape range} 
            by drawing the \gap{shadow} of the graph on the \gap{$y$}-axis.
        }
    \end{itemize}
}

\myProblemsWithContent[Find the {\bfseries\itshape domain} of the following relations.]
{
    \begin{myTikzpictureGrid}{0.35} [-1]{7}[-3]{4}
        \tkzDefPoint(1,3){A} 
        \tkzDrawPoint[shape=circle, size=6pt, color=black, fill=black, line width=1.5pt, ](A)
        \tkzDefPoint(6,-2){A} 
        \tkzDrawPoint[shape=circle, size=6pt, color=black, fill=white, line width=1.5pt, ](A)
        \tkzFct[ solid, ultra thick, samples=100, domain=1:6,]{4-x}
    \end{myTikzpictureGrid}
    \vspace{0.6in}
}
{
    \begin{myTikzpictureGrid}{0.35} [-1]{12}[-1]{5}
        \tkzDefPoint(11,4){A} 
        \tkzDrawPoint[shape=circle, size=6pt, color=black, fill=black, line width=1.5pt, ](A)
        \tkzDefPoint(2,1){A} 
        \tkzDrawPoint[shape=circle, size=6pt, color=black, fill=white, line width=1.5pt, ](A)
        \tkzFct[ solid, ultra thick, samples=100, domain=-1:11,]{sqrt(x-2)+1}
    \end{myTikzpictureGrid}
}
