\section{Vertical Line Test}

\begin{myConceptSteps}{~to determine of a relation is a function\dots}
    \myStep{graph}{\gap{Sketch} the graph of the relation. }
    \myStep{vertical lines}{Consider a \gap{vertical line} at every $x$-value. }
    \myStep{$y$-values}{Count the number of points where the lines \gap{intersect} the graph.}
    \myStep{test}{For functions, the lines must intersect \gap{only once} (or not at all).}
\end{myConceptSteps}
