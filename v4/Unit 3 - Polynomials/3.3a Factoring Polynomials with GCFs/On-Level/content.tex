    \myObjective{factor}{a polynomial by removing the greatest common factor (\gap{GCF})}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{coefficient}{a \gap{number} in front of variables}
    \myDefinition{factors}{things that are \gap{multiplied} together}
    \myDefinition{common factor}{a factor of \gap{several} terms}
    \myDefinition{greatest common factor}{the \gap{biggest} common factor you can find (GCF)}
\end{myVocabulary}

\myCenteredBox[width=4.5in]{
    When we {\itshape factor something}, 
    we \gap{rewrite} it as several pieces 
    \gap{multiplied} together.
