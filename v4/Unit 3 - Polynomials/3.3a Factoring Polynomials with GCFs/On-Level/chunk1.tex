\begin{myConceptSteps}{To factor a polynomial by removing the GCF\dots}
    \myStep{numbers}{%
        Find the \gap{largest} integer 
        that \gap{divides} \gap{exactly} into the coefficients of every term.
        \begin{itemize}[nosep]
            \item This will {\bfseries\itshape never} be larger than the \gap{smallest} coefficient.
        \end{itemize}
        }
    \myStep{variables}{%
        Find the \gap{largest} exponent of each variable in every term.
        }
    \myStep{GCF}{Multiply those results to get the greatest common factor.}
    \myStep{rewrite}{%
        Rewrite the polynomial as the product of 
        \begin{itemize}
            \item the \gap{GCF} and 
            \item a new polynomial 
            % (in parentheses)
            where each term is the original term \gap{divided} \gap{by} the GCF
            (Write this polynomial between \gap{parentheses}.)
        \end{itemize}
        }
\end{myConceptSteps}
