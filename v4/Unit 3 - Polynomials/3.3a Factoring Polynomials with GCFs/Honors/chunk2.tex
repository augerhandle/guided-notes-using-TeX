\section{Finding The Greatest Common Factor (GCF)}

\begin{myConceptSteps}{To find the GCF of a polynomial\dots}
    \myStep{factors}{\gap{Factor} each monomial.}
    \myStep{common factors}{%
        Identify each factor that occurs in \gap{every} monomial.
        These are the \gap{common} \gap{factors}.
        }
    \myStep{GCF}{\gap{Multiply} the common factors together.}
\end{myConceptSteps}


\myProblems[Find the GCF of these polynomials.]
    {
        \centering $ 10 x^2y^3 + 15x^3y $
    }
    {
        \centering $27 a^2b^5 - 6a^3b^2$
    }
    {2in}
