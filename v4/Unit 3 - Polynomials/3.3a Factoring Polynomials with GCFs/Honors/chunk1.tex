\section{Factoring Monomials}

\begin{myConceptSteps}{To factor a monomial\dots}
    \myStep{coefficient}{%
        Rewrite the {coefficient} as its {\bfseries\itshape prime factors}.
        \begin{itemize}[nosep]
            \item Use a \gap{factor} \gap{tree}.
        \end{itemize}
    }
    \myStep{variable}{%
        Expand the variable and its exponent.
        \begin{itemize}[nosep]
            \item Example: $x^3 \rightarrow x\,x\,x$
        \end{itemize}
    }
\end{myConceptSteps}
