\section{Factoring a Polynomials by Removing the GCF}

\begin{myConceptSteps}{To factor a polynomial by removing the GCF\dots}
    \myStep{GCF}{Find the \gap{GCF} of the polynomial.}
    \myStep{rewrite}{%
        Rewrite the polynomial as the product of 
        \begin{itemize}
            \item the \gap{GCF} and 
            \item a new polynomial 
            % (in parentheses)
            where each term is the original term \gap{divided} \gap{by} the GCF
            (Write this polynomial between \gap{parentheses}.)
        \end{itemize}
        }
\end{myConceptSteps}


\myProblems[Factor these polynomials by removing the GCF.]

    \centering $ 10 x^2y^3 + 15x^3y $
}
{
    \centering $27 a^2b^5 - 6a^3b^2$
}
