\dummypart{3}{Polynomials}
\myLesson{Factors, Roots, Solutions, Zeros, and $x$-intercepts}[5]

\begin{myObjectives}
    \myObjective{convert}{%
        between the following
        \begin{itemize}[nosep]
            \item the \gap{linear} \gap{factors} of $f(x)$
            \item the \gap{$x$-intercepts} of $f(x)$
            \item the \gap{zeros} of $f(x)$
            \item the \gap{roots} or \gap{solutions} of $f(x)=0$
        \end{itemize}
    }
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{$x$-intercepts of $f(x)$}{%
        points where the graph \gap{crosses} the $x$-axis
        ($y=0$)
    }
    \myDefinition{zeros of $f(x)$}{%
        $x$ values where $f(x)$ is \gap{zero}
    }
    \myDefinition{roots/solutions}{$x$ values you get when you solve $f(x)=0$}
    \myDefinition{linear factor}{$(x-a)$ that divides into $f(x)$ with no \gap{remainder}} 
\end{myVocabulary}
