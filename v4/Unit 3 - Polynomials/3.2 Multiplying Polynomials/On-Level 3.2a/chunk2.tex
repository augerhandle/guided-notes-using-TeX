\section{Multiplying Monomials $\times$ Polynomials}

\myCenteredBox[width=5in]{
    \large
    To multiply a {\bfseries\itshape monomial} (one term) 
    times a 
    {\bfseries\itshape polynomial} (several terms),
    you \gap{distribute}.
}

\myProblems
    {
        $ (-7x^2)  (5x^3 - 6x) $
    }
    {
        $ (2z^2)  (-4z^3 + z) $
    }
    {2.25in}


\myWideProblem
    {
        $ (2xy)  (x^2 -3xy + y^2) $
    }
    {3in}
