% \section{Using Remainders to Check for Divisors}

\myCenteredBox[width=4.5in]{
    A linear binomial $(x-a)$ is a {factor} of a polynomial, $P(x)$, 
    if the \gap{remainder} of 
    {\( \frac{P(x)}{x-a} \) }
    is \gap{zero}.
}

\myProblem[Is $(x-1)$ a factor of this polynomial? Explain why or why not.]
{
    \centering
    $
        9x^3 - 18x + 9
    $
}
