\section{Adding and Subtracting Polynomials}

To {\bfseries\itshape add} and {\bfseries\itshape subtract} polynomials,
\begin{itemize}
    \item \gap{distribute} any \gap{minus signs}
    \item \gap{combine} \gap{like} \gap{terms}.
\end{itemize}


\myProblems
    {
        $ (2x^3 + x^2 + x + 9) - (x^3 + 2x + 5)) $
    }
    {
        $ (6x^2 - 2x + 1) - (x^2 - 3x - 5)) $
    }
    [\normalsize]{1.25in}

    }
