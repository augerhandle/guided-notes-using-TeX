\section{Factoring Differences of Squares}

\myWideProblem[Multiply these polynomials using the {\bfseries\itshape Box Method}]
    {
        \centering $ (a+b) (a-b) $
    }
    {1in}


\myCenteredBox[width=5in]{
    A \gap{difference} \gap{of} \gap{squares}
    may be rewritten like this:
    \Large
    \begin{align*}{}
        (a^2 - b^2) = (a+b)(a-b)
    \end{align*}
}

        $ z^2 - 121 $
    }
    {
        $ 9x^2 - 16 $
    }
    [\normalsize]{1.4in}
