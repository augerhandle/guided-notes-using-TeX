
\section{Factoring Differences of Cubes}
\myWideProblem[Multiply these polynomials using the {\bfseries\itshape Box Method}]
    {
        \centering $ (a-b)(a^2 + ab + b^2) $
    }
    {1in}



\myCenteredBox[width=5in]{
    A \gap{difference} \gap{of} \gap{cubes}
    may be rewritten like this:
    \Large
    \begin{align*}{}
        (a^3 - b^3) = (a-b)(a^2 + ab + b^2)
    \end{align*}
}

        $ 125x^3 - 8 $
    }
