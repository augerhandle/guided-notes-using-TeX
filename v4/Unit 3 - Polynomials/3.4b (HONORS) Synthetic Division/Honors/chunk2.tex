\section*{Some Uses for Synthetic Division}
\subsection*{Identifying Factors (The Factor Theorem)}

\myCenteredBox[]{
    $(x-a)$ is a {factor} of a polynomial, $P(x)$, 
    if the \gap{remainder} of 
    {\( \frac{P(x)}{x-a} \) }
    is \gap{zero}.
}

\myProblem[Is $(x-1)$ a factor of this polynomial? Explain why or why not.]
{
    \centering
    $
        9x^3 - 18x + 9
    $
}
{2in}


\subsection*{Evaluating Polynomials (The Remainder Theorem)}

\myCenteredBox[]{
    If $R$ is the remainder of dividing {\( \frac{P(x)}{x-a} \) },
    then $P(a)$ ($P$ evaluated at $x=a$) is 
    $P(a) = R$.
}

\myProblem[Use synthetic division to evaluate the following polynomial at $x=2$.]
{
    \centering
    $
        P(x) = 9x^3 - 18x + 9
    $
}
{2in}
