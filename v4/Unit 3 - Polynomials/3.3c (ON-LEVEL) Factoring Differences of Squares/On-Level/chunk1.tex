\section{Factoring Differences of Squares}

\myCenteredBox[width=5in]{
    When a polynomial is a 
    \gap{difference} \gap{of} \gap{squares},
    it may be rewritten like this:
    \Large
    \begin{align*}{}
        (a^2 - b^2) = (a+b)(a-b)
    \end{align*}
}


\myProblems[Factor these polynomials.]
    {
        $ x^2 - 16 $
    }
    {
        $ x^2 - 144 $
    }
    {1.5in}

    \myProblems
    {
        $ 25x^2 - 64 $
    }
    {
        $ 4x^2 - 25 $
    }
    {1.5in}
