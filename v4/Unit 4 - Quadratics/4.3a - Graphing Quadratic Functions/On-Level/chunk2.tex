\section{Sketching Quadratic Functions using Tables}


\begin{myConceptSteps}{To graph a quadratic function\dots}
    \myStep{table}{Create an \gap{$x$-$y$ table} with at least five rows.}
    \myStep{$x$}{Choose at least five \gap{$x$ values} and put them in the $x$ column.}
    \myStep{evaluate}{%
        Substitute those $x$ values into the function in the \gap{$y$ column}.
        \begin{itemize}[nosep]
            \item Show your work.
            \item Make sure you put the $x$ values in \gap{parentheses}.
            \item The result is the $y$ value.
        \end{itemize}
        }
    \myStep{points}{Plot the ($x$,$y$) \gap{points} on a graph.}
    \myStep{connect}{Connect the dots \gap{smoothly}.}
\end{myConceptSteps}
