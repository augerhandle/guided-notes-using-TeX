\section{Sketching Quadratic Functions using Tables}


\begin{myConceptSteps}{To graph a quadratic function\dots}
    \myStep{table}{Create an $x$-$y$ \gap{table} with at least five rows.}
    \myStep{$x$}{Choose at least five $x$ values and put them in the $x$ column.}
    \myStep{evaluate}{%
        Substitute those $x$ values into the function.
        \begin{itemize}[nosep]
            \item Show your work.
            \item Make sure you put $x$ in \gap{parentheses}.
            \item The result is the $y$ value.
        \end{itemize}
        }
    \myStep{points}{Plot the ($x$,$y$) \gap{points} from each row on a graph.}
    \myStep{connect}{Connect the dots \gap{smoothly}.}
\end{myConceptSteps}
