


\begin{myConcept}{The Zero Product Property}
    If several things multiplied together are \gap{zero}, 
    then at least one of them is \gap{zero}.\\[2\baselineskip]
    %
    If $A$, $B$, $C$, \dots are \gap{factors} of a polynomial,
    then you solve the equation 
    \[(A)(B)(C)\dots=0\]
    by solving these \gap{simpler} equations:
    \begin{center}
        \hfill
        $A=0$
        \hfill
        $B=0$ 
        \hfill
        $C=0$ 
        \hfill 
        $\dots=0$
        \hfill{\,}
    \end{center}
\end{myConcept}
