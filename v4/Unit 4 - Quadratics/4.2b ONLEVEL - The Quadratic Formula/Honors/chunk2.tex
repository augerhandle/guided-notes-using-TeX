\section{Solving Equations with the Quadratic Formula}


\begin{myConceptSteps}{To solve a quadratic equation using the {\bfseries\itshape Quadratic Formula}\dots}
    \myStep{standard form}{%
        Write the equation like this: \gap{$ax^2 + bx + c = 0 $}.
        \begin{itemize}[nosep]
            \item One side of the equation must be equal to \gap{zero}.
        \end{itemize}
        }
    \myStep{constants}{Find $a$, $b$, and $c$. Calculate the $D = b^2 - 4ac$.}
    \myStep{substitute}{%
        Substitute the constants into the quadratic formula,
        \[
            x = \frac 
                    {-b \pm \myRoot{D}}
                    {2a}
        \]
            }
    \myStep{solutions}{You will get \gap{2}, \gap{1}, or \gap{0} solutions.}
\end{myConceptSteps}

\myProblems[Use the Quadratic Formula to solve these equations.]
{
    $ 2x^2 - 5x + 2 = 0 $
}
{
    $ z^2 - 4 = -2z $
}
