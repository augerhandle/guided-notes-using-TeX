\begin{myConceptSteps}{To rewrite a quadratic function into vertex form by completing the square\dots}
    \myStep{standard form}{Write it as \gap{$ax^2 + bx + c$}.}
    \myStep{leading coefficient}{%
        Factor \gap{$a$} from {first} \gap{2} {terms}: \gap{$a\left[x^2 + \frac{b}{a}x\right] + c$}.
    }
    \myStep{complete the square}{%
        {Complete the square} for the expression 
        \gap{inside} the brackets.
    }
    \myStep{distribute}{Distribute \gap{$a$} into the brackets.}
    \myStep{simplify}{Combine the \gap{last} \gap{two} terms.}
\end{myConceptSteps}
