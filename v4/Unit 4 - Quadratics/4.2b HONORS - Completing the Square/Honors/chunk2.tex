\section{Recognizing Perfect Square Trinomials}

\myCenteredBox[width=6in]{
    Whenever the \gap{first} and \gap{last} 
    terms of a trinomial are perfect squares,
    you should suspect that it factors like this:
    {\large\[
        a^2 + 2ab + b^2 \quad \bm{=} \quad (a+b)^2
    \]}
    If it does, it is called 
    a \gap{perfect} \gap{square} trinomial.
}

\begin{myConceptSteps}{To identify perfect square trinomials\dots}
    \myStep{perfect squares}[?]{%
        Are the first and last terms perfect squares?
        If so,  
        \begin{itemize}[nosep]
            \item Find \gap{a} as the {\bfseries\itshape square root} of the \gap{first} term.
            \item Find \gap{b} as the {\bfseries\itshape square root} of the \gap{last} term.
        \end{itemize}
        }
    \myStep{$2ab$}{Calculate \gap{$2ab$} using the results of step 1.}
    \myStep{middle term}{%
        If the middle term equals $2ab$, 
        then it's a perfect square trinomial.
        }
\end{myConceptSteps}
