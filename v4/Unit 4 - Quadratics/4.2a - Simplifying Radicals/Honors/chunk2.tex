\begin{myConceptSteps}{To simplify \gap{square} \gap{root} radicals\dots}
    \myStep{factor}{Factor the \gap{number} under the radical:
        \begin{itemize}[nosep]
            \item Use a \gap{factor} \gap{tree}.
        \end{itemize}
    }
    \myStep{move}{Remove \gap{pairs} of factors from under the radical:
        \begin{itemize}[nosep]
            \item Replace every factor pair with \gap{one} factor \gap{outside} the radical.
            \item If nothing remains under the radical, it goes away since $\myRoot{1} = 1$.
        \end{itemize}
    }
    \myStep{simplify}[!]{Multiply all the numbers:
        \begin{itemize}[nosep]
            \item Multiply together all the remaining numbers \gap{under} the radical.
            \item Multiply together all the numbers \gap{in} \gap{front} of the radical.
        \end{itemize}
    }
\end{myConceptSteps}
