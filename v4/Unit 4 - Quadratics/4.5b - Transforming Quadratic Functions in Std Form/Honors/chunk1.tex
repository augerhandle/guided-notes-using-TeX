\myCenteredBox[width=6in]{
    This is really just two lessons combined:
    \vspace{1\baselineskip}
    \begin{itemize}
        \item Convert to vertex form---Lesson \gap{4.5a}.
        \item Find the transformation/attribute information---Lesson \gap{4.4b}.
    \end{itemize}
}

\begin{myConceptSteps}{To find transformation details for quadratic functions written in standard form\dots}
    \myStep{vertex form}{%
        Convert the function to \gap{vertex form} by completing the square. 
        (See {\bfseries\itshape Lesson 4.5a})
    }
    \myStep{constants}{Find the values of $\bm{a}$, $\bm{h}$, and $\bm{k}$.}
    \myStep{transformations}{%
        Use the constant values and the {\itshape one-stop-shop} 
        (\gap{left} side)
        in the {\bfseries\itshape Lesson 4.4b} notes to 
        list all the transformations involved.
    }
    \myStep{attributes}{%
        Use the constant values and the {\itshape one-stop-shop} 
        (\gap{right} side)
        in the {\bfseries\itshape Lesson 4.4b} notes to find the attribute values. 
}
\end{myConceptSteps}
