\begin{itemize}[fullwidth]
    \item 
        {\bfseries\itshape Solving} an equation means%
        \dotfill 
        finding the \gap{roots/solutions}.
    \item 
        {\bfseries\itshape Roots/solutions}%
        \dotfill 
        have the same values as \gap{zeros}.
    \item 
        {\bfseries\itshape Zeros} are%
        \dotfill 
        the $x$-coordinates of \gap{$x$-intercepts}.
    \item 
        You can see $\bm{x}$-{\bfseries\itshape{}intercepts}%
        \dotfill 
        on a \gap{graph} of the function.
\end{itemize}
