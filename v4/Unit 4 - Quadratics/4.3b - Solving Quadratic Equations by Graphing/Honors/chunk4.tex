\section*{Using the TI-84 to Find Zeros}

\begin{myConceptSteps}{To find each zero of a function using a TI-84\dots}
    \myStep{graph the function}{%
        \begin{itemize}[nosep]
            \item Press the \myKey{Y=} key, and enter the function.
            \item Press the \myKey{GRAPH} key.
        \end{itemize}
    }
    \myStep{zero command}{%
        \begin{itemize}[nosep]
            \item Press \myKey{2nd}\myKey{TRACE} to get the CALC menu.
            \item Press \myKey{2} to select the {\ttfamily zero} command
        \end{itemize}
    }
    \myStep{left bound}{%
        \begin{itemize}[nosep]
            \item The calculator asks for a {\itshape Left Bound}.
            \item It wants you to select a point to the left of the $x$-intercept.
            \item Use the arrow keys to move the {\itshape cursor} to the left of the $x$-intercept.
            \item Press \myKey{ENTER}.
        \end{itemize}
    }
    \myStep{right bound}{%
        \begin{itemize}[nosep]
            \item The calculator asks for a {\itshape Right Bound}.
            \item It wants you to select a point to the right of the $x$-intercept.
            \item Use the arrow keys to move the {\itshape cursor} to the right of the $x$-intercept.
            \item Press \myKey{ENTER}.
        \end{itemize}
    }
    \myStep{guess}{%
        \begin{itemize}[nosep]
            \item The calculator prompts for a {\itshape Guess}.
            \item Use the arrow keys to move the {\itshape cursor} 
                close to the $x$-intercept.
            \item Press \myKey{ENTER}.
        \end{itemize}
    }
    \myStep{zero}{%
        \begin{itemize}[nosep]
            \item The calculator will display the $x$ and $y$ coordinates 
                of the $x$-intercept.
            \item The zero is the $x$-coordinate.
        \end{itemize}
    }
\end{myConceptSteps}
