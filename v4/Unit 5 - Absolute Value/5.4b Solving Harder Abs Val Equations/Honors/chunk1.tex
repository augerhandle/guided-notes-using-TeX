\section{Two Absolute Value Expressions}

\subsection{Two Identical Absolute Values}

\begin{tcolorbox}[center,width=5.5in,]
    \raggedright
    If the equation includes an absolute value \gap{several} \gap{times},
    then \gap{collect} them together.\\[0.5\baselineskip]
    And solve the simpler absolute value equation.
\end{tcolorbox}

\myWideProblemWithContent[%
    Solve this equation. 
    Notice that it has two \myEmph{identical} absolute value expressions.
    ]
{
    \begin{center} $6|3-3b|=72-2|3-3b| $ \end{center}
    \tcblower
    \phantom{empty line of text}\\[0.5\baselineskip]
    \begin{minipage}{0.5\linewidth}
        {\,}
    \end{minipage}
    % \hfil
    \rule{0.5pt}{2.25in}
    \begin{minipage}{0.5\linewidth}
        \vfill
        \raggedleft\gap{$b=-2, 4$}
    \end{minipage}
}




\subsection{Two Different Absolute Values}

\begin{tcolorbox}[center,width=5in,]
    If the equation consists of \gap{two} \gap{different} absolute value expressions
    on each side of the equation,
    then the inside expressions must either be \gap{equal} or \gap{opposites}.
\end{tcolorbox}


\myWideProblemWithContent[%
    Solve this equation. 
    Notice that two \myEmph{different} absolute value expressions are equal to each other.
    ]
{
    \begin{center}
        $|z-2|=|2z+3| $
    \end{center}
    \tcblower
    \begin{minipage}{0.5\linewidth}
        {\,}
    \end{minipage}
    % \hfil
    \rule{0.5pt}{1.5in}
    \begin{minipage}{0.5\linewidth}
        \vfill
        \raggedleft\gap{$z=-5, -\frac{1}{3}$}
    \end{minipage}
}
