This is an absolute value equation:
\begin{equation*}
    5 - |6x + 14| = |6x+14| + 23 
\end{equation*}
%
You will learn to \myEmph{solve} these.

\begin{tcolorbox}[center,width=4in,]
    Before you can solve absolute value equations,
    you must know how to \gap{isolate} the absolute value expressions.
\end{tcolorbox}


\myProblems[Isolate the absolute value expressions in these equations.]
{
    $|x| + 5 = 0$
}
{
    $2|x+1| = 20$
}{0.5in}

\myProblems
{
    $2|x-2| -3 = 7$
}
{
    $5 + |x+3| = 2$
