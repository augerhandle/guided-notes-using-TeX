\section{The Graph of Transformed Absolute Value Functions}

\begin{myConceptSteps}{To quickly sketch the graph of a transformed absolute value function\dots}
    \myStep{vertex}{%
        Shift the \gap{vertex} by $\bm{h}$ and $\bm{k}$. 
        Put a \gap{dot} at $(\bm{h},\bm{k})$.
    }
    \myStep{right line}{%
        Draw a \gap{line} starting at the vertex with \gap{slope} $\bm{a}$.
    }
    \myStep{left line}{Draw a mirror image of the right line on the \gap{left} side of the vertex.}
\end{myConceptSteps}


\myProblemsWithContent[Sketch the graphs of these functions.]
{
    $g(x) = 2|x - 3| -1$
    \tcblower 
    \centering
    ${a} =$ \gap{$2$},\quad ${h} =$ \gap{$3$},\quad ${k} =$ \gap{$-1$}\\[\baselineskip]
    \begin{myTikzpictureGrid}{0.35} [-3]{7}{5}
        \whenTEACHER{
            \tkzFct[ solid, ultra thick, samples=500, domain=-10:10,]{2*abs(x-3)-1}
        }
        % \tkzText(5,7){$f(x) = x^2$}
    \end{myTikzpictureGrid}
}
{
    $g(x) = \frac{1}{2}|x+3| -2 $
    \tcblower 
    \centering
    ${a} =$ \gap{$\frac{1}{2}$},\quad ${h} =$ \gap{$-3$},\quad ${k} =$ \gap{$-2$}\\[\baselineskip]
    \begin{myTikzpictureGrid}{0.35} [-8]{2}{5}
        \whenTEACHER{
            \tkzFct[ solid, ultra thick, samples=500, domain=-10:10,]{0.5*abs(x+3)-2}
        }
        % \tkzText(5,7){$f(x) = x^2$}
    \end{myTikzpictureGrid}
}
