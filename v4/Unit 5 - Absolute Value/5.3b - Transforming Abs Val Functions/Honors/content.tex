\begin{myObjectives}
    \myObjective{find}{$a$, $h$, $k$ in an absolute value function}
    \myObjective{find}{the attributes of absolute value functions}
    \myObjective{describe}{%
        the transformations that convert an absolute value 
        parent function into a transformed function 
        }
    \myObjective{convert}{%
        between different forms of absolute value functions:
        \begin{itemize}
            \item absolute value functions written in terms of $a$, $h$, $k$,
            \item graphs of absolute value functions, and 
            \item descriptions of the transformations 
        \end{itemize}
    }
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{function family}{a group of functions that are \gap{related} to each other}
    \myDefinition{parent function}{the \gap{simplest} function in a function family}
    \myDefinition{attribute}{feature, characteristic, something interesting}
\end{myVocabulary}
