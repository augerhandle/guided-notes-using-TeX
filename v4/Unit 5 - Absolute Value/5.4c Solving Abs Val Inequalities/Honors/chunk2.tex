\section{How to solve absolute value inequalities}

\begin{myConceptSteps}{To solve absolute value equations\dots}
    \myStep{isolate}{%
        Get the absolute value expression \gap{by} \gap{itself}.
        \begin{itemize}[nosep]
            \item[{\Large\ding{43}}] 
                \gap{Reverse} the inequality
                if you \myEmph{multiply/divide} 
                by a \gap{negative}.
        \end{itemize} 
    }
    \myStep{check}{When the absolute value expression is
        \begin{center}
            \renewcommand{\arraystretch}{1.75}
            \begin{tabular}{|lcl|}
                \hline
                {less than a negative} 
                & {\LARGE\ding{224} }
                & {\itshape no solution}
                    \\
                \hline
                {greater than a negative}    
                    & {\LARGE\ding{224} }
                    & {\itshape all real numbers}
                    \\
                \hline
            \end{tabular}
        \end{center}
    }
    \myStep{rewrite}{Rewrite the inequality depending on the symbol:
        \begin{center}
            \renewcommand{\arraystretch}{1.75}
            \large
            \begin{tabular}{|ccc|}
                \hline
                $|ax+b| > c$ 
                & {\LARGE\ding{224} }
                & $ax+b < -c$ \, \myEmph{or} \, $ax+b > c$
                    \\
                \hline
                $|ax+b| < c$    
                    & {\LARGE\ding{224} }
                    & $-c < ax+b < c$
                    \\
                \hline
            \end{tabular}
        \end{center}
    }
    \myStep{solve}{Solve the \gap{simpler} inequalities.}
\end{myConceptSteps}
