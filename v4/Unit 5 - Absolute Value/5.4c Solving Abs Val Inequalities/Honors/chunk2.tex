\section{Solving}

\begin{myConceptSteps}{To solve absolute value inequalities\dots}
    \myStep{isolate}{Get the absolute value expression \gap{by} \gap{itself}.
        \begin{tcolorbox}[center,width=5in,]
            \begin{itemize}
                \item [{\Large\ding{43}}]
                \gap{Reverse} the inequality
                if you \myEmph{multiply/divide} 
                by a \gap{negative}.
                \end{itemize}
        \end{tcolorbox}
    }
    \myStep{check}{Watch out for\gap{no solution} or \gap{all real numbers}.}
    \myStep{rewrite}{Rewrite the inequality depending on the symbol:
        \begin{center}
            \renewcommand{\arraystretch}{1.75}
            \large
            \begin{tabular}{|ccc|}
                \hline
                $|ax+b| > c$ 
                & {\LARGE\ding{224} }
                & $ax+b < -c$ \, \myEmph{or} \, $ax+b > c$
                    \\
                \hline
                $|ax+b| < c$    
                    & {\LARGE\ding{224} }
                    & $-c < ax+b < c$
                    \\
                \hline
            \end{tabular}
        \end{center}
    }
    \myStep{solve}{Solve the \gap{simpler} inequalities.}
\end{myConceptSteps}
