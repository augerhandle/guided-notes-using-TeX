\section{How to solve absolute value equations}

\begin{myConceptSteps}{To solve absolute value equations\dots}
    \myStep{isolate}{Get the absolute value expression \gap{by} \gap{itself}.}
    \myStep{check}{If the expression equals a \gap{negative}, the answer is \gap{no} \gap{solution}.}
    \myStep{split}{Write two equations for the \gap{postive} and \gap{negative} cases}
    \myStep{solve}{solve each of the \gap{simpler} equations}
\end{myConceptSteps}
