\section{Shifting the Parent Function}

\begin{tcolorbox}[center,width=4in,]
    When you shift the parent function, 
    you shift its \gap{vertex}, too.
    \end{tcolorbox}

\myWideProblemWithContent[%
    Use a graphing calculator to graph these functions.\vspace{-0.75\baselineskip}
    \begin{itemize}[nosep]
        \item Sketch the graph of the {\bfseries\itshape parent function}, $f(x)=|x|$.
        \item Sketch the graph that you see in the calculator.
        \item Draw a dot where the vertex is.
        \item Describe what happened.
    \end{itemize}
    ]
{
    $g(x) = |x| -4$
    \tcblower
    \begin{minipage}{0.325\textwidth}
        \begin{myTikzpictureGrid}{0.4} {6}{6}
            % \tkzText(13,8){$x - y = 0$}
            %\tkzFct[ solid, ultra thick, samples=100, domain=-6:6,]{abs(x)-4}
            % \tkzText(-13,7.25){$x + 2y = 6$}
        \end{myTikzpictureGrid}
    \end{minipage}
    \begin{minipage}{0.675\textwidth}
        The parent function was shifted\\
        \gap{vertically} \gap{down} by \gap{4}.
    \end{minipage}
}
