\section{Expressions with Variables}


\begin{myConceptSteps}{To evaluate absolute values of variable expressions\dots}[
    \begin{tcolorbox}[center,width=5in,colback=white,]
        \centering
        Remember: {\bfseries\itshape never distribute} anything to the inside of $|\cdots|$\,!
    \end{tcolorbox}
]
    \myStep{substitute}{Substitute the given value of the variable into the expression.}
    \myStep{evaluate}{Evaluate the resulting absolute value expression with numbers.}
\end{myConceptSteps}
