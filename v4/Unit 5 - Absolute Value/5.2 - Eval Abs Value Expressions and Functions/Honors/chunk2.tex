\myProblems[Evaluate these absolute value expressions using the given value of the variable.]
{
    $-5|3x+2|$ \quad where $x=-2$ % = -5|3*(-2) + 2| = -5|-6+2| = -5|-4| = -5*4 = -20
}
{
    $-10|x^2 - 5x - 9|$ \quad where $x=-2$ % = -10|(-2)^2 -5(-2) - 9| = -10|4+10-9| = -10|5| = -50
}{2in}[{\raggedleft result = \gap{-20}}][{\raggedleft result = \gap{-50}}]
