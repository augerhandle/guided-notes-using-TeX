\section{Expressions with Numbers}

\begin{myConceptSteps}{To evaluate absolute values of expressions of {\bfseries\itshape only numbers}\dots}
    \myStep{inside first}{
        Focus on what is \gap{inside} the $|\cdots|$. 
        \begin{itemize}
            \item Ignore everything outside the $|\cdots|$ at the beginning.
            \item Never distribute anything to the inside.
        \end{itemize}
    }
    \myStep{simplify}{
        Simplify the expression inside $|\cdots|$. Use GEMDAS.
        \begin{itemize}
            \item The result is a \gap{positive} \gap{number}.
        \end{itemize}
    }
    \myStep{evaluate}{
        Replace $|\cdots|$ by the absolute value of the number. ({\bfseries\itshape Lesson 5.1})
        \begin{itemize}
            \item There should be \gap{no} \gap{vertical} \gap{bars} left.
        \end{itemize}
    }
    \myStep{Solve}{
        Simplify whatever is left.
    }
\end{myConceptSteps}
