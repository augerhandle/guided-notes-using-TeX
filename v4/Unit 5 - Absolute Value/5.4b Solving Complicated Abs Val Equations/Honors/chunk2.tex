\section{Extraneous Solutions}

\begin{tcolorbox}[center,width=5.75in,]
    If the equation has the variable \gap{inside} and \gap{outside}
    the absolute value, 
    some of your solutions might be \gap{extraneous}.\\[0.5\baselineskip]
    So you \myEmph{must} \gap{check} your answers to make sure they work.
\end{tcolorbox}

\myWideProblemWithContent[Solve this equation. Make sure to check for extraneous solutions.]
{
    \begin{center}
        $2|4-n|=-3n$
    \end{center}
    \tcblower
    \begin{minipage}{0.5\linewidth}
        {\,}
    \end{minipage}
    % \hfil
    \rule{0.5pt}{2in}
    \begin{minipage}{0.5\linewidth}
        \vfill
        \raggedleft\gap{$n=-8$}
    \end{minipage}
}
