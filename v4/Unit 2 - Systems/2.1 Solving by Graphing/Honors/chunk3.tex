\section{Verifying Solutions}

\begin{myConceptSteps}{To determine if a point is a solution of a system of 2 equations\dots}
    \myStep{substitute}{
        Replace the variables by the point's \gap{coordinates}}.
    \myStep{check}{
        If both equations are true, the point \gap{is} a solution.}
\end{myConceptSteps}


\myProblemsWithContent[Determine if the specified point is a solution of the system.]
{
    Is the point $(1,1)$ a solution to this system?
    \begin{align*}
        x+y &= 2  \\
        6x+y &= -3
    \end{align*}  
    \vspace{2in}
}
{
    Is the point $(1,-2)$ a solution to this system?
    \begin{align*}
        x-y &= 3 \\
        5x+y &= 3 
    \end{align*}  
}[\normalsize]
