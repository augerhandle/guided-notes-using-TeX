\section{Elimination by Multiplying and then Adding}

\myCenteredBox[]{
    What do you do with a system like this?\\
    \begin{center}
        \vspace{-0.5\baselineskip}
        \sysdelim..
        \systeme{
            2x-8y=-12,
            2x+7y=-27
        }
        \vspace{0.5\baselineskip}
    \end{center}
    If there are no \gap{negatives}, 
    \gap{multiply} one of the equations by \gap{$-1$} to 
    get \gap{negatives}.
}
