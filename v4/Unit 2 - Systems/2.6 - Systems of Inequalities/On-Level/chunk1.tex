\section{Graphing a 2-variable inequality}

\begin{myConceptSteps}{To sketch the solution of a 2-variable inequality\dots}
  \myStep{rewrite}{%
    Write the inequality in \gap{slope-intercept} form
    unless the boundary is a \gap{vertical} line.
  }
  \myStep{boundary line}{%
    Sketch the line.
    \begin{itemize}[fullwidth]
      \item Use a \gap{dashed} line for {\Large $<$} and {\Large $>$}.
      \item Use a \gap{solid} line for {\Large $\le$} and {\Large $\ge$}.
    \end{itemize}
  }
  \myStep{shading}{%
    \begin{itemize}[fullwidth]
      \item For {\Large $>$} and {\Large $\ge$}, 
      draw an arrow \gap{up} from the line. Shade \gap{above}.
      \item For {\Large $<$} and {\Large $\le$},
      draw an arrow \gap{down} from the line. Shade \gap{below}.
      \item {\itshape For vertical lines,}
        \begin{itemize}
          \item for {\Large $>$} and {\Large $\ge$}, 
          shade to the \gap{right}.
          \item for {\Large $<$} and {\Large $\le$},
          shade to the \gap{left}.
        \end{itemize}
    \end{itemize}
  }
\end{myConceptSteps}
