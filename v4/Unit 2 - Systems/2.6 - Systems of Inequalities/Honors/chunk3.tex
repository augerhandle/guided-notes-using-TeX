\section{Is a point a solution?}

\begin{myConcept}{To determine if a point is a solution to a system of inequalities\dots}
  \begin{itemize}[fullwidth]
    \item Sketch the graph of the solution.
    \item The point \gap{is a solution} {\bfseries\itshape only if}
    \begin{itemize}
      \item it is \gap{inside} the shading, or 
      \item it is \gap{on} a solid line and \gap{not on} any dashed lines.
    \end{itemize}
  \end{itemize}
\end{myConcept}
