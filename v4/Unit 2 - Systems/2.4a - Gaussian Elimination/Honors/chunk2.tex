\section{Gaussian Elimination}

\begin{myConceptSteps}{To transform a system to row echelon form\dots}
  \myStep{left-to-right}{%
      Transform each \gap{column} starting from the
      \gap{left} and moving to the \gap{right}.
      }
  \myStep{ones}{%
      Use {\bfseries\itshape row operations} to get $1$s \gap{on the diagonal}.
      }
  \myStep{zeros}{%
      Use {\bfseries\itshape row operations} to get $0$s \gap{below the diagonal}.
    }
  \myStep{column-by-column}{%
    Finish the {\bfseries\itshape ones} and {\bfseries\itshape zeros} 
    \gap{before} you move to the next column.
}
\end{myConceptSteps}
