\noindent
But what do you do with a system like this?
\begin{center}
    \centering
    \sysdelim..
    \systeme{
        -2x-4y=2,
        -3x+3y=12
    }
\end{center}

\myCenteredBox[width=5.5in]{
    It problems like this,
    you have to multiply \gap{both} equations by \gap{different} numbers
    to get negative coefficients to line up.
}

\begin{myConceptSteps}{To solve systems by multiplying two equations and using elimination\dots}
    \myStep{organize}{Rewrite the system so it is \gap{aligned}.}
    \myStep{opposites}{
        Multiply both equations by \gap{two} numbers 
        so that the coefficients of one of the variables 
        are \gap{negatives} of each other.
        }
    \myStep{first variable}{
        Add the equations. This will \gap{eliminate} a variable.
        Solve for the \gap{other} variable.}
    \myStep{second variable}{
        Substitute the value of the \gap{first} variable 
        into one of the equations.
        Solve for the \gap{other} variable.}
\end{myConceptSteps}
