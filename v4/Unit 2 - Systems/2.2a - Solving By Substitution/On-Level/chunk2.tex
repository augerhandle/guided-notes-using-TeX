\section{Solving Simple Systems Algebraically}

\begin{myConceptSteps}{To solve a simple system by substitution\dots}
    \myStep{substitute}{Replace \gap{$y$} in the first equation by the expression in the second equations.}
    \myStep{$x$}{Solve that equation for \gap{$x$}. You will get a number.}
    \myStep{$y$}{Substitute the $x$-value into either one of the original equations and simplify.}
\end{myConceptSteps}

\myWideProblem[Solve this simple system by substitution.]
{
    \vspace{-\baselineskip}
    \begin{align*}
        y &= -2x + 3 \\
        y &= x-3
    \end{align*}
