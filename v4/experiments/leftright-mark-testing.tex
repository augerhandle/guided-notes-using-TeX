% \makepsmarks is the commmand that automates the creation of 
% \leftmark and \rightmark by the various sectioning commands.
% I am redefining it here (as discussed in the memman doc) 
% to do what I want.
%
% \leftmark is set at the FIRST occurance of \markboth or \markright.
% \rightmark is set at the LAST occurance of them.
%
\makeatletter
\makepsmarks{headings}{%
    \def\partmark##1{%
        \markboth{%
        % \ifnum\c@secnumdepth > \m@ne
        %     \if@mainmatter
              \partname\ \thepart\ % ie, "Unit 1"
        %     \fi
        % \fi
        ##1}{}
    }%
    \def\chaptermark##1{%
        \markright{%
            % \ifnum\c@secnumdepth > \z@
                \thepart.\thechapter \ % ie, "<unit#>.<lesson#>"
            % \fi
        ##1}%
    }
}
\makeatother
