\dummypart{9}{Linear Regression and Data Models}
\myLesson{Introduction to Linear Regression}[2]

\begin{myObjectives}
    \myObjective{identify}{whether the data in a scatterplot are 
        \begin{itemize}[nosep]
            \item \gap{strongly}, \gap{weakly} or \gap{not} 
                linearly correlated 
            \item \gap{positively} or \gap{negatively} correlated
        \end{itemize}
    }
    \myObjective{sketch}{an approximate line of \gap{best} \gap{fit}}
    \myObjective{write}{the equation of the \gap{line} of \gap{best} \gap{fit} using 
        \myEmph{linear regression} and {\scshape Desmos}
    }
\end{myObjectives}
            
\begin{myVocabulary}
    \myDefinition{scatterplot}{a plot of a bunch of $(x,y)$ points}
    \myDefinition{correlated}{two things that are \gap{related} to each other}
    \myDefinition{line of best fit}{%
        a line that \gap{almost} 
        goes through all the points in a scatterplot
    }
    \myDefinition{linear regression}{%
        a way to find the line of \gap{best} \gap{fit} 
        for many points
    }
\end{myVocabulary}
