\section{Evaluating Simple Logarithms}

\begin{minipage}{0.725\textwidth}
\begin{myConceptSteps}{To evaluate logs without a calculator\dots}[]
    \myStep{equation}{Write it in logarithmic form by adding $= x$. }
    \myStep{rewrite}{Rewrite in \gap{exponential} form.}
    \myStep{result}{The value of \gap{$x$} that makes the equation true. }
\end{myConceptSteps}
\end{minipage}
\begin{minipage}{0.25\textwidth}
    \whenTEACHER{
        \tiny
        $log_4(64) = $?\\[0.25\onelineskip]
        1. log form: $log_4(64)=x$ \\
        \quad $b=4$, $A=64$, $x=x$\\
        2. exp form: $b^x=A \Rightarrow 4^x=64$\\
        3. use cheat sheet!\\
        \quad $4^3 = 64$\\[0.25\onelineskip]
        so \fbox{$x=3$}
    }
\end{minipage}
