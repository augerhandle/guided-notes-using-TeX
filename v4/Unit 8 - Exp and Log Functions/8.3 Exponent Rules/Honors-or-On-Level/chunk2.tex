\begin{tcolorbox}[]
    {\bfseries Product Rule.}
    To multiply two identical bases,
    write the base and \myEmph{add} the exponents.
    \[b^m b^n = b^{m+n}\]
\end{tcolorbox}
\begin{tcolorbox}[]
    {\bfseries Quotient Rule.}
    To divide two identical bases,
    write the base and \myEmph{subtract} the exponents.
    \[\frac{b^m}{b^n} = b^{m-n}\]
\end{tcolorbox}


\begin{tcolorbox}[enhanced,opacityframe=0,]
    \begin{itemize}
        \item $2^3 2^5 = 2^8$
        % \item $x^5 x^2 = x^7$
        \item $(x^2 y^3) (xy^2) = x^3 y^5$
    \end{itemize}
    \end{tcolorbox}
    \begin{tcolorbox}[enhanced,opacityframe=0,]
        \begin{itemize}
            \item $\frac{2^5}{2^3} = 2^2$
            \item $\frac{x^5 y^4 z^2}{x^2 y^2 z} = x^3 y^2 z$
        \end{itemize}
    \end{tcolorbox}
