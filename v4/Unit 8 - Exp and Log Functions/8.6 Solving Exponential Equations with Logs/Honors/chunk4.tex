\section{Solving Exponential Equations with Inverse Functions}

\begin{myConceptSteps}{To solve equations with inverse functions\dots}
    \myStep{isolate}{Get the exponential term by itself.}
    \myStep{base}{Find the base $b$ of the exponential term.}
    \myStep{$log_b(\dots) = log_b(\dots)$}{Take the logarithm base $b$ of \gap{both} \gap{sides} of the equation.}
    \myStep{solve}{Solve for the variable.}
\end{myConceptSteps}
