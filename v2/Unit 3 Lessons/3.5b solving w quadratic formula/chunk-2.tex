\section*{Solutions, roots, zeros, $x$-intercepts}

\begin{center}
    \begin{tcolorbox}[width=5.5in]
        The following words mean (almost) the \gap{same thing}:
        {\large
        \begin{itemize}[itemsep=0in]
            \item {\bfseries\itshape solution}
            \item {\bfseries\itshape root}
            \item {\bfseries\itshape zero}
            \item {\bfseries\itshape $x$-intercept}
        \end{itemize}
        }
    \end{tcolorbox}
\end{center}

\myBlankExample{0.5in}{
    What is the difference between the following problem statements?
    \begin{itemize}
        \item {\bfseries\itshape Solve} this equation: $2x^2 + 3 + 4 = 0$.
        \item {\bfseries\itshape Find the solution} of this equation: $2x^2 + 3 + 4 = 0$.
        \item {\bfseries\itshape Find the roots} of this equation: $2x^2 + 3 + 4 = 0$.
        \item {\bfseries\itshape Find the zeros} of this equation: $2x^2 + 3 + 4 = 0$.
        \item {\bfseries\itshape Find the $x$-intercepts} of this equation: $2x^2 + 3 + 4 = 0$.
    \end{itemize}
}
