

\section*{Factoring expressions by removing the GCF}


\begin{myConceptSteps}{
    To factor an expression by removing the GCF\dots
}
\myStep{terms}{
    Identify each of the terms in the expression.
}
\myStep{GCF}{
    Find the GCF of all of the terms.
}
\myStep{factor}{
    Divide all the terms by the GCF,
    and write the GCF in front.
}
\end{myConceptSteps}
