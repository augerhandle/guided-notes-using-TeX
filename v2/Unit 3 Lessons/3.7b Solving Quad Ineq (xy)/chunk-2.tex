\section*{Solving quadratic inequalities with two variables}

We will solve inequalities that look like this:

\begin{center}
\begin{minipage}{0.4\textwidth}
    \begin{itemize}[itemsep=0in]
        \item $y <    ax^2 + bx + c$
        \item $y \leq ax^2 + bx + c$
    \end{itemize}
\end{minipage}
\begin{minipage}{0.4\textwidth}
    \begin{itemize}[itemsep=0in]
        \item $y >    ax^2 + bx + c$
        \item $y \geq ax^2 + bx + c$
    \end{itemize}
\end{minipage}
\end{center}



\begin{taggedblock}{pre-AP}
    We are now working in a {\bfseries\itshape two-dimensional} universe
    (not the one-dimensional number line),
    but the idea of a {\bfseries\itshape boundary} and 
    {\bfseries\itshape partitioning} the universe still applies.

    \begin{center}
        \begin{tcolorbox}[width=6in]
            \begin{itemize}
                \item The parabola is the \gap{boundary}.
                \item It might or might not be part of the solution. 
                But we cannot use a circle to indicate this.
                Instead, we draw it \gap{dashed} or \gap{solid}.
                \item The boundary \gap{partitions} the two-dimensional universe into two regions:
                    \begin{itemize}
                        \item[$\circ$] a greater-than part (\gap{above} the boundary)
                        \item[$\circ$] a less-than part (\gap{below} the boundary)
                    \end{itemize}
                \item We shade the part that satisfies the the inequality.
            \end{itemize}
        \end{tcolorbox}
    \end{center}
\end{taggedblock}



\begin{myConceptSteps}{
    To sketch the graph of a quadratic inequality with {\bfseries\itshape two variables}\dots
}
\myStep{boundary}{
    Decide if the parabola is 
    \gap{dashed} or \gap{solid}
    \begin{center}
        \fbox{\Large
        \begin{tabular}{l|c|l}
            {\bfseries\itshape dashed} & $>$, $<$       & not equals \\
            {\bfseries\itshape solid}  & $\geq$, $\leq$ & equals included \\
        \end{tabular}
        }
    \end{center}
}
\myStep{graph}{
    Sketch the graph of the \gap{parabola} as dashed or solid.
    (We will use {\scshape Desmos} or a {\scshape TI-84} to find the graph.)
}
\myStep{shade}{
    Shade in the graph above or below the line.
    \begin{center}
        \fbox{\Large
        \begin{tabular}{l|c|l}
            {\bfseries\itshape above} & $>$, $\geq$ &greater... \\
            {\bfseries\itshape below} & $<$, $\leq$ &less... \\
        \end{tabular}
        }
    \end{center}
}
\end{myConceptSteps}

    \myBlankExample{3in}{
        Sketch the graph of the solution of this quadratic inequality:
        \[
            y > x^2 + 4x + 6
        \]
    }
    \myBlankExample{3in}{
        Sketch the graph of the solution of this quadratic inequality:
        \[
            y < -x^2 -x - 2
        \]
    }
