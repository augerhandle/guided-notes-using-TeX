\section*{Does a point satisfy a quadratic inequality with two variables?}



\begin{myConceptSteps}{
    To determine if a point in the plane
    satsifies a quadratic inequality with {\bfseries\itshape two variables}\dots
}
\myStep{sketch}{
    \gap{Sketch} the graph of the quadratic inequality, 
    with shading above/below the dashed/solid boundary.
}
\myStep{plot}{
    Plot the \gap{point} in the plane.
    The point satisfies the inequality if either
    \begin{itemize}
        \item the point is in the shaded region, or
        \item the boundary is solid and the point is on the boundary.
    \end{itemize}
}
\end{myConceptSteps}

\begin{taggedblock}{on-level}
    \myBlankExample{3in}{
        Determine if the points $(0,0)$ and $(4,-1)$ satisfy this inequality:
        \[
            y \leq x^2 - 8x + 12 
        \]
    }
\end{taggedblock}


\begin{taggedblock}{pre-AP}
    \myBlankExample{3in}{
        Determine if the point $(4,-1)$ satisfies this inequality:
        \[
            y > -x^2 + 4x -6 
        \]
    }
    \myBlankExample{3in}{
        Determine if the point $(2,-2)$ satisfies the inequality 
        in the previous example.
    }
\end{taggedblock}
