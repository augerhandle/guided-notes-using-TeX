\section*{Solving quadratic inequalities}

We will solve inequalities that look like this:

\begin{center}
\begin{minipage}{0.4\textwidth}
    \begin{itemize}[itemsep=0in]
        \item $ax^2 + bx + c > 0$
        \item $ax^2 + bx + c \geq 0$
    \end{itemize}
\end{minipage}
\begin{minipage}{0.4\textwidth}
    \begin{itemize}[itemsep=0in]
        \item $ax^2 + bx + c < 0$
        \item $ax^2 + bx + c \leq 0$
    \end{itemize}
\end{minipage}
\end{center}

\begin{myConceptSteps}{
    To solve a quadratic inequality with {\bfseries\itshape one variable}\dots
}
    \myStep{standard form}{
        Move terms to the \gap{left},
        leaving \gap{zero} on the right.
    }
    \myStep{solve it}{
        Solve the associated \gap{equation} by
        \begin{itemize}[itemsep=0in]
            \item factoring or
            \item the quadratic formula
        \end{itemize}
    }
    \myStep{intervals}{
        Plot the solutions on a number line
        as \gap{circles} (filled-in or empty).
        This divides the number line into \gap{intervals}.
    }
    \myStep{test points}{
        Evaluate the function at a point
        \gap{inside} each interval.
    }
    \myStep{solution}{
        \begin{itemize}[itemsep=0in]
            \item On a number line, \gap{shade} intervals 
            where the tests satisfy the inequality.
            \item Write the answer as an \gap{inequality}.
        \end{itemize}
    }
\end{myConceptSteps}
