\section*{Factoring quadratic expressions of the form $x^2 + bx + c$}

\begin{myConceptSteps}{
    To factor a quadratic expression of the form $x^2 + bx + c$\dots
}
\myStep{$b$ and $c$}{
    Identify the values of $b$ and $c$.
}
\myStep{two numbers}{
    Find two numbers that
    \begin{itemize}[itemsep=0in]
        \item \gap{multiply} together to give $c$
        \item \gap{add} together to give $b$
    \end{itemize}
}
\myStep{rewrite}{
    Rewrite the expression as 
    \begin{center}
        $x^2 + bx + c = $
        \underline{\phantom{\hspace{1.5in}}}
        \,
        \underline{\phantom{\hspace{1.5in}}}
    \end{center}
}
\end{myConceptSteps}

\begin{center}
    \begin{tcolorbox}[width=2.5in]
        This only works if \gap{$a=1$}
    \end{tcolorbox}
\end{center}
