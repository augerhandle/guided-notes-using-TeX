

\section*{Evaluating quadratic functions with numbers as inputs}

\begin{myConceptSteps}{To evaluate a quadratic function given a number as input\dots}
    \myStep{substitute}{%
        Substitute the number for the \gap{input variable} wherever is appears.
        Put \gap{parentheses} around the number when you substitute.
    }
    \myStep{simpify}{%
        Simplify the expression.
        You should get a \gap{number} as the output.
    }
\end{myConceptSteps}



\begin{taggedblock}{on-level}
    \myBlankExample{2in}{
        \Large
        For the function $f(x) = 3x^2 + x$,
        find $f(-2)$.
    }
\end{taggedblock}
\begin{taggedblock}{pre-AP}
    \myBlankExample{2in}{
        \Large
        For the function $f(x) = -3x^2 - x$,
        find $f(-2)$.
    }
\end{taggedblock}
