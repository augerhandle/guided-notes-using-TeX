\section*{Evaluating quadratic functions with expressions as inputs}

\begin{myConceptSteps}{To evaluate a quadratic function given an algebraic expression as input\dots}
    \myStep{substitute}{%
        Substitute the expression for the \gap{input variable} wherever is appears.
        Put \gap{parentheses} around the expression when you substitute.
    }
    \myStep{simpify}{%
        Simplify the expression.
        You should get a new \gap{expression} as the output.
    }
\end{myConceptSteps}


\myBlankExample{1.75in}{
    \Large
    For the function $f(x) = x^2 - x - 1$,
    find $f(2t)$.
}

\myBlankExample{3in}{
    \Large
    For the function $g(x) = 2(x-1)^2 - 3$,
    find $f(z-2)$.
}
