\section*{Adding/Subtracting complex numbers}

{\bfseries\itshape Complex numbers} are a combination 
of a \gap{real} number and an \gap{imaginary} number.

\begin{center}
    \begin{tcolorbox}[width=5in]
        {\bfseries\itshape Imaginary numbers} have {\bfseries\itshape two parts} 
        and are written
        \vskip1em
        \begin{center}
            \begin{tcolorbox}[width=1.5in,height=4em,valign=center,colback=white,before=,after=\hfill]
            {
                \huge
                    \phantom{$a + bi$}
            }
            \end{tcolorbox}
        \end{center}
        where $a$ and $b$ are real numbers.
    \end{tcolorbox}
\end{center}

\begin{myConceptSteps}{
    To add or subtract complex numbers\dots
    }
    \myStep{distribute}{
        Distribute any \gap{negatives}.
    }
    \myStep{combine}{
        Combine like terms (\gap{real} parts and \gap{imaginary} parts).
    }
\end{myConceptSteps}

\myBlankExample{1.5in}{
    Add these complex numbers:
    $(1 - 2i) + (-3 + 4i)$.
}

\begin{taggedblock}{on-level}
    \myBlankExample{1.5in}{
        Subtract these complex numbers:
        $(5 + 7i) - (4 - 3i)$.
    }
\end{taggedblock}
\begin{taggedblock}{pre-AP}
    \myBlankExample{1.5in}{
        Subtract these complex numbers:
        $(-5 - 7i) - (-4 - 3i) - 5i$.
    }
\end{taggedblock}
