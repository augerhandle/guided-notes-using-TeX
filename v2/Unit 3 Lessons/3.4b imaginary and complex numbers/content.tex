\begin{myObjectives}
    \myObjective{simplify}{powers of $i$}
    \myObjective{sketch}{the graph of complex numbers in the complex plane}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{complex number}{
        a ``number'' with a {\bfseries\itshape real part} and 
        an {\bfseries\itshape imaginary part}
        (e.g., $2 + 5i$)
    }
    \myDefinition{complex plane}{
        a 2-D plane where the $x$-axis is reals 
        and the $y$-axis is imaginaries.
    }
\end{myVocabulary}
