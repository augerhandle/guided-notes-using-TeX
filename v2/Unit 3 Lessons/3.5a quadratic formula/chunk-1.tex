\section*{The Quadratic Formula}

\begin{center}
\begin{tcolorbox}[width=5.5in]
    Given a quadratic equation in standard form 
    {\LARGE
        \[
            a x^2 + bx + c = 0
        \]
    }
    the solution for $x$ can be calculated from
    {\bfseries\itshape the Quadratic Formula}:
    {\LARGE
        \[
            x = \frac
                {-b \pm \sqrt{b^2 - 4ac}}
                {2a}
        \]
    }
\end{tcolorbox}
\end{center}
