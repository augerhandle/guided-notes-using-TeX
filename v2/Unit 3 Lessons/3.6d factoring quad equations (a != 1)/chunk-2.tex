\section*{Solving quadratic equations by factoring}

\begin{myConceptSteps}{
    To solve a quadratic equation of the form $ax^2 + bx + c = 0$\dots
}
\myStep{rewrite}{
    Rewrite the equation in {\bfseries\itshape standard form}: \underline{\hspace{2in}}
}
\myStep{factor}{Factor the quadratic expression by grouping.}
\myStep{solve}{
    Solve the factored quadratic equation using the
    {\bfseries\itshape Zero Product Property}
}
\end{myConceptSteps}
