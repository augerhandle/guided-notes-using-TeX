\section*{Factoring quadratic expressions by grouping}

\begin{center}
    \begin{tcolorbox}[width=4in]
        Before you begin,
        \begin{center}
            \LARGE\bfseries\itshape
            try to factor out a {GCF}.
        \end{center}
    \end{tcolorbox}
\end{center}

\begin{myConceptSteps}{
    To factor a quadratic expression $ax^2 + bx + c$ by grouping\dots
}
    \myStep{$a$, $b$, $c$, $ac$}{
        Identify the values of $a$, $b$, and $c$
        and \gap{calculate} $ac$.
    }
    \myStep{two numbers}{
        Find two numbers that
        \begin{itemize}[itemsep=0in]
            \item \gap{multiply} together to give {\LARGE $ac$}
            \item \gap{add} together to give {\LARGE $b$}
        \end{itemize}
        Call these two numbers {\LARGE $m$} and {\LARGE $n$}.
        (It does not matter which number is $m$ and which is $n$.)
    }
    \myStep{split $bx$}{
        Split the middle term $bx$ into \gap{$mx + nx$}.
    }
    \myStep{group}{
        Use parentheses to \gap{group} the four terms into \gap{two pairs}.
    }
    \myStep{GCFs}{
        Factor a \gap{GCF} out of each of the two pairs.
        (If there is no GCF, write a 1 instead.)
    }
    \myStep{shared binomial}{
        Factor the \gap{shared binomial} out of the two pairs.
    }
\end{myConceptSteps}

\begin{taggedblock}{on-level}
    \myBlankExample{3in}{
        Factor this quadratic expression:
        \(
           3k^2 + 4k - 15
        \)
    }
\end{taggedblock}
