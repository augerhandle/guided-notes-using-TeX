

\section*{Standard form}

\begin{center}
\begin{tcolorbox}[width=5.5in]
    {
        A quadratic function may be written in 
        {\bfseries\itshape standard form}
        as follows.
        \vskip1em
        {
            \Huge
            \begin{center}
        $y = {\mathbf a}x^2 + {\mathbf b}x + {\mathbf c}$
            \end{center}
        }
        \vskip1.75em
    }
\end{tcolorbox}
\end{center}

\noindent {$\mathbf a$}, {$\mathbf b$}, and {$\mathbf c$} are constant numbers.

\begin{center}
    \begin{tcolorbox}[width=4in]
        This is the {\bfseries\itshape same} constant {$\mathbf a$} that is in vertex form.
    \end{tcolorbox}
\end{center}

\begin{taggedblock}{on-level}
    \myBlankExample{1in}{
        Find the constants $a$, $b$, and $c$ in this quadratic function:
        $y = 3x^2 + 5x - 7$
    }
\end{taggedblock}
