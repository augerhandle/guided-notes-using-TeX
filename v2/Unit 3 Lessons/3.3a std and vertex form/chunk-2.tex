

\section*{Finding the vertex and axis of symmetry}

\begin{myConceptSteps}{
    To find the coordinates of the vertex of a quadratic function in standard form\dots
    }
    \myStep{$\mathbf a$ and $\mathbf b$}{
        Find $\mathbf a$ and $\mathbf b$ 
        from the standard form equation.
    }
    \myStep{$x$}{
        The $x$-coordinate of the vertex is $x = \frac{-b}{2a}$.
    }
    \myStep{$y$}{
        To find the $y$-coordinate of the vertex,
        substitute the $x$-coordinate into the function.
    }
\end{myConceptSteps}

\myBlankExample{1.5in}{
    Find the coordinates of the vertex of this quadratic function:
    $y = 2x^2 - 16x - 42$
}


\begin{myConceptSteps}{
    To find the equation of the axis of symmetry of a quadratic function in standard form\dots
    }
    \myStep{$\mathbf a$ and $\mathbf b$}{
        Find $\mathbf a$ and $\mathbf b$ 
        from the standard form equation.
    }
    \myStep{$x$}{
        The equation of the axis of symmetry is $x = \frac{-b}{2a}$.
        (This is the same as the $x$-coordinate of the vertex.)
    }
\end{myConceptSteps}

\myBlankExample{1.5in}{
    Find the equation of the axis of symmetry of this quadratic function:
    $y = 4x^2 - 16x -3$
}
