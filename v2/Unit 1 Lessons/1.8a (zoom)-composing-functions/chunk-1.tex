Today we are going to learn how to ``join'' two functions together ---
taking the output of the first one and passing that as an input to the second one.
This is called \emph{composing functions}.

\begin{myConceptSteps}{To evaluate a composite function at a number\dots}
    \myStep{input}{Pass the input (a number) to the {\bfseries\itshape nearest function} on the left.}
    \myStep{evaluate}{Evaluate that function. (You will get a number.)}
    \myStep{pass}{Pass that number to the next function to the left.}
    \myStep{evaluate}{Evaluate the that next function. (You will get a number.) This is the output of the composite function.}
\end{myConceptSteps}

\begin{center}
    \begin{tcolorbox}[width=3.5in]
        \(   (f\circ g)(x) = f( g(x))    \)
        \vskip1em
        Pass the input to $g$ first (because $g$ is closest to $x$).
        Then pass the result of $g(x)$ to $f$.
    \end{tcolorbox}
\end{center}

\begin{center}
    \begin{tcolorbox}[width=3.5in]
        \(   (g\circ f)(x) = g( f(x))    \)
        \vskip1em
        Pass the input to $f$ first (because $f$ is closest to $x$).
        Then pass the result of $f(x)$ to $g$.
    \end{tcolorbox}
\end{center}

    Evaluate the composite functions

    \begin{align*}
        f(x) &= x + 2 \\
        g(x) &= 3x - 4
    \end{align*}
}
