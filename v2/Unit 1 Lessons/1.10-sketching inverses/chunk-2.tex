\section*{Sketching the graph of an inverse function}

\begin{myConceptSteps}{To sketch the inverse of $f(x)$ given its equation\dots}
    \myStep{inverse}{Find $f^{-1}(x)$ using what you learned in Lesson 1.9.}
    \myStep{sketch}{
        Sketch the graph of that inverse function 
        (by hand if you know how  
        or using a graphing calculator or Desmos if you don't).
    }
\end{myConceptSteps}

\begin{taggedblock}{on-level}
\myBlankExample{3in}{
    Sketch the graph of the inverse of
    \( f(x) = 3x \)
}
\end{taggedblock}

\myBlankExample{3in}{
    Sketch the graph of the inverse of
    \( v(t) = 2t-5 \)
}

\begin{taggedblock}{pre-AP}
    \myBlankExample{3in}{
    Sketch the graph of the inverse of
    \( f(x) = \frac{7}{2}x- \frac{9}{2} \)
}
\end{taggedblock}
