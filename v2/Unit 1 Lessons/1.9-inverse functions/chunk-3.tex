\section*{Finding the inverse of continuous functions}

\begin{myConceptSteps}{To find the inverse of a function specified by a rule, $f(x)=$\dots}
    \myStep{rewrite}{Replace $f(x)$ with $y$ so that you have $y=\cdots$.}
    \myStep{swap}{Replace the original $y$ by $x$. Replace all the original $x$'s by $y$. (This is ``swapping'' $x$ and $y$.)}
    \myStep{solve}{Solve the new equation for $y$.}
    \myStep{inverse notation}{Replace the $y$ by $f^{-1}(x)$.}
\end{myConceptSteps}

\myBlankExample{4in}{
    Find $f^{-1}$, the inverse of the function,
    \( f(x) = x-8 \).
}
