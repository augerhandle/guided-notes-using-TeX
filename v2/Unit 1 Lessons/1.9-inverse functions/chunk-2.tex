\section*{Finding the inverse of discrete functions}

\begin{myConceptSteps}{To find the inverse of a relation given as a bunch of points\dots}
    \myStep{count}{Count the number of ordered pairs in the original relation.}
    \myStep{new set}{Create an ``empty'' set of the same number of ordered pairs.}
    \myStep{swap}{Fill in the empty ordered pairs by reversing (swapping) the $x$ and $y$ values in each of the original pairs.}
\end{myConceptSteps}


\myBlankExample{0.75in}{
    Find the inverse of this discrete relation:
    \[ 
        \{
            (1,10), (2,8), (5,-3), (5,7)
        \}
    \]
}

\myBlankExample{0.75in}{
    Find the inverse of this discrete relation:
    \[ 
        \{
            (2,10), (2,8), (2,0), (2,-1), (2,-4)
        \}
    \]
}
