\begin{myObjectives}
    \myObjective{find}{the domain and range of a function}
    \myObjective{find}{$x$- and $y$-intercepts on a graph}
    \myObjective{find}{the maximum and minimum points on a graph}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{domain}{
        the set of all the inputs in a relation 
        (usually $x$'s)
    }
    \myDefinition{range}{
        the set of all the outputs in a relation 
        (usually $y$'s)
    }
    \myDefinition{discrete relation}{
        a relation that is a bunch of separate points
    }
    \myDefinition{continuous relation}{
        a relation that can be graphed as a smooth line or curve
    }
    \myDefinition{intercept}{
        where a curve crosses the $x$- or $y$-axis
    }
    \myDefinition{minimum}{
        the smallest $y$ value of a relation (the $y$ coordinate of the lowest point)
    }
    \myDefinition{maximum}{
        the largest $y$ value of a relation (the $y$ coordinate of the highest point)
    }
\end{myVocabulary}
