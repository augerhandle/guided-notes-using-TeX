\section*{Determining if two functions are inverses of each other}

\begin{myConceptSteps}{To determine if $f(x)$ and $g(x)$ are inverses of each other\dots}
    \myStep{$f \circ g$}{Verify that ($f \circ g)(x) \equiv f(g(x)) = x$.}
    \myStep{$g \circ f$}{Verify that ($g \circ f)(x) \equiv g(f(x)) = x$.}
\end{myConceptSteps}

They must \emph{both} equal $x$ in order to be inverses of each other.
If either one does not equal $x$, then they are not inverses.


\begin{taggedblock}{on-level}
    \myBlankExample{2in}{
        Determine if $f$ and $g$ are inverse functions, where
        \( f(x) = 5x \)
        and
        \(     g(x) = \frac{1}{5}x \) 
    }

    \myBlankExample{2in}{
        Determine if $f$ and $g$ are inverse functions, where
        \( f(x) = x+3 \)
        and
        \(     g(x) = \frac{1}{3}x \) 
    }
\end{taggedblock}
