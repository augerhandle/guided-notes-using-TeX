\myLesson{Graphing and Transforming Absolute Value Functions}[2][c]

\begin{myObjectives}
    \myObjective{plot}{the 3 DOTS associated with the absolute value parent function}
    \myObjective{write}{the $a$/$h$/$k$ form of a transformed absolute value function}
    \myObjective{explain}{the effects of $a$/$h$/$k$ on the parent function}
    \myObjective{sketch}{the graph of a transformed absolute value function given $a$/$h$/$k$ using the 3 DOTS}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{3 DOTS}{Three special points that we will use to help us transform parent functions all year.}
    \myDefinition{$a$/$h$/$k$}{The 3 transformation parameters that we will use all year.}
    \myDefinition{$a$}{The parameter the leads to reflections and vertical stretches/compressions.}
    \myDefinition{$h$}{The parameter the leads horizontal shifts.}
    \myDefinition{$k$}{The parameter the leads vertical shifts.}
\end{myVocabulary}
