

\section*{Absolute value expressions of numbers}

\begin{myConceptSteps}{To evaluate an absolute value expression of numbers\dots}
    \myStep{evaluate}{Evaluate what is inside the vertical bars. This will give you a number.}
    \myStep{positive or negative}{Determine whether the number inside the vertical bars is positive or negative.}
    \myStep{result}{If the number is positive or zero, that is the result. If the number is negative, the result is the opposite (positive) value.}
\end{myConceptSteps}

    Evaluate $|100-4|$
