

\section*{Absolute value expressions of numbers}

\begin{myConceptSteps}{To evaluate an absolute value expression of numbers\dots}
    \myStep{simplify}{
        Simplify what is \gap{inside} the vertical bars using the order of operations (GEMDAS). 
        This will give you the absolute value of a number.
    }
    \myStep{evaluate}{
        Evaluate the \gap{absolute value} of that number.
    }
\end{myConceptSteps}

Remember that the result will always be \gap{positive (or zero)}.

\myBlankExample{0.75in}{
    Evaluate $|10+3-9|$
}

\myBlankExample{0.75in}{
    Evaluate $|1+9-20|$
}

\begin{taggedblock}{on-level}
    \myBlankExample{0.75in}{
        Evaluate $|3(5-10) + 5|$
    }
\end{taggedblock}
\begin{taggedblock}{pre-AP}
    \myBlankExample{0.75in}{
        Evaluate $\left|\frac{2}{3}(75-15) - 4\cdot25/2\right|$
    }
\end{taggedblock}
