

\section*{Simple absolute values}

\begin{myConceptSteps}{To evaluate an absolute value of a number\dots}
    \myStep{positive}{%
        If the number is positive or zero,
        then the absolute value is that number.
    }
    \myStep{negative}{%
        If the number is negative, 
        the absolute value is the number without the minus sign.
    }
\end{myConceptSteps}

\begin{center}
    \begin{tcolorbox}[width=4in]
        The absolute value is {\bfseries\itshape always} positive (or zero).
        It is {\bfseries\itshape never} negative.
    \end{tcolorbox}
\end{center}

\myBlankExample{0.5in}{
    Evaluate $|23|$
}

\myBlankExample{0.5in}{

}
