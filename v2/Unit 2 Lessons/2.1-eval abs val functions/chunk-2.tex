

\section*{Simple absolute values}

\begin{myConceptSteps}{To evaluate an absolute value of a number\dots}
    \myStep{positive or negative}{Determine whether the number inside the vertical bars is positive or negative.}
    \myStep{result}{If the number is positive or zero, that is the result. If the number is negative, the result is the opposite (positive) value.}
\end{myConceptSteps}

\begin{center}
    \begin{tcolorbox}[width=4in]
        Remember that the absolute value is {\bfseries\itshape always} positive (or zero).
        It is {\bfseries\itshape never} negative.
    \end{tcolorbox}
\end{center}

\myBlankExample{0.5in}{
    Evaluate $|23|$
}

\myBlankExample{0.5in}{
    Evaluate $|-8|$
}

\myBlankExample{0.5in}{
