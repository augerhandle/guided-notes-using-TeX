

\section*{Absolute value of algebraic expressions given the value of all the variables}

\begin{myConceptSteps}{To evaluate an absolute value of an algebraic expression given the value of all the variables\dots}
    \myStep{substitute}{Substitute the variable values into the expression.}
    \myStep{evaluate}{Evaluate what is inside the vertical bars. This will give you a number.}
    \myStep{positive or negative}{Determine whether the number inside the vertical bars is positive or negative.}
    \myStep{result}{If the number is positive or zero, that is the result. If the number is negative, the result is the opposite (positive) value.}
\end{myConceptSteps}

\myBlankExample{1.25in}{
    Evaluate $|x+5|$ given that $x=11$.
}

\myBlankExample{1.25in}{
    Evaluate $|2x-38|$ given that $x=4$.
}

\myBlankExample{1.25in}{
    Evaluate $|3x - 48|$ given that $x=16$.
}
