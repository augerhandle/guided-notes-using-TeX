

\section*{Absolute value of algebraic expressions given the value of all the variables}

\begin{myConceptSteps}{To evaluate an absolute value of an algebraic expression given the value of all the variables\dots}
    \myStep{substitute}{
        Substitute the variable values into the expression.
    }
    \myStep{simplify}{
        Simplify what is inside the vertical bars using the order of operations (GEMDAS). 
        This will give you the absolute value of a number.
    }
    \myStep{evaluate}{
        Evaluate the absolute value of that number.
        Remember that the result will always be positive (or zero).
    }
\end{myConceptSteps}

\myBlankExample{1.25in}{
    Evaluate $|x+5|$ given that $x=-11$.
}

\begin{taggedblock}{on-level}
    \myBlankExample{1.25in}{
