\subsection*{Reflections Across the $x$-axis}

\begin{center}
    \begin{tcolorbox}[width=4in]
        A {\bfseries\itshape reflection across the $x$-axis} 
        moves all the points on the graph of the parent function
        to the {\bfseries\itshape other side of the $x$-axis}.
    \end{tcolorbox}
\end{center}

\myBlankExample{1in}{Explain what a ``reflection across the $x$-axis'' is.}

\myBlankExample{3in}{
    Sketch the following:
    \begin{itemize}[itemsep=0in]
        \item the graph of the absolute value parent function
        \item the graph after the parent has been reflected across the $x$-axis
    \end{itemize}
    Your answer should have two graphs on it.
}
