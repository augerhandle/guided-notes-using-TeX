    \subsection*{Combined transformations}
    
    \myBlankExample{1.75in}{
        Using a graphing calculator or app,
        sketch the graphs of
            \begin{itemize}[itemsep=0in]\Large
                \item $y = |x-2| + 4$
                \item $y = -|x+3| - 2$
                \item $y = 3|x-4|$
                \item $y = -3|x+4|-1$
            \end{itemize}
        What kind of transformations are these?
    }    

    \begin{center}
        \begin{tcolorbox}[width=4.2in]
            The general form of the transformed absolute value function is
            \LARGE
            \[
                f(x) = a | x - h | + k
            \]
        \end{tcolorbox}
    \end{center}
