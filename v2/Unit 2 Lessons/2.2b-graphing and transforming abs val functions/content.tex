\myLesson{Graphing and Transforming Absolute Value Functions}[2][b]

\begin{myObjectives}
    \myObjective{list}{the four transformations}
    \myObjective{explain}{the meaning of each of the transformations}
    \myObjective{sketch}{the result of the four transformations applied to the graph of the absolute value parent function}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{horizontal shift}{a transformation where points move {\bfseries\itshape left or right}}
    \myDefinition{reflection}{a transformation where points are reflected to the other side of a line}
    \myDefinition{vertical stretch}{a transformation where $y$-coordinates {\bfseries\itshape increase}}
    \myDefinition{vertical compression}{a transformation where $y$-coordinates {\bfseries\itshape decrease}}
    \myDefinition{vertical shift}{a transformation where points move {\bfseries\itshape up or down}}
\end{myVocabulary}
