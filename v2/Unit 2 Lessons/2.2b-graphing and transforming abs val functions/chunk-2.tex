\subsection*{Vertical Shifts}

\begin{center}
    \begin{tcolorbox}[width=4in]
        A {\bfseries\itshape vertical shift} 
        moves all the points on the graph of the parent function
        to the {\bfseries\itshape up or down}.
    \end{tcolorbox}
\end{center}
\myBlankExample{2in}{
    Sketch the following:
    \begin{itemize}[itemsep=0in]
        \item the graph of the absolute value parent function
        \item the graph after the parent has been vertically shifted up by 2
        \item the graph after the parent has been vertically shifted down by 1
    \end{itemize}
    Your answer should have three graphs on it.
}



\begin{taggedblock}{on-level}
    \myBlankExample{2in}{
        Using a graphing calculator or app,
        sketch the graphs of
            \begin{itemize}[itemsep=0in]
                \item $y = |x| - 3$
                \item $y = |x| + 2$
            \end{itemize}
        What kind of transformations are these?
    }    
\end{taggedblock}
