\section*{Solving compound inequalities connected by ``or''}

\myBlankExample{2in}{
    Solve the following compound inequality  and sketch its graph on a number line.
    \[
        x+5 \geq 10
        \qquad\text{\itshape OR}\qquad
        x-1 < -1
    \]
}
