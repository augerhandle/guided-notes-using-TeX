



\section*{Solving compound inequalities connected by ``and''}
\myBlankExample{3in}{
    Solve the following compound inequality and sketch its graph on a number line.
    \[
        10 \geq x+ 6 
        \qquad\text{\itshape AND}\qquad
        x+ 6 > 9
    \]
    (This inequality can also be written as a single three-part inequality:
    \(
        10 \geq x+ 6 > 9
    \))
}

\myBlankExample{1in}{
    Rewrite this compound inequality connected by ``and'' as a three-part inequality:
    \[
        8 < 2x + 1
        \qquad\text{\itshape AND}\qquad
        2x + 1 \leq 12
    \]
}

\myBlankExample{1in}{
    Rewrite this three-part compound inequality as two inequalities connected by ``and'':
    \[  -5 < 2x + 1 \leq 10 \]
}
