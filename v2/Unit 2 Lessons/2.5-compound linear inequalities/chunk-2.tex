



\section*{Compound inequalities connected by ``and''}

\begin{center}
    \begin{tcolorbox}[width=4in]
        The graph of a compound inequality that is connected with ``and''
        is the {\bfseries\itshape intersection} (overlap) of the graphs of the two number line.
    \end{tcolorbox}
\end{center}

\myBlankExample{3in}{
    Solve the following compound inequality:
    \[
        10 \geq x+ 6 
        \qquad\text{and}\qquad
        x+ 6 > 9
    \]
    (This inequality can also be written as a single three-part inequality:
    \(
        10 \geq x+ 6 > 9
    \))
}

\myBlankExample{1in}{
    Rewrite this compound inequality connected by ``and'' as a three-part inequality:
    \[
        8 < 2x + 1
        \qquad\text{and}\qquad
        2x + 1 \leq 12
    \]
}

\myBlankExample{1in}{
    Rewrite this three-part compound inequality as two inequalities connected by ``and'':
    \[  -5 < 2x + 1 \leq 10 \]
}
