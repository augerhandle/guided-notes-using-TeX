
\section*{Solving systems of equations by graphing}


\begin{myConceptSteps}{
    To solve a system of equations with two variables by graphing\dots
}
    \myStep{graph}{
        Sketch the graph of each \gap{equation}.
    }
    \myStep{intersections}{
        Highlight where the graphs \gap{intersect}.
    }
    \myStep{coordinates}{
        The solution is the coordinates of all the \gap{points}
        where the graphs intersect.
    }
\end{myConceptSteps}


\begin{taggedblock}{on-level}
    \myBlankExample{2.5in}{
        Which of the following points are solutions of this system of equations:
        \begin{equation*}
            \begin{cases}
                y &= -x + 4 \\
                y &=  x + 2 \\
            \end{cases}
        \end{equation*}
        \begin{minipage}{0.48\textwidth}
            \begin{itemize}
                \item[$\circ$] $(-2,0)$
                \item[$\circ$] $(0,2)$
            \end{itemize}    
        \end{minipage}
        \begin{minipage}{0.48\textwidth}
            \begin{itemize}
                \item[$\circ$] $(0,4)$
                \item[$\circ$] $(1,3)$
            \end{itemize}    
        \end{minipage}
    }
    \myBlankExample{2.5in}{
        Solve this system by graphing.
        \begin{equation*}
            \begin{cases}
                y &= -3x + 10 \\
                y &=  6x + 1 \\
            \end{cases}
        \end{equation*}
        How many solutions are there?
    }
    \myBlankExample{3in}{
        Solve this system by graphing.
        \begin{equation*}
            \begin{cases}
                y &= x^2 + 1 \\
                y &=  x+3 \\
            \end{cases}
        \end{equation*}
        How many solutions are there?
    }
\end{taggedblock}
