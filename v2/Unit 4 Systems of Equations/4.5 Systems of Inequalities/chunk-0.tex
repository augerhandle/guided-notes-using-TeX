\noindent
Remember that to sketch an inequality (in two variables), we 
\begin{itemize}[itemsep=0in]
    \item drew the boundary as a \gap{dashed} or \gap{solid}
    \item shaded \gap{above} or \gap{below} the curve.
\end{itemize}.

\begin{center}
    \fbox{\Large
    \begin{tabular}{l|c|l}
        {\bfseries\itshape dashed} & $>$, $<$       & no equals \\
        {\bfseries\itshape solid}  & $\geq$, $\leq$ & equals included \\
    \end{tabular}
    }
\end{center}
\begin{center}
    \fbox{\Large
    \begin{tabular}{l|c|l}
        {\bfseries\itshape above} & $>$, $\geq$ &greater... \\
        {\bfseries\itshape below} & $<$, $\leq$ &less... \\
    \end{tabular}
    }
\end{center}
