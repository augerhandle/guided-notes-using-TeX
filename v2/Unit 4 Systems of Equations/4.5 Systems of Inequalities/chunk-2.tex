\begin{myConceptSteps}{
    To determine if a point is a solution to a system of inequalities\dots
}
    \myStep{sketch}{
        Sketch the system of inequalities.
    }
    \myStep{plot}{
        Plot the point.
        The point is a solution if
        \begin{itemize}
            \item is \gap{in} the shaded region 
            \item is \gap{not on} a dashed line
        \end{itemize}
    }
\end{myConceptSteps}

\myBlankExample{3in}{
    Determine if the following are solutions of this linear inequality:
    \begin{align*}
        y &> x-4\\
        y &<= -2x+2
    \end{align*}
    {\Large
    $(0,0)$\hfill
    $(0,5)$\hfill
    $(5,0)$\hfill
    $(0,-5)$\hfill
    $(-5,0)$  
    }
}
