\begin{myConceptSteps}{
    To sketch a system of inequalities\dots
}
    \myStep{sketches}{
        Sketch each inequality. 
        (Use the {\scshape Desmos graphing calculator} to help you.)
    }
    \myStep{intersection}{
        The solution is where the shaded regions \gap{overlap}.
    }
\end{myConceptSteps}

\myBlankExample{3in}{
    Sketch the solution of this linear inequality:
    {\Large
    \begin{align*}
        y &> x-4\\
        y &<= -2x+2
    \end{align*}
    }
}
