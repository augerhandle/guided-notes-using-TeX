
\section*{Solving systems of linear equations by elimination}


\begin{myConceptSteps}{
    To solve a system of linear equations by elimination\dots
}
    \myStep{rewrite}{
        Rewrite the equations so that the variables \gap{line up}.
    }
    \myStep{multiply}{
        \gap{Multiply} one or both of the equations by a number
        so that the equations have \gap{matching} coefficients for one of the variables.
    }
    \myStep{eliminate}{
        \gap{Add} or \gap{subtract} the equations to eliminate
        the variable with matching coefficients.
    }
    \myStep{solve}{
        \gap{Solve} that equation for the second variable.
    }
    \myStep{back-substitute}{
        \gap{Substitute} that result into one of the original equations,
        and solve for the other variable.
    }
\end{myConceptSteps}
