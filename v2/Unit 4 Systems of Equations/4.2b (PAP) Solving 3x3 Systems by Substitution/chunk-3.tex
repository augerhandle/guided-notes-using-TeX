\section*{Examples of systems with {\bfseries\itshape no solutions} and {\bfseries\itshape infinite solutions}}

\begin{myCenteredBox}{width=5in}
    What do solutions look like when there are\vskip1em
    \begin{itemize}[itemsep=0.5em]
        \item \gap{no} solutions?
        \item \gap{infinite} solutions?
    \end{itemize}
\end{myCenteredBox}



\myBlankExample{2.1in}{
    Solve the following system by {\bfseries\itshape substitution}.
    \begin{equation*}
        \begin{cases}
            &y = 2x -5\\
            &y = 2x + 4\\
        \end{cases}
    \end{equation*}
}
\myBlankExample{2.1in}{
    Solve the following system by {\bfseries\itshape substitution}.
    \begin{equation*}
        \begin{cases}
            &y = 2x -5\\
            &5y = 10x -25\\
        \end{cases}
    \end{equation*}
}
