\section*{Solving systems of linear/quadratic equations by substitution}

\begin{myConceptSteps}{
    To solve a system of linear/quadratic equations by substitution\dots
}
    \myStep{solve}{
        Solve the \gap{linear} for one of the variables.
        The result will often be an \gap{expression}.
    }
    \myStep{substitute}{
        Substitute the result into the \gap{quadratic} equation, and
        solve for the other variable.
        The result will usually be two \gap{numbers}.
    }
    \myStep{back-substitute}{
        Substitute those numbers back into the expression for the first variable,
        and solve for the first variable.
        The result will be one or two \gap{numbers}.
    }
    \myStep{solution}{
        Write the solution in $(x,y)$ form as \gap{points}.
    }
\end{myConceptSteps}


\myBlankExample{2.75in}{
    Solve the following system by {\bfseries\itshape substitution}.
    \begin{equation*}
        \begin{cases}
            &y = x^2+4x+3\\
            &-2x + y =6\\
        \end{cases}
    \end{equation*}
}
