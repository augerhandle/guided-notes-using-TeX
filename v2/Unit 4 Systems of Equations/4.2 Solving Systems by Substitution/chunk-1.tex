
\section*{Solving systems of linear equations by substitution}


\begin{myConceptSteps}{
    To solve a system of linear equations by substitution\dots
}
    \myStep{solve}{
        Solve one the equations for one of the variables.
        The result will often be an \gap{expression}.
    }
    \myStep{substitute}{
        Substitute the result into the other equation, and
        solve for the second variable.
        The result will be a \gap{number}.
    }
    \myStep{back-substitute}{
        Substitute that number back into the expression for the first variable,
        and solve for the first variable.
        The result will be a \gap{number}.
    }
    \myStep{solution}{
        Write the solution in $(x,y)$ form as a \gap{point}.
    }
\end{myConceptSteps}


\myBlankExample{2.75in}{
    Solve the following system by {\bfseries\itshape substitution}.
    \begin{equation*}
        \begin{cases}
            &y = -3x + 7\\
            &y = 2x - 3\\
        \end{cases}
    \end{equation*}
}
