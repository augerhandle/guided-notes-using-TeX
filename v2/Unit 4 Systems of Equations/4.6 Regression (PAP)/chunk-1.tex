
\section*{Two equations, two unknowns}

You know how to solve two equations and two unknowns.

\begin{myConceptSteps}{
    To find the equation of a line going through two points\dots
}
    \myStep{variables}{
        Let the equation of the line be $y=mx+b$.
        Then write down $m$ and $b$ as the unknown variables.
    }
    \myStep{equations}{Write a $2\times2$ system of equations.
        \begin{itemize}
            \item Evaluate $y=mx+b$ using $x$ and $y$ from each of the two points.
            This gives you \gap{two} equations.
            \item \gap{Reorder} the equations so the variables \gap{line up} 
            and the constant numbers are on the \gap{right}.
        \end{itemize}
    }
    \myStep{matrices}{
        Fill in \gap{$A$} and \gap{$B$}.
    }
    \myStep{solve}{
        Use the Desmos matrix \gap{calculator} 
        to calculate the solution as {\boldmath $A^{-1}B$}.
    }
\end{myConceptSteps}
