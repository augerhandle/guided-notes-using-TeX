

\begin{myConceptSteps}{
    To solve a system of linear equations with matrices\dots
}
    \myStep{variables}{
        Determine what the unknown variables in the word problem are,
        and write them down.
    }
    \myStep{equations}{Write the system of equations.
        \begin{itemize}
            \item Figure out the \gap{equations} that come from the word problem.
            \item \gap{Reorder} them so the variables \gap{line up} 
            and the constant numbers are on the \gap{right}.
        \end{itemize}
    }
    \myStep{matrices}{
        Fill in \gap{$A$} and \gap{$B$}.
    }
    \myStep{solve}{
        Use the Desmos matrix \gap{calculator} 
        to calculate the solution as {\boldmath $A^{-1}B$}.
    }
\end{myConceptSteps}




\begin{myExample}{
    The convenience store is selling apples for \$0.50 each
    and oranges for \$0.65 each. 
    Guadalupe buys 9 items of fruit (apples and oranges) and pays \$5.10.
    How many apples and how many oranges did Guadalupe buy?
}
    \myWideMatrixTable
    \vspace{1em}
    Guadalupe bought \gap{five} apples and \gap{four} oranges.
\end{myExample}
