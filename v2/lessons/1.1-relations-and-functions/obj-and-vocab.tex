\begin{myObjectives}
    \myObjective{find}{the domain and range of a relation}
    \myObjective{determine}{if a relation is a function}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{relation}{
        a set of relationships between inputs and outputs
        (usually $x$'s and $y$'s)
    }
    \myDefinition{function}{
        a special kind of relation
        (only one output for each input)
    }
    \myDefinition{domain}{
        the set of all the inputs in a relation 
        (usually the $x$'s)
    }
    \myDefinition{range}{
        the set of all the outputs in a relation 
        (usually the $y$'s)
}
    \myDefinition{vertical}{
        up and down
    }
    \myDefinition{ordered pairs}{
        pairs of $x$'s and $y$'s For example, $(-2,13)$ is an ordered pair.
    }
\end{myVocabulary}
