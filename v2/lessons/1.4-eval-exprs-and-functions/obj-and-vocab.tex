\begin{myObjectives}
    \myObjective{simplify}{expressions using the \emph{order of operations}}
    \myObjective{evaluate}{expressions given the value of all the variables}
    \myObjective{evaluate}{a function given the value of its input variable}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{operation}{some kind of mathematical calculation ($+$, $-$, multiply, etc...)}
    \myDefinition{expression}{numbers and variables combined using various operations}
    \myDefinition{order of operations}{which operations should be done before the others}
    \myDefinition{evaluate}{apply \emph{order of operations} to an expression (to get a number)}
    \myDefinition{simplify}{apply \emph{order of operations} to an expression (to get a simpler expression)}
\end{myVocabulary}
