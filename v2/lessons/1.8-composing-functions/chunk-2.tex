\section*{Evaluating composite functions using numbers as inputs}

\begin{myConceptSteps}{To evaluate a composite function at a number\dots}
    \myStep{input}{Pass the input (a number) to the {\bfseries\itshape nearest function} on the left.}
    \myStep{evaluate}{Evaluate that function. (You will get a number.)}
    \myStep{pass}{Pass that number to the next function to the left.}
    \myStep{evaluate}{Evaluate the that next function. (You will get a number.) This is the output of the composite function.}
\end{myConceptSteps}

\begin{center}
    \begin{tcolorbox}[width=4in]
        When the functions are $f$ and $g$ and the input is $2$,
        \begin{align*}
            (f \circ g)(2) &= f(g(2)) \qquad\dots\text{\itshape evaluate $g(2)$ first and then $f$.}\\
            (g \circ f)(2) &= g(f(2)) \qquad\dots\text{\itshape evaluate $f(2)$ first and then $g$.}
        \end{align*}
    \end{tcolorbox}
\end{center}



\myBlankExample{3in}{
    Evaluate the composite functions 
    $(f \circ g)(1)$ and $(g \circ f)(1)$ where
    \begin{align*}
        f(x) &= x + 2 \\
        g(x) &= 3x - 4
    \end{align*}
}
\myBlankExample{3in}{
    Evaluate the composite functions 
    $(f \circ g)(-2)$ and $(g \circ f)(-2)$ where
    \begin{align*}
        f(x) &= 2x + 1 \\
        g(x) &= 4x - 3
    \end{align*}
}
