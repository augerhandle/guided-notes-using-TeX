\section*{Composing two functions}

What would you do with
$(f \circ g)(x)$
where the input is a {\bfseries\itshape variable} instead of a number!?
It turns out that the ideas are similar to what you just learned.


\begin{myConceptSteps}{To compose two functions\dots}
    \myStep{input}{Pass the input variable to the {\bfseries\itshape nearest function} on the left.}
    \myStep{evaluate}{Evaluate that function. (You will get an expression --- not a number.)}
    \myStep{pass}{Pass that expression to the next function to the left. Use parentheses!}
    \myStep{evaluate}{Evaluate the that next function. (You will get another expression.) This is the output of the composite function.}
\end{myConceptSteps}


\begin{center}
    \begin{tcolorbox}[width=5in]
        When the functions are $f$ and $g$ and the input is $x$,
        \begin{align*}
            (f \circ g)(x) &= f(g(x)) \qquad\dots\text{\itshape evaluate $g(x)$ first and then $f$.}\\
            (g \circ f)(x) &= g(f(x)) \qquad\dots\text{\itshape evaluate $f(x)$ first and then $g$.}
        \end{align*}
    \end{tcolorbox}
\end{center}



\begin{taggedblock}{on-level}
    \myBlankExample{3in}{
        Evaluate the composite functions 
        $(f \circ g)(x)$ and $(g \circ f)(x)$ where
        \begin{align*}
            f(x) &= x + 2 \\
            g(x) &= 3x - 4
        \end{align*}
    }
\end{taggedblock}

\myBlankExample{2.5in}{
    Evaluate the composite functions 
    $(f \circ g)(x)$ and $(g \circ f)(x)$ where
    \begin{align*}
        f(x) &= 2x + 1 \\
        g(x) &= 4x - 3
    \end{align*}
}

\begin{taggedblock}{pre-AP}
\myBlankExample{3.5in}{
    Evaluate $(v \circ w)(x)$ and $(w \circ v)(x)$ where
    \begin{align*}
        v(t) &= t + 1 \\
        w(t) &= t^2 + 2t + 3
    \end{align*}    
}

    You can compose a function with itself.\nopagebreak
    \myBlankExample{2.5in}{
        Evaluate $(w \circ w)(t)$ where
        \begin{align*}
            w(z) &= 5z - 8
        \end{align*}    
    }

    You can also pass an expression as the input.

    \myBlankExample{3.5in}{
        Evaluate $(f \circ g)(2x-1)$ and $(f \circ g)(2x-1)$ where
        \begin{align*}
            f(t) &= t + 1 \\
            g(t) &= 3 + 2t
        \end{align*}    
    }


\end{taggedblock}
