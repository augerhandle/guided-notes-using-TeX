\begin{myObjectives}
    \myObjective{find}{the domain and range of a continuous function}
\end{myObjectives}

\begin{myVocabulary}
    \myDefinition{domain}{
        the set of all the inputs in a function 
        (usually $x$'s)
    }
    \myDefinition{range}{
        the set of all the outputs in a function 
        (usually $y$'s)
    }
    \myDefinition{continuous function}{
        a function that can be graphed as a smooth line or curve
    }
\end{myVocabulary}
