\begin{myConceptSteps}{To solve multi-step equations\dots}
    \myStep{distribute}{Distribute terms wherever you can.}
    \myStep{simplify}{Simplify the equation by combining like terms.}
    \myStep{move}{
        Move all terms with the variable to one side of the equation
        and all numbers to the other side, 
        using addition and subtraction as inverse operations.
    }
    \myStep{simplify}{Combine like terms again, if necessary.}
    \myStep{solve}{
        Solve the one-step equation,
        using multiplication and division as inverse operations.
    }
\end{myConceptSteps}
